187 void
188 TimerConfigure(uint32_t ui32Base, uint32_t ui32Config)
189 {
190  //
191  // Check the arguments.
192  //
193  ASSERT(TimerBaseValid(ui32Base));
194  ASSERT((ui32Config == GPTIMER_CFG_ONE_SHOT) ||
195  (ui32Config == GPTIMER_CFG_ONE_SHOT_UP) ||
196  (ui32Config == GPTIMER_CFG_PERIODIC) ||
197  (ui32Config == GPTIMER_CFG_PERIODIC_UP) ||
198  ((ui32Config & 0xff000000) == GPTIMER_CFG_SPLIT_PAIR));
199  ASSERT(((ui32Config & 0xff000000) != GPTIMER_CFG_SPLIT_PAIR) ||
200  ((((ui32Config & 0x000000ff) == GPTIMER_CFG_A_ONE_SHOT) ||
201  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_ONE_SHOT_UP) ||
202  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_PERIODIC) ||
203  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_PERIODIC_UP) ||
204  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_CAP_COUNT) ||
205  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_CAP_TIME) ||
206  ((ui32Config & 0x000000ff) == GPTIMER_CFG_A_PWM)) &&
207  (((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_ONE_SHOT) ||
208  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_ONE_SHOT_UP) ||
209  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_PERIODIC) ||
210  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_PERIODIC_UP) ||
211  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_CAP_COUNT) ||
212  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_CAP_COUNT_UP) ||
213  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_CAP_TIME) ||
214  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_CAP_TIME_UP) ||
215  ((ui32Config & 0x0000ff00) == GPTIMER_CFG_B_PWM))));
216 
217  //
218  // Disable the timers.
219  //
220  HWREG(ui32Base + GPTIMER_O_CTL) &= ~(GPTIMER_CTL_TAEN | GPTIMER_CTL_TBEN);
221 
222  //
223  // Set the global timer configuration.
224  //
225  HWREG(ui32Base + GPTIMER_O_CFG) = ui32Config >> 24;
226 
227  //
228  // Set the configuration of the A and B timers. Note that the B timer
229  // configuration is ignored by the hardware in 32-bit modes.
230  //
231  HWREG(ui32Base + GPTIMER_O_TAMR) = (ui32Config & 255) | GPTIMER_TAMR_TAPWMIE;
232  HWREG(ui32Base + GPTIMER_O_TBMR) =
233  ((ui32Config >> 8) & 255) | GPTIMER_TBMR_TBPWMIE;
234 }
