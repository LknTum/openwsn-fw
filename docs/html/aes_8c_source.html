90 uint8_t AESLoadKey(uint8_t *pui8Key , uint8_t ui8KeyLocation)
91 {
92  static uint32_t ui32temp[4];
93  uint8_t * pui8temp = (uint8_t *)ui32temp;
94  uint8_t i;
95 
96  g_ui8CurrentAESOp = AES_KEYL0AD;
97  // The key address needs to be 4 byte aligned
98  for(i = 0; i < KEY_BLENGTH; i++)
99  {
100  pui8temp[i] = pui8Key[i];
101  }
102  IntDisable(INT_AES);
103 
104  // workaround for AES registers not retained after PM2
105  HWREG(AES_CTRL_INT_CFG) |= AES_CTRL_INT_CFG_LEVEL;
106  HWREG(AES_CTRL_INT_EN) |= (AES_CTRL_INT_EN_DMA_IN_DONE |
107  AES_CTRL_INT_EN_RESULT_AV);
108 
109  // configure master control module
110  HWREG(AES_CTRL_ALG_SEL) &= (~AES_CTRL_ALG_SEL_KEYSTORE);
111  HWREG(AES_CTRL_ALG_SEL) |= AES_CTRL_ALG_SEL_KEYSTORE;
112 
113  // clear any outstanding events
114  HWREG(AES_CTRL_INT_CLR) |= (AES_CTRL_INT_CLR_DMA_IN_DONE |
115  AES_CTRL_INT_CLR_RESULT_AV);
116 
117  //configure key store module (area, size)
118  HWREG(AES_KEY_STORE_SIZE) &= KEY_STORE_SIZE_BITS;
119 
120  // 128-bit key size
121  HWREG(AES_KEY_STORE_SIZE) |= KEY_STORE_SIZE_128;
122 
123  // enable keys to write (e.g. Key 0)
124  HWREG(AES_KEY_STORE_WRITE_AREA) = (0x00000001 << ui8KeyLocation);
125 
126 
127  // configure DMAC
128  // enable DMA channel 0
129  HWREG(AES_DMAC_CH0_CTRL) |= 0x000000001;
130 
131  // base address of the key in ext. memory
132  HWREG(AES_DMAC_CH0_EXTADDR) = (uint32_t)pui8temp;
133 
134  // total key length in bytes (e.g. 16 for 1 x 128-bit key)
135  HWREG(AES_DMAC_CH0_DMALENGTH) = 0x10;
136 
137  // wait for operation completed
138  do
139  {
140  ASM_NOP;
141  }
142  while((!(HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_RESULT_AV)));
143 
144  // check for absence of errors in DMA and key store
145  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_DMA_BUS_ERR))
146  {
147  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_DMA_BUS_ERR;
148  return (AES_DMA_BUS_ERROR);
149  }
150  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_KEY_ST_WR_ERR))
151  {
152  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_KEY_ST_WR_ERR;
153  return (AES_KEYSTORE_WRITE_ERROR);
154  }
155 
156  // acknowledge the interrupt
157  HWREG(AES_CTRL_INT_CLR) |= (AES_CTRL_INT_CLR_DMA_IN_DONE |
158  AES_CTRL_INT_CLR_RESULT_AV);
159 
160  // disable master control/DMA clock
161  HWREG(AES_CTRL_ALG_SEL) = 0x00000000;
162 
163  // check status, if error return error code
164  if((HWREG(AES_KEY_STORE_WRITTEN_AREA) & 0x7) != 0x1)
165  {
166  g_ui8CurrentAESOp = AES_NONE;
167  return (AES_KEYSTORE_WRITE_ERROR);
168  }
169 
170  for(i = 0; i < KEY_BLENGTH; i++)
171  {
172  pui8temp[i] = 0;
173  }
174 
175  g_ui8CurrentAESOp = AES_NONE;
176  return (AES_SUCCESS);
177 }

206 uint8_t AESECBStart(uint8_t *pui8MsgIn,
207  uint8_t *pui8MsgOut,
208  uint8_t ui8KeyLocation,
209  uint8_t ui8Encrypt,
210  uint8_t ui8IntEnable)
211 {
212  // workaround for AES registers not retained after PM2
213  g_ui8CurrentAESOp = AES_ECB;
214  HWREG(AES_CTRL_INT_CFG) = AES_CTRL_INT_CFG_LEVEL;
215  HWREG(AES_CTRL_INT_EN) = AES_CTRL_INT_EN_RESULT_AV;
216  if(ui8IntEnable)
217  {
218  IntPendClear(INT_AES);
219  IntEnable(INT_AES);
220  }
221 
222  // configure the master control module
223  // enable the DMA path to the AES engine
224  HWREG(AES_CTRL_ALG_SEL) = AES_CTRL_ALG_SEL_AES;
225  // clear any outstanding events
226  HWREG(AES_CTRL_INT_CLR) |= (AES_CTRL_INT_CLR_DMA_IN_DONE |
227  AES_CTRL_INT_CLR_RESULT_AV);
228 
229  HWREG(AES_KEY_STORE_READ_AREA) = (uint32_t)ui8KeyLocation;
230 
231  //wait until key is loaded to the AES module
232  do
233  {
234  ASM_NOP;
235  }
236  while((HWREG(AES_KEY_STORE_READ_AREA) & AES_KEY_STORE_READ_AREA_BUSY));
237 
238  // check for Key Store read error
239  if((HWREG(AES_CTRL_INT_STAT)& AES_CTRL_INT_STAT_KEY_ST_RD_ERR))
240  {
241  // Clear Key Store Read error
242  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_KEY_ST_RD_ERR;
243  return (AES_KEYSTORE_READ_ERROR);
244  }
245 
246  // configure AES engine
247  // program AES-ECB-128 encryption and no IV
248  if(ui8Encrypt)
249  {
250  HWREG(AES_AES_CTRL) = 0x0000000C;
251  }
252  else
253  {
254  HWREG(AES_AES_CTRL) = 0x00000008;
255  }
256 
257  // write length of the message (lo)
258  HWREG(AES_AES_C_LENGTH_0) = (uint32_t) AES_ECB_LENGTH;
259  // write length of the message (hi)
260  HWREG(AES_AES_C_LENGTH_1) = 0;
261 
262  // configure DMAC
263  // enable DMA channel 0
264  HWREG(AES_DMAC_CH0_CTRL) = AES_DMAC_CH0_CTRL_EN;
265 
266  // base address of the input data in ext. memory
267  HWREG(AES_DMAC_CH0_EXTADDR) = (uint32_t)pui8MsgIn;
268 
269  // input data length in bytes, equal to the message
270  HWREG(AES_DMAC_CH0_DMALENGTH) = AES_ECB_LENGTH;
271 
272  // length (may be non-block size aligned)
273  HWREG(AES_DMAC_CH1_CTRL) = AES_DMAC_CH1_CTRL_EN; // enable DMA channel 1
274 
275  // base address of the output data buffer
276  HWREG(AES_DMAC_CH1_EXTADDR) = (uint32_t)pui8MsgOut;
277 
278  // output data length in bytes, equal to the result
279  HWREG(AES_DMAC_CH1_DMALENGTH) = AES_ECB_LENGTH;
280 
281  return (AES_SUCCESS);
282 }

293 uint8_t AESECBCheckResult(void)
294 {
295  return (((HWREGB(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_RESULT_AV)) ||
296  ((HWREGB(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_DMA_BUS_ERR)) ||
297  ((HWREGB(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_KEY_ST_WR_ERR)) ||
298  ((HWREGB(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_KEY_ST_RD_ERR)));
299 }

309 uint8_t AESECBGetResult(void)
310 {
311  //check for errors
312  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_DMA_BUS_ERR))
313  {
314  // clear the DMA error bit
315  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_DMA_BUS_ERR;
316  return (AES_DMA_BUS_ERROR);
317  }
318  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_KEY_ST_WR_ERR))
319  {
320  // clear the Key Store Write error bit
321  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_KEY_ST_WR_ERR;
322  return (AES_KEYSTORE_WRITE_ERROR);
323  }
324  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_KEY_ST_RD_ERR))
325  {
326  // clear the Key Store Read error bit
327  HWREG(AES_CTRL_INT_CLR) |= AES_CTRL_INT_CLR_KEY_ST_RD_ERR;
328  return (AES_KEYSTORE_READ_ERROR);
329  }
330 
331  // if no errors then AES ECB operation was successful, disable AES
332  // interrupt
333  IntDisable(INT_AES);
334 
335  //clear DMA done and result available bits
336  HWREG(AES_CTRL_INT_CLR) |= (AES_CTRL_INT_CLR_DMA_IN_DONE |
337  AES_CTRL_INT_CLR_RESULT_AV);
338 
339  // result has already been copied to the output buffer by DMA
340  HWREG(AES_CTRL_ALG_SEL) = 0x00000000; // disable master control/DMA clock
341  HWREG(AES_AES_CTRL) = 0x00000000; // clear mode
342  g_ui8CurrentAESOp = AES_NONE;
343  return (AES_SUCCESS);
344 }
