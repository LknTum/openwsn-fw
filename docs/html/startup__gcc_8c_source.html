309 void
310 ResetISR (void)
311 {
312  uint32_t *pui32Src, *pui32Dest;
313 
314  //
315  // Workaround for PM debug issue
316  //
317  HWREG(SYS_CTRL_EMUOVR) = 0xFF;
318 
319 
320  /* Workaround for J-Link debug issue */
321  HWREG(NVIC_VTABLE) = (uint32_t)gVectors;
322 
323  //
324  // Copy the data segment initializers from flash to SRAM.
325  //
326  pui32Src = &_etext;
327  for(pui32Dest = &_data; pui32Dest < &_edata; )
328  {
329  *pui32Dest++ = *pui32Src++;
330  }
331 
332  //
333  // Zero fill the bss segment.
334  //
335  __asm(" ldr r0, =_bss\n"
336  " ldr r1, =_ebss\n"
337  " mov r2, #0\n"
338  " .thumb_func\n"
339  "zero_loop:\n"
340  " cmp r0, r1\n"
341  " it lt\n"
342  " strlt r2, [r0], #4\n"
343  " blt zero_loop");
344 
345 #ifdef CC2538_USE_ALTERNATE_INTERRUPT_MAP
346  //
347  // Enable alternate interrupt mapping
348  //
349  HWREG(SYS_CTRL_I_MAP) |= 1;
350 #endif
351  //
352  // Call the application's entry point.
353  //
354  main();
355 
356  //
357  // End here if return from main()
358  //
359  while(1)
360  {
