49 void radiotimer_start(PORT_RADIOTIMER_WIDTH period) {
50  // radio's SFD pin connected to P4.1
51  P4DIR &= ~0x02; // input
52  P4SEL |= 0x02; // in CCI1a/B mode
53 
54  // CCR0 contains period of counter
55  // do not interrupt when counter reaches TBCCR0, but when it resets
56  TBCCR0 = period-1;
57 
58  // CCR1 in capture mode
59  TBCCTL1 = CM_3+SCS+CAP+CCIE;
60  TBCCR1 = 0;
61

63  TBCCTL2 = 0;
64  TBCCR2 = 0;

124 kick_scheduler_t radiotimer_isr() {
125  uint16_t tbiv_local;
126 
127  // reading TBIV returns the value of the highest pending interrupt flag
128  // and automatically resets that flag. We therefore copy its value to the
129  // tbiv_local local variable exactly once. If there is more than one
130  // interrupt pending, we will reenter this function after having just left
131  // it.
132  tbiv_local = TBIV;
133 
134  switch (tbiv_local) {
135  case 0x0002: // CCR1 fires
136  if (TBCCTL1 & CCI) {
137  // SFD pin is high: this was the start of a frame
138  if (radiotimer_vars.startFrameCb!=NULL) {
139  radiotimer_vars.startFrameCb(TBCCR1);
140  // kick the OS
141  return KICK_SCHEDULER;
142  }
143  } else {
144  // SFD pin is low: this was the end of a frame
145  if (radiotimer_vars.endFrameCb!=NULL) {
146  radiotimer_vars.endFrameCb(TBCCR1);
147  // kick the OS
148  return KICK_SCHEDULER;
149  }
150  }
151  break;
152  case 0x0004: // CCR2 fires
153  if (radiotimer_vars.compareCb!=NULL) {
154  radiotimer_vars.compareCb();
155  // kick the OS
156  return KICK_SCHEDULER;
157  }
158  break;
159  case 0x0006: // CCR3 fires
160  break;
161  case 0x0008: // CCR4 fires
162  break;
163  case 0x000a: // CCR5 fires
164  break;
165  case 0x000c: // CCR6 fires
166  break;
167  case 0x000e: // timer overflow
168  if (radiotimer_vars.overflowCb!=NULL) {
169  radiotimer_vars.overflowCb();
170  // kick the OS
171  return KICK_SCHEDULER;
172  }
173  break;
174  }
175  return DO_NOT_KICK_SCHEDULER;
176 }
