239 void
240 IOCPinConfigPeriphOutput(uint32_t ui32Port, uint8_t ui8Pins,
241  uint32_t ui32OutputSignal)
242 {
243  uint32_t ui32PortAddr;
244  uint32_t ui32PinNo;
245  uint32_t ui32PinBit;
246 
247  //
248  // Check the arguments
249  //
250  ASSERT((ui32Port == GPIO_A_BASE) || (ui32Port == GPIO_B_BASE) ||
251  (ui32Port == GPIO_C_BASE) || (ui32Port == GPIO_D_BASE));
252  ASSERT(ui8Pins != 0);
253 
254  //
255  // Initialize to default value
256  //
257  ui32PortAddr = IOC_PA0_SEL;
258 
259  //
260  // Look for specified port pins to be configured, multiple pins are allowed
261  //
262  for(ui32PinNo = 0; ui32PinNo < 8; ui32PinNo++)
263  {
264  ui32PinBit = (ui8Pins >> ui32PinNo) & 0x00000001;
265  if(ui32PinBit != 0)
266  {
267  //
268  // Find register addresses for configuring specified port pin
269  //
270  switch(ui32Port)
271  {
272  case GPIO_A_BASE:
273  ui32PortAddr = g_pui32IOCPortASignSelectReg[ui32PinNo];
274  break;
275 
276  case GPIO_B_BASE:
277  ui32PortAddr = g_pui32IOCPortBSignSelectReg[ui32PinNo];
278  break;
279 
280  case GPIO_C_BASE:
281  ui32PortAddr = g_pui32IOCPortCSignSelectReg[ui32PinNo];
282  break;
283 
284  case GPIO_D_BASE:
285  ui32PortAddr = g_pui32IOCPortDSignSelectReg[ui32PinNo];
286  break;
287 
288  default:
289  // Default to port A pin 0
290  ui32PortAddr = IOC_PA0_SEL;
291  break;
292  }
293 
294  //
295  // Set the mux for the desired port pin to select the desired
296  // peripheral output signal
297  //
298  HWREG(ui32PortAddr) = ui32OutputSignal;
299  }
300  }
