46 #define GPIO_O_DATA 0x00000000 // This is the data register. In
47  // software control mode, values
48  // written in the GPIODATA register
49  // are transferred onto the GPOUT
50  // pins if the respective pins have
51  // been configured as outputs
52  // through the GPIODIR register. A
53  // read from GPIODATA returns the
54  // last bit value written if the
55  // respective pins are configured
56  // as output, or it returns the
57  // value on the corresponding input
58  // GPIN bit when these are
59  // configured as inputs.
60 #define GPIO_O_DIR 0x00000400 // The DIR register is the data
61  // direction register. All bits are
62  // cleared by a reset; therefore,
63  // the GPIO pins are input by
64  // default.
65 #define GPIO_O_IS 0x00000404 // The IS register is the
66  // interrupt sense register.
67 #define GPIO_O_IBE 0x00000408 // The IBE register is the
68  // interrupt both-edges register.
69  // When the corresponding bit in IS
70  // is set to detect edges, bits set
71  // to high in IBE configure the
72  // corresponding pin to detect both
73  // rising and falling edges,
74  // regardless of the corresponding
75  // bit in the IEV (interrupt event
76  // register). Clearing a bit
77  // configures the pin to be
78  // controlled by IEV.
79 #define GPIO_O_IEV 0x0000040C // The IEV register is the
80  // interrupt event register. Bits
81  // set to high in IEV configure the
82  // corresponding pin to detect
83  // rising edges or high levels,
84  // depending on the corresponding
85  // bit value in IS. Clearing a bit
86  // configures the pin to detect
87  // falling edges or low levels,
88  // depending on the corresponding
89  // bit value in IS.
90 #define GPIO_O_IE 0x00000410 // The IE register is the
91  // interrupt mask register. Bits
92  // set to high in IE allow the
93  // corresponding pins to trigger
94  // their individual interrupts and
95  // the combined GPIOINTR line.
96  // Clearing a bit disables
97  // interrupt triggering on that
98  // pin.
99 #define GPIO_O_RIS 0x00000414 // The RIS register is the raw
100  // interrupt status register. Bits
101  // read high in RIS reflect the
102  // status of interrupts trigger
103  // conditions detected (raw, before
104  // masking), indicating that all
105  // the requirements are met, before
106  // they are finally allowed to
107  // trigger by IE. Bits read as 0
108  // indicate that corresponding
109  // input pins have not initiated an
110  // interrupt.
111 #define GPIO_O_MIS 0x00000418 // The MIS register is the masked
112  // interrupt status register. Bits
113  // read high in MIS reflect the
114  // status of input lines triggering
115  // an interrupt. Bits read as low
116  // indicate that either no
117  // interrupt has been generated, or
118  // the interrupt is masked. MIS is
119  // the state of the interrupt after
120  // masking.
121 #define GPIO_O_IC 0x0000041C // The IC register is the
122  // interrupt clear register.
123  // Writing 1 to a bit in this
124  // register clears the
125  // corresponding interrupt edge
126  // detection logic register.
127  // Writing 0 has no effect.
128 #define GPIO_O_AFSEL 0x00000420 // The AFSEL register is the mode
129  // control select register. Writing
130  // 1 to any bit in this register
131  // selects the hardware
132  // (peripheral) control for the
133  // corresponding GPIO line. All
134  // bits are cleared by a reset,
135  // therefore no GPIO line is set to
136  // hardware control by default.
137 #define GPIO_O_GPIOLOCK 0x00000520 // A write of the value 0x4C4F434B
138  // to the GPIOLOCK register unlocks
139  // the GPIO commit register
140  // (GPIOCR) for write access. A
141  // write of any other value
142  // reapplies the lock, preventing
143  // any register updates. Any write
144  // to the commit register (GPIOCR)
145  // causes the lock register to be
146  // locked.
147 #define GPIO_O_GPIOCR 0x00000524 // The GPIOCR register is the
148  // commit register. The value of
149  // the GPIOCR register determines
150  // which bits of the AFSEL register
151  // is committed when a write to the
152  // AFSEL register is performed. If
153  // a bit in the GPIOCR register is
154  // 0, the data being written to the
155  // corresponding bit in the AFSEL
156  // register is not committed and
157  // retains its previous value. If a
158  // bit in the GPIOCR register is
159  // set to 1, the data being written
160  // to the corresponding bit of the
161  // AFSEL register is committed to
162  // the register and will reflect
163  // the new value. The contents of
164  // the GPIOCR register can only be
165  // modified if the GPIOLOCK
166  // register is unlocked. Writes to
167  // the GPIOCR register will be
168  // ignored if the GPIOLOCK register
169  // is locked. Any write to the
170  // commit register causes the lock
171  // register to be locked.
172 #define GPIO_O_PMUX 0x00000700 // The PMUX register can be used
173  // to output external decouple
174  // control and clock_32k on I/O
175  // pins. Decouple control can be
176  // output on specific PB pins and
177  // clock_32k can be output on a
178  // specific PA or PB pin. These
179  // features override the current
180  // setting of the selected pin when
181  // enabled. The pin is set to
182  // output, pull-up and -down
183  // disabled, and analog mode
184  // disabled.
185 #define GPIO_O_P_EDGE_CTRL 0x00000704 // The port edge control register

187  // each port input causes that port
188  // to generate a power-up interrupt

190 #define GPIO_O_USB_CTRL 0x00000708 // This register is used to
191  // control which edge of the USB
192  // controller input generates a
193  // power-up interrupt to the
194  // system.
195 #define GPIO_O_PI_IEN 0x00000710 // The power-up interrupt enable
196  // register selects, for its

198  // whether interrupts are enabled

200 #define GPIO_O_IRQ_DETECT_ACK 0x00000718 // If the IRQ detect ACK register

203  // enabled I/O port is responsible

205  // interrupt to the system. Writing
206  // the IRQ detect ACK register is
207  // used to clear any number of
208  // individual port bits that may be
209  // signaling that an edge was
210  // detected as configured by the
