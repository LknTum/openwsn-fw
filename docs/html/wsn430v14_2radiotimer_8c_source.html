109 kick_scheduler_t radiotimer_isr(void) {
110  PORT_RADIOTIMER_WIDTH taiv_local;
111 
112  // reading TAIV returns the value of the highest pending interrupt flag
113  // and automatically resets that flag. We therefore copy its value to the
114  // tbiv_local local variable exactly once. If there is more than one
115  // interrupt pending, we will reenter this function after having just left
116  // it.
117  taiv_local = TAIV;
118 
119  switch (taiv_local) {
120  case 0x0002: // CCR1 fires
121  if (radiotimer_vars.compareCb!=NULL) {
122  // call the callback
123  radiotimer_vars.compareCb();
124  // kick the OS
125  return KICK_SCHEDULER;
126  }
127  break;
128  case 0x0004: // CCR2 fires
129  break;
130  case 0x000a: // timer overflow
131  if (radiotimer_vars.overflowCb!=NULL) {
132  radiotimer_vars.overflowCb();
133  // kick the OS
134  return KICK_SCHEDULER;
135  }
136  break;
137  }
138  return DO_NOT_KICK_SCHEDULER;
139 }
