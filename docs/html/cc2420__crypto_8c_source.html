227 static owerror_t cc2420_crypto_load_key(uint8_t key[16], uint8_t* /* out */ key_index) {
228  static uint8_t loaded_key[2][16]; // to save some SPI transfers, keep a copy of
229  // loaded keys in MCU RAM.
230  static uint8_t num_keys_loaded = 0;
231  cc2420_status_t status;
232  uint8_t reversed[16];
233  uint8_t i;
234  uint8_t next_key_index;
235 
236  // verify if crystal oscillator is stable
237  cc2420_spiStrobe(CC2420_SNOP, &status);
238 
239  if (status.xosc16m_stable) { // green light only if CC2420 is ready
240 
241  for (i = 0; i < 2; i++) {
242  if (num_keys_loaded > i && memcmp(loaded_key[i], key, 16) == 0) { // first cond. to avoid all zero key hit
243  *key_index = i;
244  return E_SUCCESS;
245  }
246  }
247 
248  next_key_index = num_keys_loaded & 0x01;
249 
250  memcpy(loaded_key[next_key_index], key, CC2420_KEY_LEN);
251  memcpy(reversed, loaded_key[next_key_index], CC2420_KEY_LEN);
252  // Need to transfer it little-endian
253  reverse(reversed, CC2420_KEY_LEN);
254 
255  // Load the key in key RAM
256  cc2420_spiWriteRam(next_key_index == 0 ? CC2420_RAM_KEY0_ADDR : CC2420_RAM_KEY1_ADDR,
257  &status, reversed, CC2420_KEY_LEN);
258 
259  *key_index = next_key_index;
260  num_keys_loaded++;
261  return E_SUCCESS;
262  }
263  return E_FAIL;
264 }

318 static void create_cc2420_nonce(uint8_t l,
319  uint8_t len_mac,
320  uint8_t len_a,
321  uint8_t* nonce154,
322  uint8_t* buffer) {
323  // Create nonce
324  buffer[0] = 0x00; // set flags to zero including reserved
325  buffer[0] |= 0x07 & (l-1); // field L
326  buffer[0] |= len_a != 0 ? 0x08 : 0; // field Adata in CC2420's nonce (see datasheet)
327  memcpy(&buffer[1], nonce154, 13);
328  if (l == 3) { buffer[13] = 0; }
329  buffer[14] = 0x00; // block counter
330  buffer[15] = 0x01; // block counter
331  reverse(buffer, 16);
332 }
