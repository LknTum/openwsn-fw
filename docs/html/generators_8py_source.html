52 # find the first docstring in 's' at or after 'index'
53 # a docstring is an @ preceded by whitespace and followed by
54 # at least one letter
55 def _find_docstring_tag(s, index):

57  at = find(s, "@", index)

61  index = at + 1

63 # process a docstring s. returns a tuple of
64 # - base documentation
65 # - list of (tag, description) pairs (in the same order as in s)
66 def nd_docstring(s):
67  tags = []
68  at = _find_docstring_tag(s, 0)
69  if at < 0:
70  return (s, tags)
71  base = s[:at]
72  while at >= 0:
73  # find end of tag
74  tagend = at + 1
75  while tagend < len(s) and not s[tagend].isspace():
76  tagend += 1
77  nextat = _find_docstring_tag(s, tagend)
78  if nextat == -1:
79  tagvalend = len(s)
80  else:
81  tagvalend = nextat
82  tags.append((s[at + 1:tagend], s[tagend + 1:tagvalend]))
83  at = nextat
84  return (base, tags)

180 # Return a (head, body) pair for type xmltype, declaring name (a string)
181 # with type qualifiers quals (a string). isstar should be true if name
182 # starts with a *.
183 # The user-friendly representation for this type is head + " " + body
184 def typename_full(xmltype, name, quals, isstar):
185  # hack around nesC 1.2.1 schema bug (uses typedef, not typename here)
186  tdef = xml_tagset(xmltype, ["typedef", "typename"])
187  if tdef:
