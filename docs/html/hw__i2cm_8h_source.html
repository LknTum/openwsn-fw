46 #define I2CM_SA 0x40020000 // I2C master slave address This
47  // register consists of eight bits,
48  // seven address bits (A6-A0), and
49  // a receive and send bit, which
50  // determines if the next operation
51  // is a receive (high) or transmit
52  // (low).
53 #define I2CM_CTRL 0x40020004 // I2C master control and status
54  // This register accesses status
55  // bits when read and control bits
56  // when written. When read, the
57  // status register indicates the
58  // state of the I2C bus controller.
59  // When written, the control
60  // register configures the I2C
61  // controller operation. The START
62  // bit generates the START or
63  // REPEATED START condition. The
64  // STOP bit determines if the cycle
65  // stops at the end of the data
66  // cycle or continues on to a
67  // repeated START condition. To
68  // generate a single transmit
69  // cycle, the I2C master slave
70  // address (I2CMSA) register is
71  // written with the desired
72  // address, the R/S bit is cleared,
73  // and this register is written
74  // with ACK = X (0 or 1), STOP = 1,
75  // START = 1, and RUN = 1 to
76  // perform the operation and stop.
77  // When the operation is completed
78  // (or aborted due an error), an
79  // interrupt becomes active and the
80  // data may be read from the I2CMDR
81  // register. When the I2C module
82  // operates in master receiver
83  // mode, the ACK bit is normally
84  // set, causing the I2C bus
85  // controller to automatically
86  // transmit an acknowledge after
87  // each byte. This bit must be
88  // cleared when the I2C bus
89  // controller requires no further
90  // data to be transmitted from the
91  // slave transmitter.
92 #define I2CM_STAT 0x40020004 // I2C master control and status
93  // This register accesses status
94  // bits when read and control bits
95  // when written. When read, the
96  // status register indicates the
97  // state of the I2C bus controller.
98  // When written, the control
99  // register configures the I2C
100  // controller operation. The START
101  // bit generates the START or
102  // REPEATED START condition. The
103  // STOP bit determines if the cycle
104  // stops at the end of the data
105  // cycle or continues on to a
106  // repeated START condition. To
107  // generate a single transmit
108  // cycle, the I2C master slave
109  // address (I2CMSA) register is
110  // written with the desired
111  // address, the R/S bit is cleared,
112  // and this register is written
113  // with ACK = X (0 or 1), STOP = 1,
114  // START = 1, and RUN = 1 to
115  // perform the operation and stop.
116  // When the operation is completed
117  // (or aborted due an error), an
118  // interrupt becomes active and the
119  // data may be read from the I2CMDR
120  // register. When the I2C module
121  // operates in master receiver
