70 #define SMWDTHROSC_WDCTL_CLR_M 0x000000F0 // Clear timer When 0xA followed
71  // by 0x5 is written to these bits,
72  // the timer is loaded with 0x0000.
73  // Note that 0x5 must be written
74  // within one watchdog clock period
75  // Twdt after 0xA was written for
76  // the clearing to take effect
77  // (ensured). If 0x5 is written
78  // between Twdt and 2Twdt after 0xA
79  // was written, the clearing may
80  // take effect, but there is no
81  // guarantee. If 0x5 is written >
82  // 2Twdt after 0xA was written, the
83  // timer will not be cleared. If a
84  // value other than 0x5 is written
85  // after 0xA has been written, the
86  // clear sequence is aborted. If
87  // 0xA is written, this starts a
88  // new clear sequence. Writing to
89  // these bits when EN = 0 has no
90  // effect.
91 #define SMWDTHROSC_WDCTL_CLR_S 4
92 #define SMWDTHROSC_WDCTL_EN 0x00000008 // Enable timer When 1 is written
93  // to this bit the timer is enabled
94  // and starts incrementing. The
95  // interval setting specified by
96  // INT[1:0] is used. Writing 0 to
97  // this bit have no effect.
98 #define SMWDTHROSC_WDCTL_EN_M 0x00000008
99 #define SMWDTHROSC_WDCTL_EN_S 3
100 #define SMWDTHROSC_WDCTL_INT_M 0x00000003 // Timer interval select These
101  // bits select the timer interval
102  // as follows: 00: Twdt x 32768 01:
103  // Twdt x 8192 10: Twdt x 512 11:
104  // Twdt x 64 Writing these bits
105  // when EN = 1 has no effect.
106 #define SMWDTHROSC_WDCTL_INT_S 0
