46 #define GPTIMER_O_CFG 0x00000000 // GPTM configuration This
47  // register configures the global
48  // operation of the GPTM. The value
49  // written to this register
50  // determines whether the GPTM is
51  // in 32-bit mode (concatenated
52  // timers) or in 16-bit mode
53  // (individual, split timers).
54 #define GPTIMER_O_TAMR 0x00000004 // GPTM Timer A mode This register
55  // configures the GPTM based on the
56  // configuration selected in the
57  // CFG register. This register
58  // controls the modes for Timer A
59  // when it is used individually.
60  // When Timer A and Timer B are
61  // concatenated, this register
62  // controls the modes for both
63  // Timer A and Timer B, and the
64  // contents of TBMR are ignored.
65 #define GPTIMER_O_TBMR 0x00000008 // GPTM Timer B mode This register
66  // configures the GPTM based on the
67  // configuration selected in the
68  // CFG register. This register
69  // controls the modes for Timer B
70  // when it is used individually.
71  // When Timer A and Timer B are
72  // concatenated, this register is
73  // ignored and TBMR controls the
74  // modes for both Timer A and Timer
75  // B.
76 #define GPTIMER_O_CTL 0x0000000C // GPTM control This register is
77  // used alongside the CFG and TnMR
78  // registers to fine-tune the timer
79  // configuration, and to enable
80  // other features such as timer
81  // stall.
82 #define GPTIMER_O_SYNC 0x00000010 // GPTM synchronize Note: This
83  // register is implemented on GPTM
84  // 0 base address only. This
85  // register does however, allow
86  // software to synchronize a number
87  // of timers.
88 #define GPTIMER_O_IMR 0x00000018 // GPTM interrupt mask This
89  // register allows software to
90  // enable and disable GPTM
91  // controller-level interrupts.
92  // Setting a bit enables the
93  // corresponding interrupt, while
94  // clearing a bit disables it.
95 #define GPTIMER_O_RIS 0x0000001C // GPTM raw interrupt status This
96  // register shows the state of the
97  // GPTM internal interrupt signal.
98  // These bits are set whether or
99  // not the interrupt is masked in
100  // the IMR register. Each bit can
101  // be cleared by writing 1 to its
102  // corresponding bit in ICR.
103 #define GPTIMER_O_MIS 0x00000020 // GPTM masked interrupt status
104  // This register shows the state of
105  // the GPTM controller-level
106  // interrupt. If an interrupt is
107  // unmasked in IMR, and there is an
108  // event that causes the interrupt
109  // to be asserted, the
110  // corresponding bit is set in this
111  // register. All bits are cleared
112  // by writing 1 to the
113  // corresponding bit in ICR.
114 #define GPTIMER_O_ICR 0x00000024 // GPTM interrupt clear This
115  // register is used to clear the
116  // status bits in the RIS and MIS
117  // registers. Writing 1 to a bit
118  // clears the corresponding bit in
119  // the RIS and MIS registers.
120 #define GPTIMER_O_TAILR 0x00000028 // GPTM Timer A interval load When
121  // the Timer is counting down, this
122  // register is used to load the
123  // starting count value into the
124  // Timer. When the Timer is
125  // counting up, this register sets
126  // the upper bound for the timeout
127  // event. When a GPTM is configured
128  // to one of the 32-bit modes,
129  // TAILR appears as a 32-bit
130  // register (the upper 16-bits
131  // correspond to the contents of
132  // the GPTM Timer B Interval Load
133  // (TBILR) register). In a 16-bit
134  // mode, the upper 16 bits of this
135  // register read as 0s and have no
136  // effect on the state of TBILR.
137 #define GPTIMER_O_TBILR 0x0000002C // GPTM Timer B interval load When
138  // the Timer is counting down, this
139  // register is used to load the
140  // starting count value into the
141  // Timer. When the Timer is
142  // counting up, this register sets
143  // the upper bound for the time-out
144  // event. When a GPTM is configured
145  // to one of the 32-bit modes, the
146  // contents of bits [15:0] in this
147  // register are loaded into the
148  // upper 16 bits of the TAILR
149  // register. Reads from this
150  // register return the current
151  // value of Timer B and writes are
152  // ignored. In a 16-bit mode, bits
153  // [15:0] are used for the load
154  // value. Bits [31:16] are reserved
155  // in both cases.
156 #define GPTIMER_O_TAMATCHR 0x00000030 // GPTM Timer A match This
157  // register is loaded with a match
158  // value. Interrupts can be
159  // generated when the Timer value
160  // is equal to the value in this
161  // register in one-shot or periodic
162  // mode. When a GPTM is configured
163  // to one of the 32-bit modes,
164  // TAMATCHR appears as a 32-bit
165  // register (the upper 16-bits
166  // correspond to the contents of
167  // the GPTM Timer B match
168  // (GPTMTBMATCHR) register). In a
169  // 16-bit mode, the upper 16 bits
170  // of this register read as 0s and
171  // have no effect on the state of
172  // TBMATCHR.
173 #define GPTIMER_O_TBMATCHR 0x00000034 // PTM Timer B match This register
174  // is loaded with a match value.
175  // Interrupts can be generated when
176  // the Timer value is equal to the
177  // value in this register in
178  // one-shot or periodic mode. When
179  // a GPTM is configured to one of
180  // the 32-bit modes, the contents
181  // of bits [15:0] in this register
182  // are loaded into the upper 16
183  // bits of the TAMATCHR register.
184  // Reads from this register return
185  // the current match value of Timer
186  // B and writes are ignored. In a
187  // 16-bit mode, bits [15:0] are
188  // used for the match value. Bits
189  // [31:16] are reserved in both
190  // cases.
191 #define GPTIMER_O_TAPR 0x00000038 // GPTM Timer A prescale This
192  // register allows software to
193  // extend the range of the 16-bit
194  // Timers in periodic and one-shot
195  // modes.
196 #define GPTIMER_O_TBPR 0x0000003C // GPTM Timer B prescale This
197  // register allows software to
198  // extend the range of the 16-bit
199  // Timers in periodic and one-shot
200  // modes.
201 #define GPTIMER_O_TAPMR 0x00000040 // GPTM Timer A prescale match
202  // This register effectively
203  // extends the range of TAMATCHR to

205  // 16-bit, one-shot or periodic
206  // mode.
207 #define GPTIMER_O_TBPMR 0x00000044 // GPTM Timer B prescale match
208  // This register effectively
209  // extends the range ofMTBMATCHR to
210  // 24 bits when operating in
211  // 16-bit, one-shot or periodic
212  // mode.
213 #define GPTIMER_O_TAR 0x00000048 // GPTM Timer A This register
214  // shows the current value of the
215  // Timer A counter. When a GPTM is
216  // configured to one of the 32-bit
217  // modes, TAR appears as a 32-bit
218  // register (the upper 16-bits
219  // correspond to the contents of
220  // the GPTM Timer B (TBR)
221  // register). In the16-bit Input
222  // edge count, input edge time, and
223  // PWM modes, bits [15:0] contain
224  // the value of the counter and
225  // bits 23:16 contain the value of
226  // the prescaler, which is the
227  // upper 8 bits of the count. Bits
228  // [31:24] always read as 0. To
229  // read the value of the prescaler
230  // in 16-bit, one-shot and periodic
231  // modes, read bits [23:16] in the
232  // TAV register.
233 #define GPTIMER_O_TBR 0x0000004C // GPTM Timer B This register
234  // shows the current value of the
235  // Timer B counter. When a GPTM is
236  // configured to one of the 32-bit
237  // modes, the contents of bits
238  // [15:0] in this register are
239  // loaded into the upper 16 bits of
240  // the TAR register. Reads from
241  // this register return the current
242  // value of Timer B. In a 16-bit
243  // mode, bits 15:0 contain the
244  // value of the counter and bits
245  // [23:16] contain the value of the
246  // prescaler in Input edge count,
247  // input edge time, and PWM modes,
248  // which is the upper 8 bits of the
249  // count. Bits [31:24] always read
250  // as 0. To read the value of the
251  // prescaler in 16-bit, one-shot
252  // and periodic modes, read bits
253  // [23:16] in the TBV register.

260  // the ISR entry when using the
261  // snapshot feature with the
262  // periodic operating mode. When
263  // written, the value written into
264  // this register is loaded into the
265  // TAR register on the next clock
266  // cycle. When a GPTM is configured
267  // to one of the 32-bit modes, TAV
268  // appears as a 32-bit register
269  // (the upper 16-bits correspond to
270  // the contents of the GPTM Timer B
271  // Value (TBV) register). In a
272  // 16-bit mode, bits [15:0] contain
273  // the value of the counter and
274  // bits [23:16] contain the
275  // current, free-running value of
276  // the prescaler, which is the
277  // upper 8 bits of the count in
278  // input edge count, input edge
279  // time, PWM and one-shot or
280  // periodic up count modes. In
281  // one-shot or periodic down count
282  // modes, the prescaler stored in
283  // [23:16] is a true prescaler,
284  // meaning bits [23:16] count down
285  // before decrementing the value in
286  // bits [15:0]. The prescaler its
287  // [31:24] always read as 0.

300  // bits 15:0 in this register are

302  // the TAV register. Reads from
303  // this register return the current
304  // free-running value of Timer B.
