46 #define UART_O_DR 0x00000000 // UART data Important: This
47  // register is read-sensitive. See
48  // the register description for
49  // details. This register is the
50  // data register (the interface to
51  // the FIFOs). For transmitted
52  // data, if the FIFO is enabled,
53  // data written to this location is
54  // pushed onto the transmit FIFO.
55  // If the FIFO is disabled, data is
56  // stored in the transmitter
57  // holding register (the bottom
58  // word of the transmit FIFO). A
59  // write to this register initiates
60  // a transmission from the UART.
61  // For received data, if the FIFO
62  // is enabled, the data byte and
63  // the 4-bit status (break, frame,
64  // parity, and overrun) is pushed
65  // onto the 12-bit wide receive
66  // FIFO. If the FIFO is disabled,
67  // the data byte and status are
68  // stored in the receiving holding
69  // register (the bottom word of the
70  // receive FIFO). The received data
71  // can be retrieved by reading this
72  // register.
73 #define UART_O_RSR 0x00000004 // UART receive status and error
74  // clear The RSR/ECR register is
75  // the receive status register and
76  // error clear register. In
77  // addition to the DR register,
78  // receive status can also be read
79  // from the RSR register. If the
80  // status is read from this
81  // register, then the status
82  // information corresponds to the
83  // entry read from DR before
84  // reading RSR. The status
85  // information for overrun is set
86  // immediately when an overrun
87  // condition occurs. The RSR
88  // register cannot be written.
89  // Read-only status register
90 #define UART_O_ECR 0x00000004 // UART receive status and error
91  // clear The RSR/ECR register is
92  // the receive status
93  // register/error clear register. A
94  // write of any value to the ECR
95  // register clears the framing,
96  // parity, break, and overrun
97  // errors. All the bits are cleared
98  // on reset. Write-only error clear
99  // register
100 #define UART_O_FR 0x00000018 // UART flag The FR register is
101  // the flag register. After reset,
102  // the TXFF, RXFF, and BUSY bits
103  // are 0, and TXFE and RXFE bits
104  // are 1. The CTS bit indicate the
105  // modem flow control. Note that
106  // the modem bits are only
107  // implemented on UART1 and are
108  // tied inactive on UART0. Due to
109  // this difference, the reset state
110  // of the UART0 FR register is
111  // 0x90, while UART1 FR register
112  // reset state 0x197 .
113 #define UART_O_ILPR 0x00000020 // UART IrDA low-power register
114  // The ILPR register stores the
115  // 8-bit low-power counter divisor
116  // value used to derive the
117  // low-power SIR pulse width clock
118  // by dividing down the system
119  // clock (SysClk). All the bits are
120  // cleared when reset. The internal
121  // IrLPBaud16 clock is generated by
122  // dividing down SysClk according
123  // to the low-power divisor value
124  // written to ILPR. The duration of
125  // SIR pulses generated when
126  // low-power mode is enabled is
127  // three times the period of the
128  // IrLPBaud16 clock. The low-power
129  // divisor value is calculated as
130  // follows: ILPDVSR = SysClk /
131  // FIrLPBaud16 where FIrLPBaud16 is
132  // nominally 1.8432 MHz The divisor
133  // must be programmed such that
134  // FIrLPBaud16 is in the range 1.42
135  // MHz to 2.12 MHz, resulting in a
136  // low-power pulse duration of
137  // 1.41-2.11 us (three times the
138  // period of IrLPBaud16). The
139  // minimum frequency of IrLPBaud16
140  // ensures that pulses less than
141  // one period of IrLPBaud16 are
142  // rejected, but pulses greater
143  // than 1.4 us are accepted as
144  // valid pulses. Note: Zero is an
145  // illegal value. Programming a
146  // zero value results in no
147  // IrLPBaud16 pulses being
148  // generated.
149 #define UART_O_IBRD 0x00000024 // UART integer baud-rate divisor
150  // The IBRD register is the integer
151  // part of the baud-rate divisor
152  // value. All the bits are cleared
153  // on reset. The minimum possible
154  // divide ratio is 1 (when IBRD =
155  // 0), in which case the FBRD
156  // register is ignored. When
157  // changing the IBRD register, the
158  // new value does not take effect
159  // until transmission or reception
160  // of the current character is
161  // complete. Any changes to the
162  // baud-rate divisor must be
163  // followed by a write to the LCRH
164  // register.
165 #define UART_O_FBRD 0x00000028 // UART fractional baud-rate
166  // divisor The FBRD register is the
167  // fractional part of the baud-rate
168  // divisor value. All the bits are
169  // cleared on reset. When changing
170  // the FBRD register, the new value
171  // does not take effect until
172  // transmission or reception of the
173  // current character is complete.
174  // Any changes to the baud-rate
175  // divisor must be followed by a
176  // write to the LCRH register.
177 #define UART_O_LCRH 0x0000002C // UART line control The LCRH
178  // register is the line control
179  // register. Serial parameters such
180  // as data length, parity, and stop
181  // bit selection are implemented in
182  // this register. When updating the
183  // baud-rate divisor (IBRD and/or
184  // IFRD), the LCRH register must
185  // also be written. The write
186  // strobe for the baud-rate divisor
187  // registers is tied to the LCRH
188  // register.
189 #define UART_O_CTL 0x00000030 // UART control The CTL register
190  // is the control register. All the
191  // bits are cleared on reset except
192  // for the transmit enable (TXE)
193  // and receive enable (RXE) bits,
194  // which are set. To enable the
195  // UART module, the UARTEN bit must
196  // be set. If software requires a
197  // configuration change in the
198  // module, the UARTEN bit must be
199  // cleared before the configuration
200  // changes are written. If the UART
201  // is disabled during a transmit or
202  // receive operation, the current
203  // transaction is completed before
204  // the UART stopping. Note: The
205  // UARTCTL register should not be
206  // changed while the UART is
207  // enabled or else the results are
208  // unpredictable. The following
209  // sequence is recommended for
210  // making changes to the UARTCTL
211  // register: 1. Disable the UART.
212  // 2. Wait for the end of
213  // transmission or reception of the
214  // current character. 3. Flush the
215  // transmit FIFO by clearing bit 4
216  // (FEN) in the line control
217  // register (UARTLCRH). 4.
218  // Reprogram the control register.
219  // 5. Enable the UART.
220 #define UART_O_IFLS 0x00000034 // UART interrupt FIFO level
221  // select The IFLS register is the
222  // interrupt FIFO level select
223  // register. This register can be
224  // used to define the FIFO level at
225  // which the TXRIS and RXRIS bits
226  // in the RIS register are
227  // triggered. The interrupts are
228  // generated based on a transition
229  // through a level rather than
230  // being based on the level. That
231  // is, the interrupts are generated
232  // when the fill level progresses
233  // through the trigger level. For
234  // example, if the receive trigger
235  // level is set to the half-way
236  // mark, the interrupt is triggered
237  // as the module is receiving the
238  // 9th character. Out of reset, the
239  // TXIFLSEL and RXIFLSEL bits are
240  // configured so that the FIFOs
241  // trigger an interrupt at the
242  // half-way mark.
243 #define UART_O_IM 0x00000038 // UART interrupt mask The IM

246  // this register gives the current
247  // value of the mask on the

249  // bit allows the corresponding raw
250  // interrupt signal to be routed to
251  // the interrupt controller.
252  // Clearing a bit prevents the raw
253  // interrupt signal from being sent
254  // to the interrupt controller.

276  // raw interrupt and masked
