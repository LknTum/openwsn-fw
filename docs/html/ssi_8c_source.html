109 void
110 SSIConfigSetExpClk(uint32_t ui32Base, uint32_t ui32SSIClk,
111  uint32_t ui32Protocol, uint32_t ui32Mode,
112  uint32_t ui32BitRate, uint32_t ui32DataWidth)
113 {
114  uint32_t ui32MaxBitRate;
115  uint32_t ui32RegVal;
116  uint32_t ui32PreDiv;
117  uint32_t ui32SCR;
118  uint32_t ui32SPH_SPO;
119 
120  //
121  // Check the arguments.
122  //
123  ASSERT((ui32Base == SSI0_BASE) || (ui32Base == SSI1_BASE));
124  ASSERT((ui32Protocol == SSI_FRF_MOTO_MODE_0) ||
125  (ui32Protocol == SSI_FRF_MOTO_MODE_1) ||
126  (ui32Protocol == SSI_FRF_MOTO_MODE_2) ||
127  (ui32Protocol == SSI_FRF_MOTO_MODE_3) ||
128  (ui32Protocol == SSI_FRF_TI) ||
129  (ui32Protocol == SSI_FRF_NMW));
130  ASSERT((ui32Mode == SSI_MODE_MASTER) ||
131  (ui32Mode == SSI_MODE_SLAVE) ||
132  (ui32Mode == SSI_MODE_SLAVE_OD));
133  ASSERT(((ui32Mode == SSI_MODE_MASTER) && (ui32BitRate <= (ui32SSIClk / 2))) ||
134  ((ui32Mode != SSI_MODE_MASTER) && (ui32BitRate <= (ui32SSIClk / 12))));
135  ASSERT((ui32SSIClk / ui32BitRate) <= (254 * 256));
136  ASSERT((ui32DataWidth >= 4) && (ui32DataWidth <= 16));
137 
138  //
139  // Set the mode.
140  //
141  ui32RegVal = (ui32Mode == SSI_MODE_SLAVE_OD) ? SSI_CR1_SOD : 0;
142  ui32RegVal |= (ui32Mode == SSI_MODE_MASTER) ? 0 : SSI_CR1_MS;
143  HWREG(ui32Base + SSI_O_CR1) = ui32RegVal;
144 
145  //
146  // Set the clock predivider.
147  //
148  ui32MaxBitRate = ui32SSIClk / ui32BitRate;
149  ui32PreDiv = 0;
150  do
151  {
152  ui32PreDiv += 2;
153  ui32SCR = (ui32MaxBitRate / ui32PreDiv) - 1;
154  }
155  while(ui32SCR > 255);
156  HWREG(ui32Base + SSI_O_CPSR) = ui32PreDiv;
157 
158  //
159  // Set protocol and clock rate.
160  //
161  ui32SPH_SPO = (ui32Protocol & 3) << 6;
162  ui32Protocol &= SSI_CR0_FRF_M;
163  ui32RegVal = (ui32SCR << 8) | ui32SPH_SPO | ui32Protocol | (ui32DataWidth - 1);
164  HWREG(ui32Base + SSI_O_CR0) = ui32RegVal;
165 }
