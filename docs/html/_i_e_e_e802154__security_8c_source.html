605  case IEEE154_ASH_SLF_TYPE_ENC_MIC_128:

612  // unsecure, should not support it
613  return E_FAIL;

615  // reject anything else
616  return E_FAIL;
617  }

623  (errorparameter_t)11);
624  return E_FAIL;
625  }
626

628  outStatus = CRYPTO_ENGINE.aes_ccms_dec(a,
629  len_a,
630  c,
631  &len_c,
632  nonce,
633  2,
634  keyDescriptor->key,
635  msg->l2_authenticationLength);
636 
637  //verify if any error occurs
638  if (outStatus != E_SUCCESS){
639  openserial_printError(COMPONENT_SECURITY,ERR_SECURITY,
640  (errorparameter_t)msg->l2_frameType,
641  (errorparameter_t)12);
642  }
643 
644  packetfunctions_tossFooter(msg,msg->l2_authenticationLength);
645  return outStatus;
646 }
