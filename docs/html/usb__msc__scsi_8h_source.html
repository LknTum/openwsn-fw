155  uint_8 response_data_format;/*Bits 7..6: obsolete

162  uint_8 sccs;/*Bit 7: SCCS (0 = no embedded storage array controller
163  component present)
164  Bit 6: ACC (0 = no access controls coordinator present)
165  Bits 5..4: TPGS (0 = no support or vendor-specific support
166  for asymmetric logical unit access)
167  Bit 3: 3PC ((0 = no support for third-party copy commands)
168  Bits 2..1: reserved
169  Bit 0: PROTECT (0 = no support for protection information)*/
170  uint_8 bque;/*Bit 7: BQUE (0 = no support for basic task management)
171  Bit 6: ENCSERV (0=no support for embedded enclosure services)
172  Bit 5: VS (vendor specific)
173  Bit 4: MULTIP (0 = device has a single port)
174  Bit 3: MCHNGR (0 = no support for media changer)
175  Bits 2..1: obsolete
176  Bit 0: ADDR16 (not used with USB interface) */
177  uint_8 wbus;/*Bit 7..6: obsolete
178  Bit 5: WBUS16 (not used with USB interface)
179  Bit 4: SYNC (not used with USB interface)
180  Bit 3: LINKED (0 = no support for linked commands)
181  Bit 2: obsolete
182  Bit 1: CMDQUE (0 = no support for full task management)
183  Bit 0: VS (vendor specific)*/
184  uint_8 vendor_info[8];/*T10 VENDOR IDENTIFICATION, MSB first */
185  uint_8 product_id[16];/*PRODUCT IDENTIFICATION, MSB first*/
