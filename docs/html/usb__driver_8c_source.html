147 static void USB_Device_Init_Params(void)
148 {
149  uint_8 loop_index=0;
150 
151  g_EPn= g_EPn_max; /* 1 control endpoint */
152 
153  /*
154  Initialize USB_STATUS_DEVICE_STATE, USB_STATUS_INTERFACE,
155  USB_STATUS_ADDRESS, USB_STATUS_CURRENT_CONFIG, USB_STATUS_SOF_COUNT
156  and USB_STATUS_DEVICE to USB_STATUS_UNKNOWN
157  */
158  for(loop_index = 0; loop_index < USB_STATUS_TEST_MODE; loop_index++)
159  {
160  #ifdef OTG_BUILD
161  if(loop_index != (USB_STATUS_OTG-1)) /* Do not initialize the OTG status with 0xFFFF */
162  #endif
163  {
164  g_usb_component_status[loop_index] = USB_STATUS_UNKNOWN;
165  }
166  }
167 
168  /* Initialize status of All Endpoints to USB_STATUS_DISABLED */
169  for(loop_index = 0; loop_index < MAX_SUPPORTED_ENDPOINTS; loop_index++)
170  {
171  g_usb_ep_status[loop_index] = USB_STATUS_DISABLED;
172  }
173 }

272 uint_8 _usb_device_init_endpoint (
273  _usb_device_handle handle, /* [IN] USB Device handle */
274  uint_8 endpoint_number, /* [IN] Endpoint number*/
275  uint_16 max_packet_size, /* [IN] Maximum packet size (in bytes) for the endpoint */
276  uint_8 direction, /* [IN] Direction of transfer */
277  uint_8 endpoint_type, /* [IN] Type of endpoint */
278  uint_8 flag /* [IN] Zero termination flag */
279 )

284

287  {

289  }
290 
291  ep_str.direction = direction;
292  ep_str.ep_num = endpoint_number;
293 
294 #ifndef _MC9S08JS16_H
295  ep_str.size = max_packet_size;
296 #else
297  ep_str.size = (char)max_packet_size;
298 #endif
299  ep_str.type = endpoint_type;

468  (USB_RECV):(USB_SEND));

470  }
471  }
472  /* Set the corresponding endpoint setting */
473  g_usb_ep_status[ep_num] = setting;
474  }
