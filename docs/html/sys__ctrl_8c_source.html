199 void
200 SysCtrlClockSet(bool bExternalOsc32k, bool bInternalOsc,
201  uint32_t ui32SysDiv)
202 {
203  uint32_t ui32STA;
204  uint32_t ui32Reg;
205  uint32_t ui32TimeoutVal;
206  uint32_t ui32Osc;
207 
208  // check input parameters
209  ASSERT((ui32SysDiv == SYS_CTRL_SYSDIV_32MHZ ||
210  ui32SysDiv == SYS_CTRL_SYSDIV_16MHZ ||
211  ui32SysDiv == SYS_CTRL_SYSDIV_8MHZ ||
212  ui32SysDiv == SYS_CTRL_SYSDIV_4MHZ ||
213  ui32SysDiv == SYS_CTRL_SYSDIV_2MHZ ||
214  ui32SysDiv == SYS_CTRL_SYSDIV_1MHZ ||
215  ui32SysDiv == SYS_CTRL_SYSDIV_500KHZ ||
216  ui32SysDiv == SYS_CTRL_SYSDIV_250KHZ) &&
217  !((ui32SysDiv == SYS_CTRL_SYSDIV_32MHZ) && bInternalOsc));
218 
219  //
220  // Enable AMP detect to make sure XOSC starts correctly
221  //
222  if(!bInternalOsc)
223  {
224  ui32Reg = HWREG(SYS_CTRL_CLOCK_CTRL) | SYS_CTRL_CLOCK_CTRL_AMP_DET;
225  HWREG(SYS_CTRL_CLOCK_CTRL) = ui32Reg;
226  }
227 
228  //
229  // Set 32kHz clock, Osc and SysDiv
230  //
231  ui32Reg = HWREG(SYS_CTRL_CLOCK_CTRL);
232  ui32Reg &= ~(SYS_CTRL_CLOCK_CTRL_OSC32K | SYS_CTRL_CLOCK_CTRL_OSC |
233  SYS_CTRL_CLOCK_CTRL_SYS_DIV_M);
234  if(!bExternalOsc32k)
235  {
236  ui32Reg |= SYS_CTRL_CLOCK_CTRL_OSC32K;
237  }
238  ui32Osc = bInternalOsc ? SYS_CTRL_CLOCK_CTRL_OSC : 0;
239  ui32Reg |= ui32Osc;
240  ui32Reg |= ui32SysDiv;
241  HWREG(SYS_CTRL_CLOCK_CTRL) = ui32Reg;
242 
243  //
244  // Note: This wait loop could potentially be removed, if caution
245  // is taken in code running after this call.
246  //
247  // If we have changed Osc settings, wait until change happens
248  //
249  ui32STA = HWREG(SYS_CTRL_CLOCK_STA);
250  ui32TimeoutVal = 0;
251  while((ui32Osc != (ui32STA & SYS_CTRL_CLOCK_CTRL_OSC)) &&
252  (ui32TimeoutVal < SYS_CTRL_TIMEOUT))
253  {
254  SysCtrlDelay(16);
255  ui32STA = HWREG(SYS_CTRL_CLOCK_STA);
256  ui32TimeoutVal++;
257  }
258  ASSERT(ui32TimeoutVal < SYS_CTRL_TIMEOUT);
259 
260 } // SysCtrlClockSet

291  break;

293  ui32Clk = SYS_CTRL_8MHZ;

295  case SYS_CTRL_SYSDIV_4MHZ:
296  ui32Clk = SYS_CTRL_4MHZ;

547 void
548 SysCtrlDeepSleep(void)
549 {
550 #ifndef NO_CLOCK_DIVIDER_RESTORE
551  bool bRestoreSys;
552  bool bRestoreIO;
553  uint32_t ui32Reg;
554 
555  ui32Reg = HWREG(SYS_CTRL_CLOCK_STA);
556  bRestoreSys = (ui32Reg & SYS_CTRL_CLOCK_STA_SYS_DIV_M)==0;
557  bRestoreIO = (ui32Reg & SYS_CTRL_CLOCK_STA_IO_DIV_M)==0;
558  if(bRestoreSys || bRestoreIO)
559  {
560  ui32Reg = HWREG(SYS_CTRL_CLOCK_CTRL);
561  ui32Reg |= bRestoreSys? 0x1:0x0;
562  ui32Reg |= bRestoreIO? 0x100:0x0;
563  HWREG(SYS_CTRL_CLOCK_CTRL) = ui32Reg;
564  }
565 #endif
566 
567  //
568  // Enable deep-sleep.
569  //
570  HWREG(NVIC_SYS_CTRL) |= NVIC_SYS_CTRL_SLEEPDEEP;
571 
572  //
573  // Wait for an interrupt.
574  //
575  CPUwfi();
576 
577  //
578  // Disable deep-sleep so that a future sleep will work correctly.
579  //
580  HWREG(NVIC_SYS_CTRL) &= ~(NVIC_SYS_CTRL_SLEEPDEEP);
581 
582 #ifndef NO_CLOCK_DIVIDER_RESTORE
583  if(bRestoreSys || bRestoreIO)
584  {
585  ui32Reg = HWREG(SYS_CTRL_CLOCK_CTRL);
586  ui32Reg &= bRestoreSys ? ~SYS_CTRL_CLOCK_CTRL_SYS_DIV_M : 0xffffffff;
587  ui32Reg &= bRestoreIO ? ~SYS_CTRL_CLOCK_CTRL_IO_DIV_M : 0xffffffff;
588  HWREG(SYS_CTRL_CLOCK_CTRL) = ui32Reg;
589  }
590 #endif
591 }

616 bool
617 SysCtrlPeripheralPresent(uint32_t ui32Peripheral)
618 {
619  uint32_t ui32DieCfg;
620  uint32_t ui32Mask;
621 
622  // Check the arguments.
623  ASSERT(SysCtrlPeripheralValid(ui32Peripheral));
624 
625  if(SYS_CTRL_PERIPH_INDEX(ui32Peripheral) ==
626  SYS_CTRL_PERIPH_INDEX(SYS_CTRL_PERIPH_PKA) ||
627  SYS_CTRL_PERIPH_INDEX(ui32Peripheral) ==
628  SYS_CTRL_PERIPH_INDEX(SYS_CTRL_PERIPH_AES) ||
629  SYS_CTRL_PERIPH_INDEX(ui32Peripheral) ==
630  SYS_CTRL_PERIPH_INDEX(SYS_CTRL_PERIPH_RFC))
631  {
632  ui32DieCfg = HWREG(FLASH_CTRL_DIECFG2);
633  }
634  else
635  {
636  ui32DieCfg = HWREG(FLASH_CTRL_DIECFG1);
637  }
638 
639  ui32Mask = (g_pui32DieCfgMask[SYS_CTRL_PERIPH_INDEX(ui32Peripheral)] <<
640  SYS_CTRL_PERIPH_INSTANCE(ui32Peripheral));
641 
642  // Mask with correct mask and determine if this peripheral is
643  // permanently disabled.
644  if(ui32DieCfg & ui32Mask)
645  {
646  return(true);
647  }
648  else
649  {
650  return(false);
651  }
652 }

676 void
677 SysCtrlPeripheralReset(uint32_t ui32Peripheral)
678 {
679  volatile uint32_t ui32Delay;
680 
681  // Check the arguments.
682  ASSERT(SysCtrlPeripheralValid(ui32Peripheral));
683  ASSERT(!(ui32Peripheral == SYS_CTRL_PERIPH_RFC));
684 
685  // Put the peripheral into the reset state.
686  HWREG(g_pui32SRRegs[SYS_CTRL_PERIPH_INDEX(ui32Peripheral)]) |=
687  SYS_CTRL_PERIPH_MASKBIT(ui32Peripheral);
688 
689  // Delay for a little bit.
690  for(ui32Delay = 0; ui32Delay < 16; ui32Delay++) { }
691 
692  // Take the peripheral out of the reset state.
693  HWREG(g_pui32SRRegs[SYS_CTRL_PERIPH_INDEX(ui32Peripheral)]) &=
694  ~SYS_CTRL_PERIPH_MASKBIT(ui32Peripheral);
695 }
