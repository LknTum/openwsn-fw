176 void radio_txNow() {
177  PORT_TIMER_WIDTH val;
178  // change state
179  radio_vars.state = RADIOSTATE_TRANSMITTING;
180 
181  // send packet by pulsing the SLP_TR_CNTL pin
182  PORT_PIN_RADIO_SLP_TR_CNTL_HIGH();
183  PORT_PIN_RADIO_SLP_TR_CNTL_LOW();
184 
185  // The AT86RF231 does not generate an interrupt when the radio transmits the
186  // SFD, which messes up the MAC state machine. The danger is that, if we leave
187  // this funtion like this, any radio watchdog timer will expire.
188  // Instead, we cheat an mimick a start of frame event by calling
189  // ieee154e_startOfFrame from here. This also means that software can never catch
190  // a radio glitch by which #radio_txEnable would not be followed by a packet being
191  // transmitted (I've never seen that).
192  if (radio_vars.startFrame_cb!=NULL) {
193  // call the callback
194  val=radiotimer_getCapturedTime();
195  radio_vars.startFrame_cb(val);
196  }
197 }

404 kick_scheduler_t radio_isr() {
405  PORT_TIMER_WIDTH capturedTime;
406  uint8_t irq_status;
407 
408  // capture the time
409  capturedTime = radiotimer_getCapturedTime();
410 
411  // reading IRQ_STATUS causes radio's IRQ pin to go low
412  irq_status = radio_spiReadReg(RG_IRQ_STATUS);
413 
414  // start of frame event
415  if (irq_status & AT_IRQ_RX_START) {
416  // change state
417  radio_vars.state = RADIOSTATE_RECEIVING;
418  if (radio_vars.startFrame_cb!=NULL) {
419  // call the callback
420  radio_vars.startFrame_cb(capturedTime);
421  // kick the OS
422  return KICK_SCHEDULER;
423  } else {
424  while(1);
425  }
426  }
427  // end of frame event
428  if (irq_status & AT_IRQ_TRX_END) {
429  // change state
430  radio_vars.state = RADIOSTATE_TXRX_DONE;
431  if (radio_vars.endFrame_cb!=NULL) {
432  // call the callback
433  radio_vars.endFrame_cb(capturedTime);
434  // kick the OS
435  return KICK_SCHEDULER;
436  } else {
437  while(1);
438  }
439  }
440 
441  return DO_NOT_KICK_SCHEDULER;
