60 //*****************************************************************************
61 //
62 // This command is sent to the boot loader to indicate where
63 // to store data and how many bytes will be sent by the
64 // COMMAND_SEND_DATA commands that follow. The command
65 // consists of two 32-bit values that are both transferred MSB first.
66 // The first 32-bit value is the address to start programming data
67 // into, while the second is the 32-bit size of the data that will be
68 // sent. The Program Size parameter must be a multiple of 4. This
69 // command should be followed by a COMMAND_GET_STATUS to
70 // ensure that the program address and program size were valid
71 // for the boot loader.
72 //
73 // The format of the command is as follows:
74 //
75 // uint8_t ui8Command[9];
76 //
77 // ui8Command[0] = COMMAND_DOWNLOAD;
78 // ui8Command[1] = Program Address [31:24];
79 // ui8Command[2] = Program Address [23:16];
80 // ui8Command[3] = Program Address [15:8];
81 // ui8Command[4] = Program Address [7:0];
82 // ui8Command[5] = Program Size [31:24];
83 // ui8Command[6] = Program Size [23:16];
84 // ui8Command[7] = Program Size [15:8];
85 // ui8Command[8] = Program Size [7:0];
86 //
87 //*****************************************************************************
88 #define COMMAND_DOWNLOAD 0x21

90 //*****************************************************************************
91 //
92 // This command is sent to the boot loader to transfer execution control to the

94 // MSB first, that is the address to which execution control is transferred.
95 //
96 // The format of the command is as follows:
97 //
98 // uint8_t ui8Command[5];
99 //
100 // ui8Command[0] = COMMAND_RUN;
101 // ui8Command[1] = Run Address [31:24];
102 // ui8Command[2] = Run Address [23:16];
103 // ui8Command[3] = Run Address [15:8];
104 // ui8Command[4] = Run Address [7:0];
105 //
106 //*****************************************************************************
107 #define COMMAND_RUN 0x22

109 //*****************************************************************************
110 //
111 // This command returns the status of the last command that was issued.
112 // Typically this command should be received after every command is sent to
113 // ensure that the previous command was successful or, if unsuccessful, to
114 // properly respond to a failure. The command requires one byte in the data of
115 // the packet and the boot loader should respond by sending a packet with one
116 // byte of data that contains the current status code.
117 //
118 // The format of the command is as follows:
119 //
120 // uint8_t ui8Command[1];
121 //
122 // ui8Command[0] = COMMAND_GET_STATUS;
123 //
124 // The following are the definitions for the possible status values that can be
125 // returned from the boot loader when <tt>COMMAND_GET_STATUS</tt> is sent to
126 // the CC2538 device.
127 //
128 // COMMAND_RET_SUCCESS
129 // COMMAND_RET_UNKNOWN_CMD
130 // COMMAND_RET_INVALID_CMD
131 // COMMAND_RET_INVALID_ADD
132 // COMMAND_RET_FLASH_FAIL
133 //
134 //*****************************************************************************
135 #define COMMAND_GET_STATUS 0x23

137 //*****************************************************************************
138 //
139 // This command should only follow a COMMAND_DOWNLOAD command
140 // or another COMMAND_SEND_DATA command, if more data
141 // is needed. Consecutive send data commands automatically increment
142 // the address and continue programming from the previous
143 // location. The transfer size is limited by the maximum size of
144 // a packet, which allows up to 252 data bytes to be transferred at a
145 // time. The command terminates programming once the number
146 // of bytes indicated by the COMMAND_DOWNLOAD command has
147 // been received. Each time this function is called, it should be
148 // followed by a COMMAND_GET_STATUS command to ensure that
149 // the data was successfully programmed into the flash. If the boot
150 // loader sends a NAK to this command, the boot loader will not increment
151 // the current address which allows for retransmission of
152 // the previous data.
153 //
154 // The format of the command is as follows:
155 //
156 // uint8_t ui8Command[9];
157 //
158 // ui8Command[0] = COMMAND_SEND_DATA
159 // ui8Command[1] = Data[0];
160 // ui8Command[2] = Data[1];
161 // ui8Command[3] = Data[2];
162 // ui8Command[4] = Data[3];
163 // ui8Command[5] = Data[4];
164 // ui8Command[6] = Data[5];
165 // ui8Command[7] = Data[6];
166 // ui8Command[8] = Data[7];
167 //
168 //*****************************************************************************
169 #define COMMAND_SEND_DATA 0x24

171 //*****************************************************************************
172 //
173 // This command is used to tell the boot loader to reset. This is used after
174 // downloading a new image to the cc2538 device to cause the new application
175 // or the new boot loader to start from a reset. The normal boot sequence
176 // occurs and the image runs as if from a hardware reset. It can also be used
177 // to reset the boot loader if a critical error occurs and the host device
178 // wants to restart communication with the boot loader.
179 //
180 // The format of the command is as follows:
181 //
182 // uint8_t ui8Command[1];
183 //
184 // ui8Command[0] = COMMAND_RESET;
185 //
186 // The boot loader responds with an ACK signal to the host device before
187 // actually executing the software reset on the cc2538 device running the
188 // boot loader. This informs the updater application that the command was
189 // received successfully and the part will be reset.
190 //
191 //*****************************************************************************
192 #define COMMAND_RESET 0x25

194 //*****************************************************************************
195 //
196 // This command will erase the required flash pages. A single flash
197 // page has the size of 2KB which is the minimum size that can be
198 // erased. The command consists of two 32-bit values that are both
199 // transferred MSB first. The first 32-bit value is the address in a
200 // flash page from where the erase starts, and the second 32-bits
201 // value is the number of bytes comprised by the erase. The Boot
202 // Loader responds with an ACK signal to the host device after the
203 // actual erase operation is performed. The page erase operation
204 // will always start erasing at the start address of the page that
205 // incorporates the input Data Address. The erase will end at the
206 // end of the page which is determined from the input parameters.
207 // On CC2538™ the flash starts at address 0x00200000.
208 //
209 // The format of the command is as follows:
210 //
211 // uint8_t ui8Command[9];
212 //
213 // ui8Command[0] = COMMAND_ERASE;
214 // ui8Command[1] = Data Address [31:24];
215 // ui8Command[2] = Data Address [23:16];
216 // ui8Command[3] = Data Address [15:8];
217 // ui8Command[4] = Data Address [7:0];
218 // ui8Command[5] = Data Size [31:24];
219 // ui8Command[6] = Data Size [23:16];
220 // ui8Command[7] = Data Size [15:8];
221 // ui8Command[8] = Data Size [7:0];
222 //
223 //*****************************************************************************
224 #define COMMAND_ERASE 0x26

226 //*****************************************************************************
227 //
228 // This command is used to check a flash area using CRC32. The command consists
229 // of two 32-bit values that are both transferred MSB first. The first 32-bit
230 // value is the address in flash from where the CRC32 calculation starts and
231 // the second 32-bits value is the number of bytes comprised by the CRC32
232 // calculation. The command will send the ACK in response to the command after
233 // the actual CRC32 calculation.
234 // The result is finally returned as 4 byte in a packet. The Boot Loader will
235 // then wait for an ACK from the host as a confirmation that the packet was
236 // received.
237 //
238 // The format of the command is as follows:
239 //
240 // uint8_t ui8Command[9];
241 //
242 // ui8Command[0]= COMMAND_CRC32;
243 // ui8Command[1]= Data Address [31:24];
244 // ui8Command[2]= Data Address [23:16];
245 // ui8Command[3]= Data Address [15: 8];
246 // ui8Command[4]= Data Address [ 7: 0];
247 // ui8Command[5]= Data Size [31:24];
248 // ui8Command[6]= Data Size [23:16];
249 // ui8Command[7]= Data Size [15: 8];
250 // ui8Command[8]= Data Size [7: 0];w\end{verbatim}
251 //
252 //*****************************************************************************
253 #define COMMAND_CRC32 0x27

256 //
257 // This command will make the Boot Loader to return the value of the Chip ID.
258 // The Boot Loader should first respond by sending the ACK in response to the
259 // command and then send a packet with four bytes of data that contains the
260 // Chip ID value. The two LSB bytes hold the chip part number while the
261 // remaining bytes are reserved for future use and will have the value of 0x00.
262 // The Boot Loader will then wait for an ACK from the host as a confirmation
263 // that the packet was received.
264 //
265 // The format of the command is as follows:
266 //
267 // uint8_t ui8Command[1];
268 // ui8Command[0]= COMMAND_GET_CHIP_ID;
269 //
270 //*****************************************************************************

274 //
275 // This command is used in to switch system clock source from internal
276 // 16 MHz RC oscillator to external 32 MHz crystal oscillator.
277 // This can be used to increase transfer
278 // speed on the boot loader serial interfaces. The device will respond with an
279 // ACK on the command packet before the actual switch to the external 32 MHz
280 // crystal oscillator is done.
281 // If the command is transferred on the UART interface the host must execute a
282 // UART baud rate auto-detection [3.2.3.1.1] in order to establish the baud
283 // rate to be used while the device is running from the external 32 MHz crystal
284 // oscillator. The device must be reset in order to switch back to running from
285 // the internal 16 MHz RC oscillator.
286 //
287 // The format of the command is as follows:
288 //
289 // uint8_t ui8Command[1];
290 //
291 // ui8Command[0]= COMMAND_SET_XOSC;
292 //
293 //*****************************************************************************
294 #define COMMAND_SET_XOSC 0x29

296 //*****************************************************************************
297 //
298 // his command used for reading either an 8-bit or 32-bit word at a specified
299 // address within the device memory map.
300 // The command consists of a 32-bit value that is transferred MSB first
301 // followed by an 8-bit value specifying the read access width. The 32-bit
302 // value is the address within the device memory map to be read and the 8-bit
303 // value specifies the access width of the read operation. Allowed values for
304 // the access width are 1 and 4 (1 = 1 byte / 4 = 4 bytes).
305 // The command will send an ACK in response to the command after the specified
306 // data is read from memory. The read data is finally returned as 4 bytes
307 // (MSB first) in a packet. The Boot Loader will then wait for an ACK from
308 // the host as a confirmation that the packet was received.
309 // For an access width value of 1 the read data will be located in the LSB of
310 // the four bytes returned while the MSB bytes will have the value of 0x00.
311 // If an invalid access width is specified within the command the Boot Loader
312 // will ACK the command and respond with 4 bytes with value 0x00 in a packet.
313 // An eventual following COMMAND_GET_STATUS command will in this case report
314 // COMMAND_RET_INVALID_CMD.
315 //
316 // The format of the command is as follows:
317 //
318 // uint8_t ui8Command[6];
319 //
320 // ui8Command[0]= COMMAND_MEMORY_READ;
321 // ui8Command[1]= Data Address [31:24];
322 // ui8Command[2]= Data Address [23:16];
323 // ui8Command[3]= Data Address [15: 8];
324 // ui8Command[4]= Data Address [ 7: 0];
325 // ui8Command[5]= Read access width [7:0];
326 //
327 //*****************************************************************************
328 #define COMMAND_MEMORY_READ 0x2A
329 
330 //*****************************************************************************
331 //
332 // This command is used for writing either an 8-bit or 32-bit word to a
333 // specified address within the device memory map.
334 // The command consists of two 32-bit values that are transferred MSB first
335 // followed by an 8-bit value specifying the write access width. The first
336 // 32-bit value is the address within the device memory map to be written.
337 // The second 32-bit value is the data to be written. The 8-bit value specifies
338 // the access width of the write operation. Allowed values for the access width
339 // are 1 and 4 (1 = 1 byte / 4 = 4 bytes).
340 // The command will send an ACK in response to the command after the specified
341 // data is written to memory.
342 // For an access width value of 1 the data to be written must be located in the
343 // LSB of the four data bytes in the command.
344 // If an invalid access width is specified within the command the Boot Loader
345 // will ACK the command but it will not perform any memory write operation.
346 // An eventual following COMMAND_GET_STATUS command will in this case
347 // report COMMAND_RET_INVALID_CMD.
348 //
349 // The format of the command is as follows:
350 //
351 // uint8_t ui8Command[10];
352 //
353 // ui8Command[0]= COMMAND_MEMORY_WRITE;
354 // ui8Command[1]= Data Address [31:24];
355 // ui8Command[2]= Data Address [23:16];
356 // ui8Command[3]= Data Address [15: 8];
357 // ui8Command[4]= Data Address [ 7: 0];
358 // ui8Command[5]= Data [31:24]
359 // ui8Command[6]= Data [23:16]
360 // ui8Command[7]= Data [15: 8]
361 // ui8Command[8]= Data [7: 0]
362 // ui8Command[9]= Read access width [7:0];
363 //
364 //*****************************************************************************
365 #define COMMAND_MEMORY_WRITE 0x2B
