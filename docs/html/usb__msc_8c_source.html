182 void USB_Service_Bulk_In(PTR_USB_DEV_EVENT_STRUCT event)
183 {
184  uint_8 event_type;
185 
186  #if IMPLEMENT_QUEUING
187  uint_8 index;
188  uint_8 producer, consumer;
189  USB_ENDPOINTS *usb_ep_data = USB_Desc_Get_Endpoints(event->controller_ID);
190 
191  USB_CLASS_MSC_QUEUE queue;
192  /* map the endpoint num to the index of the endpoint structure */
193  for(index = 0; index < usb_ep_data->count; index++)
194  {
195  if(usb_ep_data->ep[index].ep_num == event->ep_num)
196  break;
197  }
198 
199  producer = g_msc.ep[index].bin_producer;
200  /* if there are no errors de-queue the queue and decrement the no. of
201  transfers left, else send the same data again */
202  g_msc.ep[index].bin_consumer++;
203  consumer = g_msc.ep[index].bin_consumer;
204 
205  if(consumer != producer)
206  {/*if bin is not empty */
207 
208  queue = g_msc.ep[index].queue[consumer%MAX_QUEUE_ELEMS];
209  (void)USB_Class_Send_Data(queue.controller_ID, queue.channel,
210  queue.app_data.data_ptr, queue.app_data.data_size);
211  return;
212  }
213  #endif
214 
215  if(g_transfer_remaining >= event->len)
216  { /* decrement the global count */
217  g_transfer_remaining -= event->len;
218  }
219 
220  /* check if there is need to stall BULK IN ENDPOINT And
221  there isn't any transfer in progress*/
222  if((g_msc.in_stall_flag == TRUE)&&(!g_transfer_remaining))
223  {
224  uint_8 component = (uint_8)(event->ep_num |
225  (uint_8)(event->direction<<COMPONENT_PREPARE_SHIFT));
226  g_msc.in_stall_flag = FALSE; /* clear the flag */
227  g_msc.in_flag = FALSE; /* clear send flag */
228  /* now, stalling the status phase - CASE 5th of THIRTEEN CASES*/
229  (void)_usb_device_set_status(&(event->controller_ID),
230  (uint_8)(component|USB_STATUS_ENDPOINT),(uint_16)USB_STATUS_STALLED);
231  return;
232  }
233 
234  /* If its not a data phase on bulk endpoint */
235  if ((!g_msc.in_flag) && (event->len == MSC_CSW_LENGTH))
236  {
237  USB_memcopy(event->buffer_ptr, &(g_msc.csw_struct), sizeof(CSW));
238  }
239 
240  if(g_msc.in_flag) /* bulk in transaction has occurred before CSW */
241  {
242  if(g_msc.param_callback != NULL)
243  {
244  event_type = (uint_8)(g_transfer_remaining?
245  USB_MSC_DEVICE_READ_REQUEST : USB_APP_SEND_COMPLETE);
246 
247  if(event_type == USB_APP_SEND_COMPLETE)
248  {
249  APP_DATA_STRUCT bulk_in_recv;
250  bulk_in_recv.data_ptr = event->buffer_ptr;
251  bulk_in_recv.data_size = event->len;
252  g_msc.param_callback(event->controller_ID, event_type,
253  (void*)&bulk_in_recv);
254  }
255  else
256  {
257  LBA_APP_STRUCT lba_data;
258  g_current_offset += event->len;
259  lba_data.offset = g_current_offset;
260  lba_data.size = (g_transfer_remaining > MSC_SEND_DATA_BUFF_SIZE) ?
261  MSC_SEND_DATA_BUFF_SIZE : g_transfer_remaining;
262  /* whichever is smaller */
263  lba_data.buff_ptr = g_msc.msc_lba_send_buff;
264 
265  g_msc.param_callback(event->controller_ID, USB_MSC_DEVICE_READ_REQUEST,
266  (void*)&lba_data);
267  (void)USB_MSC_Bulk_Send_Data(event->controller_ID, lba_data.buff_ptr,
268  lba_data.size);
269  }
270  }
271 
272  if(!g_transfer_remaining)
273  { /* marks the end of data phase */
274  g_msc.in_flag = FALSE; /* clear the flag for next CBW */
275  /* Send the command status information */
276  (void)USB_MSC_Bulk_Send_Data(event->controller_ID,
277  (uint_8_ptr)&(g_msc.csw_struct), MSC_CSW_LENGTH);
278  }
279  }
280  else if ((event->len == MSC_CSW_LENGTH) /* CSW is 13 bytes in length */
281  && (g_msc.csw_struct.signature == USB_DCSWSIGNATURE)) /*valid CSW signature*/
282  {
283  /* this flag will now be set on reset or after CSW being sent */
284  g_msc.cbw_valid_flag = TRUE;
285  /* prepare for next CBW */
286  (void)USB_MSC_Bulk_Recv_Data(&(event->controller_ID),
287  (uint_8_ptr)&(g_msc.cbw_struct),MSC_CBW_LENGTH);
288  }
289 }

302 void USB_Service_Bulk_Out(PTR_USB_DEV_EVENT_STRUCT event)
303 {
304  uint_8 event_type;
305  uint_8 error;
306 
307  /* check if there is need to stall BULK OUT ENDPOINT And
308  there isn't any transfer in progress*/
309  if((g_msc.out_stall_flag == TRUE)&&(!g_transfer_remaining))
310  {
311  uint_8 component = (uint_8)(event->ep_num |
312  (event->direction<<COMPONENT_PREPARE_SHIFT));
313  g_msc.out_stall_flag = FALSE; /* clear the flag */
314  g_msc.out_flag = FALSE; /* clear send flag */
315  /* now, stalling the status phase - CASE 5th of THIRTEEN CASES*/
316  (void)_usb_device_set_status(&(event->controller_ID),
317  (uint_8)(component|USB_STATUS_ENDPOINT),
318  (uint_16)USB_STATUS_STALLED);
319  return;
320  }
321 
322  /* If its not a data phase on bulk endpoint */
323  if ((!g_msc.out_flag) && (event->len == MSC_CBW_LENGTH) &&
324  (event->buffer_ptr != (uint_8_ptr)&(g_msc.cbw_struct)) )
325  { /* last check condition saves memcopy overhead in case data was
326  already received in correct buffer */
327  USB_memcopy(event->buffer_ptr, (uint_8_ptr)&(g_msc.cbw_struct),
328  sizeof(CBW));
329  }
330 
331  if(g_msc.out_flag) /* bulk out transaction has occurred after CBW */
332  {
333  if(g_msc.param_callback != NULL)
334  {
335  event_type = (uint_8)(g_transfer_remaining?
336  USB_MSC_DEVICE_WRITE_REQUEST : USB_APP_DATA_RECEIVED);
337 
338  if(event_type == USB_APP_DATA_RECEIVED)
339  {
340  APP_DATA_STRUCT bulk_out_recv;
341  bulk_out_recv.data_ptr = event->buffer_ptr;
342  bulk_out_recv.data_size = event->len;
343  g_msc.param_callback(event->controller_ID, event_type,(void*)&bulk_out_recv);
344  }
345  else
346  {
347  LBA_APP_STRUCT lba_data1;
348  lba_data1.buff_ptr = g_msc.msc_lba_recv_buff;
349  lba_data1.offset = g_current_offset;
350  lba_data1.size = event->len;
351  g_msc.param_callback(event->controller_ID, event_type,(void*)&lba_data1);
352  }
353  }
354 
355  if(g_transfer_remaining >= event->len)
356  { /* decrement the global count */
357  g_transfer_remaining -= event->len;
358 
359  if((g_msc.out_stall_flag == TRUE)&&(!g_transfer_remaining))
360  {
361  uint_8 component = (uint_8)(event->ep_num |
362  (event->direction<<COMPONENT_PREPARE_SHIFT));
363  g_msc.out_stall_flag = FALSE; /* clear the flag */
364  g_msc.out_flag = FALSE; /* clear send flag */
365  /* now, stalling the status phase - CASE 5th of THIRTEEN CASES*/
366  (void)_usb_device_set_status(&(event->controller_ID),
367  (uint_8)(component|USB_STATUS_ENDPOINT),
368  (uint_16)USB_STATUS_STALLED);
369  return;
370  }
371  }
372 
373  if(g_transfer_remaining)
374  { /* Transfers Remaining -- Again invoke recv_data */
375  LBA_APP_STRUCT lba_data0;
376  g_current_offset += event->len;
377  lba_data0.size = (g_transfer_remaining > MSC_RECV_DATA_BUFF_SIZE) ?
378  MSC_RECV_DATA_BUFF_SIZE : g_transfer_remaining; /* whichever is smaller */
379  lba_data0.buff_ptr = g_msc.msc_lba_recv_buff;
380  (void)USB_MSC_Bulk_Recv_Data(&(event->controller_ID),
381  lba_data0.buff_ptr,lba_data0.size);
382  return;
383 
384  }
385  else
386  { /* marks the end of data phase */
387  g_msc.out_flag = FALSE; /* clear the flag for next CBW */
388  /* Send the command status information */
389  (void)USB_MSC_Bulk_Send_Data(event->controller_ID,
390  (uint_8_ptr)&(g_msc.csw_struct), MSC_CSW_LENGTH);
391  }
392  }
393  else if(/* check for valid and meaningful CBW */
394  /* CBW received after device had sent a CSW or after a reset */
395  (g_msc.cbw_valid_flag)
396  /* CBW is 31 bytes in length */
397  && (event->len == MSC_CBW_LENGTH)
398  /* valid CBW signature*/
399  && (g_msc.cbw_struct.signature == USB_DCBWSIGNATURE)
400  /* all reserved bits should be zero*/
401  && (!((g_msc.cbw_struct.lun & 0xF0) || (g_msc.cbw_struct.cb_length & 0xE0)))
402  /* host should send command to valid LUN */
403  && (g_msc.cbw_struct.lun<g_msc.device_info.num_lun_supported)
404  /* valid cbwcb length*/
405  && ((g_msc.cbw_struct.cb_length >= 0x01)&&(g_msc.cbw_struct.cb_length <= 0x10))
406  )
407  {
408  g_msc.re_stall_flag = TRUE;
409  /* A valid CBW was received */
410  g_msc.csw_struct.signature = USB_DCSWSIGNATURE;
411  g_msc.csw_struct.tag = g_msc.cbw_struct.tag;
412  /*this flag will now be set on reset or after CSW being sent */
413  g_msc.cbw_valid_flag = FALSE;
414 
415  USB_BYTE_SWAP4((uint_32)g_msc.cbw_struct.data_length,(uint_32*)&(g_msc.cbw_struct.data_length));
416 
417  /* set flag if device is going to recv data in coming transaction */
418  g_msc.out_flag = (boolean)(( (!(g_msc.cbw_struct.flag & USB_CBW_DIRECTION_BIT))
419  && (g_msc.cbw_struct.data_length))?TRUE:FALSE);
420  /* set flag if send is going to send data in coming transaction */
421  g_msc.in_flag = (boolean)(( (g_msc.cbw_struct.flag & USB_CBW_DIRECTION_BIT)
422  && (g_msc.cbw_struct.data_length))?TRUE:FALSE);
423  /* Process the command */
424  error = process_mass_storage_command(event->controller_ID, &(g_msc.cbw_struct),
425  (uint_32*)&(g_msc.csw_struct.residue), &(g_msc.csw_struct.csw_status));
426 
427  if(error == USBERR_ENDPOINT_STALLED)
428  {
429  if(g_msc.out_flag==TRUE)
430  {
431  g_msc.out_stall_flag = TRUE;
432  g_msc.out_flag = FALSE; /* so as to send status phase */
433  }
434  else if(g_msc.in_flag==TRUE)
435  {
436  g_msc.in_stall_flag = TRUE;
437  }
438  }
439 
440  /* if there is no data phase on bulk endpoints */
441  if(!(g_msc.out_flag || g_msc.in_flag))
442  { /* Send the command status information */
443  (void)USB_MSC_Bulk_Send_Data(event->controller_ID,
444  (uint_8_ptr)&(g_msc.csw_struct), MSC_CSW_LENGTH);
445  }
446  }
447  else /* Invalid/NMreceived */
448  {
449  uint_8 direction;
450  uint_8 ep_num;
451  /* prepare the component to be sent in lower layer with
452  endpoint number and direction*/
453  uint_8 component;
454 
455  direction = USB_RECV;
456  ep_num = BULK_OUT_ENDPOINT;
457  component = (uint_8)(ep_num | (direction<<COMPONENT_PREPARE_SHIFT));
458 
459  /* in case of error Stall endpoint and update the status */
460  (void)_usb_device_set_status(&(event->controller_ID),
461  (uint_8)(component | USB_STATUS_ENDPOINT),
462  (uint_16)USB_STATUS_STALLED);
463 
464  direction = USB_SEND;
465  ep_num = BULK_IN_ENDPOINT;
466  component = (uint_8)(ep_num | (direction << COMPONENT_PREPARE_SHIFT));
467 
468  /* in case of error Stall endpoint and update the status*/
469  (void)_usb_device_set_status(&(event->controller_ID),
470  (uint_8)(component | USB_STATUS_ENDPOINT),
471  (uint_16)USB_STATUS_STALLED);
472  /* make the following CBWs invalid */
473  g_msc.cbw_valid_flag = FALSE;
474  }
475 }

657  (!setup_packet->value)&&(setup_packet->length == 0x0001)&&

676  3) preserve the value of its bulk endpoint STALL conditions

680  (!setup_packet->value)&&(!setup_packet->length) &&

689  while(count < usb_ep_data->count)

692  (USB_EP_STRUCT_PTR) (&usb_ep_data->ep[count]);

701  (void)_usb_device_set_status(&controller_ID,(uint_8)

703  (uint_16)USB_STATUS_IDLE);
704  }
705  count++;
706  }
707  g_msc.out_flag = FALSE;
708  g_msc.in_flag = FALSE;
709  g_msc.out_stall_flag = FALSE;
710  g_msc.in_stall_flag = FALSE;
711  g_msc.cbw_valid_flag = TRUE; /*making the first CBW valid */
712  g_msc.re_stall_flag = FALSE;
713  g_transfer_remaining = 0;
714  }
715  else

717  host expects stall */
718  error = USBERR_INVALID_REQ_TYPE;
719  }
720  break;

722  }

725  USB_REQUEST_CLASS_VENDOR)
