140 SOCADCSingleConfigure(uint32_t ui32Resolution, uint32_t ui32Reference)
141 {
142  uint32_t ui32Reg;
143 
144  //
145  // Check the arguments.
146  //
147  ASSERT((ui32Resolution == SOCADC_7_BIT) ||
148  (ui32Resolution == SOCADC_9_BIT) ||
149  (ui32Resolution == SOCADC_10_BIT) ||
150  (ui32Resolution == SOCADC_12_BIT));
151  ASSERT((ui32Reference == SOCADC_REF_INTERNAL) ||
152  (ui32Reference == SOCADC_REF_EXT_AIN7) ||
153  (ui32Reference == SOCADC_REF_AVDD5) ||
154  (ui32Reference == SOCADC_REF_EXT_AIN67));
155 
156  //
157  // Stop random generator
158  //
159  HWREG(SOC_ADC_ADCCON1) = 0x3c;
160 
161  ui32Reg = HWREG(SOC_ADC_ADCCON3) & ~(SOC_ADC_ADCCON3_EREF_M |
162  SOC_ADC_ADCCON3_EDIV_M);
163  HWREG(SOC_ADC_ADCCON3) = ui32Reg | ui32Resolution | ui32Reference;
164 }
