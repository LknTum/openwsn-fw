72 void
73 WatchdogEnable(uint32_t ui32Interval)
74 {
75  uint32_t ui32Regval;
76  //
77  // Check the arguments.
78  //
79  ASSERT(ui32Interval == WATCHDOG_INTERVAL_32768 ||
80  ui32Interval == WATCHDOG_INTERVAL_8192 ||
81  ui32Interval == WATCHDOG_INTERVAL_512 ||
82  ui32Interval == WATCHDOG_INTERVAL_64);
83 
84  // Disable Timer to set interval
85  HWREG(SMWDTHROSC_WDCTL) &= ~SMWDTHROSC_WDCTL_EN;
86  ui32Regval = HWREG(SMWDTHROSC_WDCTL);
87  ui32Regval &= ~SMWDTHROSC_WDCTL_INT_M;
88  ui32Regval |= ui32Interval;
89  HWREG(SMWDTHROSC_WDCTL) = ui32Regval;
90 
91  //
92  // Enable the watchdog timer module.
93  //
94  ui32Regval = HWREG(SMWDTHROSC_WDCTL);
95  ui32Regval &= ~0x4;
96  ui32Regval |= SMWDTHROSC_WDCTL_EN;
97  HWREG(SMWDTHROSC_WDCTL) = ui32Regval;
98 }

110 void
111 WatchdogClear(void)
112 {
113  uint32_t ui32Reg1;
114  uint32_t ui32Reg2;
115 
116  //
117  // Write 0xA followed by 0x5 to CLR field
118  // (0x5 also clears in timer mode)
119  //
120  ui32Reg1 = HWREG(SMWDTHROSC_WDCTL);
121  ui32Reg1 &= ~SMWDTHROSC_WDCTL_CLR_M;
122  ui32Reg2 = ui32Reg1 | (0x5 << SMWDTHROSC_WDCTL_CLR_S);
123  ui32Reg1 |= 0xa << SMWDTHROSC_WDCTL_CLR_S;
124 
125  //
126  // The following two writes must happen within 0.5 cycle of WDT clock!
127  // for clear to actually happen.
128  // Note: might need to "safe guard" this zone by disabling interrupts.
129  //
130  HWREG(SMWDTHROSC_WDCTL) = ui32Reg1;
131  HWREG(SMWDTHROSC_WDCTL) = ui32Reg2;
132 }
