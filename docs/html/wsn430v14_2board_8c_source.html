34 void board_init(void) {
35  uint8_t delay;
36 
37  // disable watchdog timer
38  WDTCTL = WDTPW + WDTHOLD;
39 
40  //===== clocking
41 
42  DCOCTL = 0; // we are not using the DCO
43  BCSCTL1 = 0; // we are not using the DCO
44  BCSCTL2 = SELM_2 | (SELS | DIVS_3) ; // MCLK=XT2, SMCLK=XT2/8
45 
46  // the MSP detected that the crystal is not running (it's normal, it is
47  // starting). It set the OFIFG flag, causing the MSP430 to switch back to
48  // the DC0. By software, we need to clear that flag, causing the MSP430 to
49  // switch back to using the XT2 as a clocking source, but verify that it
50  // stays cleared. This is explained in detail in Section 4.2.6 "Basic Clock
51  // Module Fail-Safe Operation" of slau049f, pdf page 119.
52 
53  do {
54  IFG1 &= ~OFIFG; // clear OSCFault flag
55  for (delay=0;delay<0xff;delay++) { // busy wait for at least 50us
56  __no_operation();
57  }
58  } while ((IFG1 & OFIFG) != 0); // repeat until OSCFault flag stays cleared
59 
60  //===== pins
61 
62  P3DIR |= 0x01; // [P3.0] radio VREG: output
63  P1DIR |= 0x80; // [P1.7] radio reset: output
64  P1DIR &= ~0x20; // [P1.5] radio SFD: input
65  P1IES &= ~0x20; // [P1.5] radio SFD: low->high
66  P1IFG &= ~0x20; // [P1.5] radio SFD: clear interrupt flag
67  P1IE |= 0x20; // [P1.5] radio SFD: interrupt enabled
