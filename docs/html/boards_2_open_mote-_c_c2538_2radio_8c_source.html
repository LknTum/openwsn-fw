349 void radio_getReceivedFrame(uint8_t* pBufRead,
350  uint8_t* pLenRead,
351  uint8_t maxBufLen,
352  int8_t* pRssi,
353  uint8_t* pLqi,
354  bool* pCrc) {
355  uint8_t crc_corr,i;
356 
357  uint8_t len=0;
358 
359  /* Check the length */
360  len = HWREG(RFCORE_SFR_RFDATA); //first byte is len
361 
362 
363  /* Check for validity */
364  if(len > CC2538_RF_MAX_PACKET_LEN) {
365  /* wrong len */
366  CC2538_RF_CSP_ISFLUSHRX();
367  return;
368  }
369 
370 
371  if(len <= CC2538_RF_MIN_PACKET_LEN) {
372  //too short
373  CC2538_RF_CSP_ISFLUSHRX();
374  return;
375  }
376 
377  //check if this fits to the buffer
378  if(len > maxBufLen) {
379  CC2538_RF_CSP_ISFLUSHRX();
380  return;
381  }
382 
383  // when reading the packet from the RX buffer, you get the following:
384  // - *[1B] length byte
385  // - [0-125B] packet (excluding CRC)
386  // - [1B] RSSI
387  // - *[2B] CRC
388 
389  //skip first byte is len
390  for(i = 0; i < len - 2; i++) {
391  pBufRead[i] = HWREG(RFCORE_SFR_RFDATA);
392  }
393 
394  *pRssi = ((int8_t)(HWREG(RFCORE_SFR_RFDATA)) - RSSI_OFFSET);
395  crc_corr = HWREG(RFCORE_SFR_RFDATA);
396  *pCrc = crc_corr & CRC_BIT_MASK;
397  *pLenRead = len;
398 
399  //flush it
400  CC2538_RF_CSP_ISFLUSHRX();
401 }
