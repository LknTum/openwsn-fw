46 #define UDMA_STAT 0x400FF000 // DMA status The STAT register
47  // returns the status of the uDMA
48  // controller. This register cannot
49  // be read when the uDMA controller
50  // is in the reset state.
51 #define UDMA_CFG 0x400FF004 // DMA configuration The CFG
52  // register controls the
53  // configuration of the uDMA
54  // controller.
55 #define UDMA_CTLBASE 0x400FF008 // DMA channel control base
56  // pointer The CTLBASE register
57  // must be configured so that the
58  // base pointer points to a
59  // location in system memory. The
60  // amount of system memory that
61  // must be assigned to the uDMA
62  // controller depends on the number
63  // of uDMA channels used and
64  // whether the alternate channel
65  // control data structure is used.
66  // See Section 10.2.5 for details
67  // about the Channel Control Table.
68  // The base address must be aligned
69  // on a 1024-byte boundary. This
70  // register cannot be read when the
71  // uDMA controller is in the reset
72  // state.
73 #define UDMA_ALTBASE 0x400FF00C // DMA alternate channel control
74  // base pointer The ALTBASE
75  // register returns the base
76  // address of the alternate channel
77  // control data. This register
78  // removes the necessity for
79  // application software to
80  // calculate the base address of
81  // the alternate channel control
82  // structures. This register cannot
83  // be read when the uDMA controller
84  // is in the reset state.
85 #define UDMA_WAITSTAT 0x400FF010 // DMA channel wait-on-request
86  // status This read-only register
87  // indicates that the uDMA channel
88  // is waiting on a request. A
89  // peripheral can hold off the uDMA
90  // from performing a single request
91  // until the peripheral is ready
92  // for a burst request to enhance
93  // the uDMA performance. The use of
94  // this feature is dependent on the
95  // design of the peripheral and is
96  // not controllable by software in
97  // any way. This register cannot be
98  // read when the uDMA controller is
99  // in the reset state.
100 #define UDMA_SWREQ 0x400FF014 // DMA channel software request
101  // Each bit of the SWREQ register
102  // represents the corresponding
103  // uDMA channel. Setting a bit
104  // generates a request for the
105  // specified uDMA channel.
106 #define UDMA_USEBURSTSET 0x400FF018 // DMA channel useburst set Each
107  // bit of the USEBURSTSET register
108  // represents the corresponding
109  // uDMA channel. Setting a bit
110  // disables the channel single
111  // request input from generating
112  // requests, configuring the
113  // channel to only accept burst
114  // requests. Reading the register
115  // returns the status of USEBURST.
116  // If the amount of data to
117  // transfer is a multiple of the
118  // arbitration (burst) size, the
119  // corresponding SET[n] bit is
120  // cleared after completing the
121  // final transfer. If there are
122  // fewer items remaining to
123  // transfer than the arbitration
124  // (burst) size, the uDMA
125  // controller automatically clears
126  // the corresponding SET[n] bit,
127  // allowing the remaining items to
128  // transfer using single requests.
129  // To resume transfers using burst
130  // requests, the corresponding bit
131  // must be set again. A bit must
132  // not be set if the corresponding
133  // peripheral does not support the
134  // burst request model.
135 #define UDMA_USEBURSTCLR 0x400FF01C // DMA channel useburst clear Each
136  // bit of the USEBURSTCLR register
137  // represents the corresponding
138  // uDMA channel. Setting a bit
139  // clears the corresponding SET[n]
140  // bit in the USEBURSTSET register.
141 #define UDMA_REQMASKSET 0x400FF020 // DMA channel request mask set
142  // Each bit of the REQMASKSET
143  // register represents the
144  // corresponding uDMA channel.
145  // Setting a bit disables uDMA
146  // requests for the channel.
147  // Reading the register returns the
148  // request mask status. When a uDMA
149  // channel request is masked, that
150  // means the peripheral can no
151  // longer request uDMA transfers.
152  // The channel can then be used for
153  // software-initiated transfers.
154 #define UDMA_REQMASKCLR 0x400FF024 // DMA channel request mask clear
155  // Each bit of the REQMASKCLR
156  // register represents the
157  // corresponding uDMA channel.
158  // Setting a bit clears the
159  // corresponding SET[n] bit in the
160  // REQMASKSET register.
161 #define UDMA_ENASET 0x400FF028 // DMA channel enable set Each bit
162  // of the ENASET register
163  // represents the corresponding
164  // uDMA channel. Setting a bit
165  // enables the corresponding uDMA
166  // channel. Reading the register
167  // returns the enable status of the
168  // channels. If a channel is
169  // enabled but the request mask is
170  // set (REQMASKSET), then the
171  // channel can be used for
172  // software-initiated transfers.
173 #define UDMA_ENACLR 0x400FF02C // DMA channel enable clear Each
174  // bit of the ENACLR register
175  // represents the corresponding
176  // uDMA channel. Setting a bit
177  // clears the corresponding SET[n]
178  // bit in the ENASET register.
179 #define UDMA_ALTSET 0x400FF030 // DMA channel primary alternate
180  // set Each bit of the ALTSET
181  // register represents the
182  // corresponding uDMA channel.
183  // Setting a bit configures the
184  // uDMA channel to use the
185  // alternate control data
186  // structure. Reading the register
187  // returns the status of which
188  // control data structure is in use
189  // for the corresponding uDMA
190  // channel.
191 #define UDMA_ALTCLR 0x400FF034 // DMA channel primary alternate
192  // clear Each bit of the ALTCLR
193  // register represents the
194  // corresponding uDMA channel.
195  // Setting a bit clears the
196  // corresponding SET[n] bit in the
197  // ALTSET register.
198 #define UDMA_PRIOSET 0x400FF038 // DMA channel priority set Each
199  // bit of the PRIOSET register
200  // represents the corresponding
201  // uDMA channel. Setting a bit
202  // configures the uDMA channel to
203  // have a high priority level.
204  // Reading the register returns the
205  // status of the channel priority
206  // mask.
207 #define UDMA_PRIOCLR 0x400FF03C // DMA channel priority clear Each
208  // bit of the DMAPRIOCLR register
209  // represents the corresponding
210  // uDMA channel. Setting a bit
211  // clears the corresponding SET[n]
212  // bit in the PRIOSET register.
213 #define UDMA_ERRCLR 0x400FF04C // DMA bus error clear The ERRCLR
214  // register is used to read and
215  // clear the uDMA bus error status.
216  // The error status is set if the
217  // uDMA controller encountered a
218  // bus error while performing a
219  // transfer. If a bus error occurs
220  // on a channel, that channel is
221  // automatically disabled by the
222  // uDMA controller. The other
223  // channels are unaffected.
224 #define UDMA_CHASGN 0x400FF500 // DMA channel assignment Each bit
225  // of the CHASGN register
226  // represents the corresponding
227  // uDMA channel. Setting a bit
228  // selects the secondary channel
229  // assignment as specified in the
230  // section "Channel Assignments"
231 #define UDMA_CHIS 0x400FF504 // DMA channel interrupt status
232  // Each bit of the CHIS register
233  // represents the corresponding
234  // uDMA channel. A bit is set when
235  // that uDMA channel causes a
236  // completion interrupt. The bits
237  // are cleared by writing 1.
238 #define UDMA_CHMAP0 0x400FF510 // DMA channel map select 0 Each
239  // 4-bit field of the CHMAP0
240  // register configures the uDMA
241  // channel assignment as specified
242  // in the uDMA channel assignment
243  // table in the "Channel
244  // Assignments" section.
245 #define UDMA_CHMAP1 0x400FF514 // DMA channel map select 1 Each
246  // 4-bit field of the CHMAP1
247  // register configures the uDMA
248  // channel assignment as specified
249  // in the uDMA channel assignment
250  // table in the "Channel
251  // Assignments" section.
252 #define UDMA_CHMAP2 0x400FF518 // DMA channel map select 2 Each
253  // 4-bit field of the CHMAP2
254  // register configures the uDMA
255  // channel assignment as specified
256  // in the uDMA channel assignment
257  // table in the "Channel
258  // Assignments" section.
259 #define UDMA_CHMAP3 0x400FF51C // DMA channel map select 3 Each
