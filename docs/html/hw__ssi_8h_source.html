46 #define SSI_O_CR0 0x00000000 // The CR0 register contains bit
47  // fields that control various
48  // functions within the SSI module.
49  // Functionality such as protocol
50  // mode, clock rate, and data size
51  // are configured in this register.
52 #define SSI_O_CR1 0x00000004 // The CR1 register contains bit
53  // fields that control various
54  // functions within the SSI module.
55  // Master and slave mode
56  // functionality is controlled by
57  // this register.
58 #define SSI_O_DR 0x00000008 // The DR register is 16 bits
59  // wide. When the SSIDR register is
60  // read, the entry in the receive
61  // FIFO that is pointed to by the
62  // current FIFO read pointer is
63  // accessed. When a data value is
64  // removed by the SSI receive logic
65  // from the incoming data frame, it
66  // is placed into the entry in the
67  // receive FIFO pointed to by the
68  // current FIFO write pointer. When
69  // the DR register is written to,
70  // the entry in the transmit FIFO
71  // that is pointed to by the write
72  // pointer is written to. Data
73  // values are removed from the
74  // transmit FIFO one value at a
75  // time by the transmit logic. Each
76  // data value is loaded into the
77  // transmit serial shifter, then
78  // serially shifted out onto the
79  // SSITx pin at the programmed bit
80  // rate. When a data size of less
81  // than 16 bits is selected, the
82  // user must right-justify data
83  // written to the transmit FIFO.
84  // The transmit logic ignores the
85  // unused bits. Received data less
86  // than 16 bits is automatically
87  // right-justified in the receive
88  // buffer. When the SSI is
89  // programmed for MICROWIRE frame
90  // format, the default size for
91  // transmit data is eight bits (the
92  // most significant byte is
93  // ignored). The receive data size
94  // is controlled by the programmer.
95  // The transmit FIFO and the
96  // receive FIFO are not cleared
97  // even when the SSE bit in the
98  // SSICR1 register is cleared,
99  // allowing the software to fill
100  // the transmit FIFO before
101  // enabling the SSI.
102 #define SSI_O_SR 0x0000000C // The SR register contains bits
103  // that indicate the FIFO fill
104  // status and the SSI busy status.
105 #define SSI_O_CPSR 0x00000010 // The CPSR register specifies the
106  // division factor which is used to
107  // derive the SSIClk from the
108  // system clock. The clock is
109  // further divided by a value from
110  // 1 to 256, which is 1 + SCR. SCR
111  // is programmed in the SSICR0
112  // register. The frequency of the
113  // SSIClk is defined by: SSIClk =
114  // SysClk / (CPSDVSR x (1 + SCR))
115  // The value programmed into this
116  // register must be an even number
117  // between 2 and 254. The
118  // least-significant bit of the
119  // programmed number is hard-coded
120  // to zero. If an odd number is
121  // written to this register, data
122  // read back from this register has
123  // the least-significant bit as
124  // zero.
125 #define SSI_O_IM 0x00000014 // The IM register is the
126  // interrupt mask set or clear
127  // register. It is a read/write
128  // register and all bits are
129  // cleared on reset. On a read,
130  // this register gives the current
131  // value of the mask on the
132  // corresponding interrupt. Setting
133  // a bit sets the mask, preventing
134  // the interrupt from being
135  // signaled to the interrupt
136  // controller. Clearing a bit
137  // clears the corresponding mask,
138  // enabling the interrupt to be
139  // sent to the interrupt
140  // controller.
141 #define SSI_O_RIS 0x00000018 // The RIS register is the raw
142  // interrupt status register. On a
143  // read, this register gives the
144  // current raw status value of the
145  // corresponding interrupt before
146  // masking. A write has no effect.
147 #define SSI_O_MIS 0x0000001C // The MIS register is the masked
148  // interrupt status register. On a
149  // read, this register gives the
150  // current masked status value of
151  // the corresponding interrupt. A
152  // write has no effect.
153 #define SSI_O_ICR 0x00000020 // The ICR register is the
154  // interrupt clear register. On a
155  // write of 1, the corresponding
156  // interrupt is cleared. A write of
157  // 0 has no effect.
158 #define SSI_O_DMACTL 0x00000024 // The DMACTL register is the uDMA
159  // control register.
160 #define SSI_O_CC 0x00000FC8 // SSI clock configuration The CC
161  // register controls the baud clock
162  // and system clocks sources for
163  // the SSI module. Note: If the
164  // PIOSC is used for the SSI baud
165  // clock, the system clock
166  // frequency must be at least 16
167  // MHz in run mode.
