73  RCC_RTCCLKConfig(RCC_RTCCLKSource_LSE); //Select the RTC Clock Source
74  RCC_RTCCLKCmd(ENABLE); //enable RTC
75

78  RTC_WaitForSynchro();
79 
80  RTC_SetPrescaler(1); //use 16KHz clock

82

84  RTC_SetCounter(0);
85  RTC_WaitForLastTask();
86

88  RTC_SetAlarm(period);
89  RTC_WaitForLastTask();

92 
93  //interrupt when reach alarm value
94  RTC_ClearFlag(RTC_IT_ALR);

96

99

101  EXTI_ClearITPendingBit(EXTI_Line17);

194 kick_scheduler_t radiotimer_isr() {
195  uint8_t taiv_temp = radiotimer_vars.overflowORcompare;
196  switch (taiv_temp) {
197  case RADIOTIMER_COMPARE:
198  if (radiotimer_vars.compare_cb!=NULL) {
199  RCC_Wakeup();
200  // call the callback
201  radiotimer_vars.compare_cb();
202  // kick the OS
203  return KICK_SCHEDULER;
204  }
205  break;
206  case RADIOTIMER_OVERFLOW: // timer overflows
207  if (radiotimer_vars.overflow_cb!=NULL) {
208 
209  //Wait until last write operation on RTC registers has finished
210  RTC_WaitForLastTask();
211 
212  //Set the RTC time counter to 0
213  RTC_SetCounter(0x00000000);
214  RTC_WaitForLastTask();
215  RCC_Wakeup();
216  // call the callback
217  radiotimer_vars.overflow_cb();
218  // kick the OS
219  return KICK_SCHEDULER;
220  }
221  break;
222  case RADIOTIMER_NONE: // this should not happen
223  default:
224  while(1); // this should not happen
225  }
226  return DO_NOT_KICK_SCHEDULER;
227 }
