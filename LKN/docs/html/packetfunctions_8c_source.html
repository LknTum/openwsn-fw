383 void packetfunctions_calculateChecksum(OpenQueueEntry_t* msg, uint8_t* checksum_ptr) {
384  uint8_t temp_checksum[2];
385  uint8_t little_helper[2];
386 
387  // initialize running checksum
388  temp_checksum[0] = 0;
389  temp_checksum[1] = 0;
390 
391  //===== IPv6 pseudo header
392 
393  // source address (prefix and EUI64)
394  onesComplementSum(temp_checksum,(idmanager_getMyID(ADDR_PREFIX))->prefix,8);
395  onesComplementSum(temp_checksum,(idmanager_getMyID(ADDR_64B))->addr_64b,8);

404 
405  // next header
406  little_helper[0] = 0;
407  little_helper[1] = msg->l4_protocol;
408  onesComplementSum(temp_checksum,little_helper,2);
409 
410  //===== payload
411 
412  // reset the checksum currently in the payload
413  *checksum_ptr = 0;
414  *(checksum_ptr+1) = 0;
415 
416  onesComplementSum(temp_checksum,msg->payload,msg->length);
417  temp_checksum[0] ^= 0xFF;
418  temp_checksum[1] ^= 0xFF;
419 
420  //write in packet
421  *checksum_ptr = temp_checksum[0];
422  *(checksum_ptr+1) = temp_checksum[1];

426 void onesComplementSum(uint8_t* global_sum, uint8_t* ptr, int length) {
427  uint32_t sum = 0xFFFF & (global_sum[0]<<8 | global_sum[1]);
428  while (length>1) {
429  sum += 0xFFFF & (*ptr<<8 | *(ptr+1));
430  ptr += 2;
431  length -= 2;
432  }
433  if (length) {
434  sum += (0xFF & *ptr)<<8;
435  }
436  while (sum>>16) {
437  sum = (sum & 0xFFFF)+(sum >> 16);
438  }
439  global_sum[0] = (sum>>8) & 0xFF;
440  global_sum[1] = sum & 0xFF;
441 }
