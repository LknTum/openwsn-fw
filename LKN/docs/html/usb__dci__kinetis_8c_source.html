200 static void USB_Bus_Reset_Handler (void)
201 {
202 #if !HIGH_SPEED_DEVICE
203  USB0_ERRSTAT = ERR_STAT_CLEAR_ALL; /* clear USB error flag */
204  USB0_CTL |= USB_CTL_ODDRST_MASK; /* Reset to Even buffer */
205  USB0_ADDR = 0; /* reset to default address */
206  /* Select System Clock and Disable Weak Pull Downs */
207  USB0_USBCTRL = 0x00;
208 
209  /* Clear bdt elem structure */
210  Clear_Mem((uint_8_ptr)(g_bdt_elem),
211  (sizeof(BDT_ELEM) * (MAX_BDT_INDEX >> 1)),
212  (uint_8)UNINITIALISED_VAL);
213 
214  /* Clear Memory for BDT and buffer Data */
215  Clear_Mem((uint_8_ptr)g_bdtmap,(uint_32) BYTES_1024, (uint_8)0);
216 
217  /* Initialize BDT buffer address */
218  g_bdt_address = ((uint_32)g_bdtmap + BYTES_512);
219 
220  g_trf_direction = USB_TRF_UNKNOWN;
221 
222  USB0_CTL &= ~USB_CTL_ODDRST_MASK;
223  USB0_USBTRC0 |= 0x40; /* attach CFv1 core to USB bus */
224 
225  USB0_ERREN = ERR_ENB_ENABLE_ALL; /* Enable All Error Interrupts */
226  USB0_INTEN = INTENB_BUS_RESET_VAL; /* Enable All Interrupts except RESUME */
227  USB0_CTL &= ~USB_CTL_TXSUSPENDTOKENBUSY_MASK;
228 
229 #else
230  uint_32 reg;
231 
232  //printf_info("usb0 device get bus reset\n");
233 
234  //Clear the device address
235  USBHS_DEVICEADDR &= ~USBHS_DEVICEADDR_USBADR_MASK;
236 
237  // 1. Clear all setup token semaphores
238  reg = USBHS_EPSETUPSR;
239  USBHS_EPSETUPSR = reg;
240 
241  // 2. Clear all the endpoint complete status bits
242  reg = USBHS_EPCOMPLETE;
243  USBHS_EPCOMPLETE = reg;
244 
245  // 3. Cancel all primed status
246  while(USBHS_EPPRIME);
247  USBHS_EPFLUSH = 0xffffffff;
248 
249  // 4. If reset bit is not cleared at this point force reset device
250  if(!(USBHS_PORTSC1 & USBHS_PORTSC1_PR_MASK)){
251 #ifdef SERIAL_DEBUG
252  printf("Reset took longer than expected -> forced reset\n");
253 #endif
254  USBHS_USBCMD |= USBHS_USBCMD_RST_MASK;
255  }
256 
257  // 5. Free(reset) all allocated dTDs
258  memset(g_usbd_td_buf, 0, TOTAL_QTD_SIZE);
259  memset(g_usbd_td_flag,
260  0,
261  MAX_ENDPOINT_NUMBER * 2 * MAX_DTDS_PER_EP * sizeof(g_usbd_td_flag)/sizeof(*g_usbd_td_flag)
262  );
263 
264  g_trf_direction = USB_TRF_UNKNOWN;
265 #endif
266 }

408 uint_8 USB_DCI_Init (
409  uint_8 controller_ID, /* [IN] Controller ID */
410  uint_8 bVregEn /* Enables or disables internal regulator */
411 )
412 {
413 #if !HIGH_SPEED_DEVICE
414  /* Select System Clock and Disable Weak Pull Downs */
415  USB0_USBCTRL = 0x00;
416 
417  /* save the controller_ID for future use */
418  g_dci_controller_Id = controller_ID;
419 
420  /* Clear bdt elem structure */
421  Clear_Mem((uint_8_ptr)(g_bdt_elem),
422  (sizeof(BDT_ELEM) * (MAX_BDT_INDEX >> 1)),
423  (uint_8)UNINITIALISED_VAL);
424  g_bdtmap = (BDTMAP *)((uint_32)g_Mem);
425 
426  /* Clear Memory for BDT and buffer Data */
427  Clear_Mem((uint_8_ptr)g_bdtmap,(uint_32) BYTES_1024, (uint_8)0);
428 
429  #ifndef OTG_BUILD
430  /* Hard Reset to the USB Module */
431  USB0_USBTRC0 |= USB_USBTRC0_USBRESET_MASK;
432 
433  /* loop till the Reset bit clears */
434  while((USB0_USBTRC0 & USB_USBTRC0_USBRESET_MASK))
435  {
436  };
437 
438  #endif
439 
440  g_trf_direction = USB_TRF_UNKNOWN;
441 
442  /* Asynchronous Resume Interrupt Enable */
443  USB0_USBTRC0 |= 0x40;
444 
445  if(bVregEn)
446  {
447  SIM_SOPT1 |= SIM_SOPT1_USBREGEN_MASK; // enable usb voltage regulator
448  }
449  else
450  {
451  SIM_SOPT1 &= ~SIM_SOPT1_USBREGEN_MASK; // disable usb voltage regulator
452  }
453 
454  /* Set the BDT Table address, Need to be on 512 byte boundary */
455  /* D8 Bit is masked in BDT_PAGE_01 */
456  USB0_BDTPAGE1 = (uint_8)(((uint_32)g_bdtmap >> 8)& 0xFE);
457  USB0_BDTPAGE2 = (uint_8)((uint_32)g_bdtmap >> 16);
458  USB0_BDTPAGE3 = (uint_8)((uint_32)g_bdtmap >> 24);
459 
460  /* Initialized BDT buffer address */
461  g_bdt_address = ((uint_32)g_bdtmap + BYTES_512);
462 
463  #ifndef OTG_BUILD
464  /* Pull Up configuration */
465  USB0_CONTROL = USB_CONTROL_DPPULLUPNONOTG_MASK;
466  #endif
467 
468  USB0_CTL = USB_CTL_USBENSOFEN_MASK; /* Enable USB module */
469  USB0_ISTAT = INT_STAT_CLEAR_ALL; /* Clear USB interrupts*/
470 
471  /* Remove suspend state */
472  USB0_USBCTRL &= ~USB_USBCTRL_SUSP_MASK;
473 
474  /* Enable USB RESET Interrupt */
475  USB0_INTEN |= USB_INTEN_USBRSTEN_MASK;
476 
477  /* Enable USB Sleep Interrupt */
478  USB0_INTEN |= USB_INTEN_SLEEPEN_MASK;
479 
480  USB0_OTGCTL = USB_OTGCTL_DPHIGH_MASK | USB_OTGCTL_OTGEN_MASK;
481 #else // HIGH_SPEED_DEVICE
482  uint_32 reg;
483  uint_32 count = 10000;
484 
485  /* save the controller_ID for future use */
486  g_dci_controller_Id = controller_ID;
487  memset(g_usbd_qh_buf, 0, TOTAL_QHD_SIZE);
488  memset(g_usbd_td_buf, 0, TOTAL_QTD_SIZE);
489 
490  // initialize module in device mode
491  uint_8 status;
492  status = K70_ULPI_SetDeviceMode(controller_ID);
493  if (status != USB_OK)
494  return status;
495 
496  // initialize endpoint 0
497  usbd_ep_qh_init(controller_ID, EP0, IN, 64, 0, 0, 1);
498  usbd_ep_qh_init(controller_ID, EP0, OUT, 64, 0, 0, 1);
499 
500  // setup interrupts
501  USBHS_USBINTR = USBHS_USBINTR_UE_MASK // USB interrupt
502  | USBHS_USBINTR_UEE_MASK // USB error
503  | USBHS_USBINTR_PCE_MASK // Port change
504  | USBHS_USBINTR_URE_MASK // Reset enable
505  | USBHS_USBINTR_SRE_MASK // SOF received
506  | USBHS_USBINTR_SLE_MASK // suspend received
507  | USBHS_USBINTR_SEE_MASK; // System error
508 
509  // enable pullup on DP and initiate attach event
510  USBHS_USBCMD |= USBHS_USBCMD_RS_MASK;
511 #endif
512  return USB_OK;
513 }

578 uint_8 USB_DCI_Init_EndPoint (
579  uint_8 controller_ID,/* [IN] Controller ID */
580  USB_EP_STRUCT_PTR ep_ptr, /* [IN] Pointer to Endpoint structure,
581  (endpoint number,
582  endpoint type,
583  endpoint direction,
584  max packet size) */
585  boolean flag /* [IN] Zero Termination */
586 )
587 {
588 #if !HIGH_SPEED_DEVICE
589  uint_8 bdtmap_index;
590  uint_8 bdtelem_index;
591  uint_8 ep_num = ep_ptr->ep_num;
592  uint_8 direction = ep_ptr->direction;
593  uint_32 ep_ctrl[2] = {EP_OUT, EP_IN};
594  P_BUFF_DSC temp;
595  P_BDT_ELEM bdt_elem;
596 
597  /* if the max packet size is greater than the max buffer size */
598  if(ep_ptr->size > MAX_EP_BUFFER_SIZE)
599  {
600  ep_ptr->size = MAX_EP_BUFFER_SIZE;
601  }
602 
603  /* Get the bdt index correspoding to the endpoint */
604  bdtmap_index = USB_DCI_Get_BDT_Index(ep_num, direction,
605  USB_RAM_EVEN_BUFFER);
606  bdtelem_index = (uint_8)TRANSFER_INDEX(bdtmap_index);
607 
608  /*
609  incase the bdtmap_index is invalid
610  or already initialised return with an error
611  */
612  if((bdtmap_index == INVALID_BDT_INDEX) ||
613  (g_bdt_elem[bdtelem_index].len != (uint_16)UNINITIALISED_VAL) ||
614  (g_bdt_address >= ((uint_32)g_Mem + BYTES_1024)) ||
615  (ep_ptr->type > USB_INTERRUPT_PIPE) ||
616  (ep_ptr->direction > USB_SEND))
617  {
618  return USBERR_EP_INIT_FAILED;
619  }
620 
621  bdt_elem = &g_bdt_elem[bdtelem_index];
622  /* Reset Handler resets bdtmap_index to UNINITIALISED_VAL */
623  if(bdt_elem->bdtmap_index == (uint_8)UNINITIALISED_VAL)
624  {
625  bdt_elem->bdtmap_index = 0;
626  }
627 
628  /* update bdt element structure */
629  bdt_elem->len = (uint_16)ep_ptr->size;
630  bdt_elem->flag = flag;
631  /* preserving even/odd buffer bit */
632  bdt_elem->bdtmap_index &= 0x01;
633  bdt_elem->bdtmap_index |= ((direction << 1) | (ep_num << 2));
634  bdt_elem->addr = g_bdt_address;
635  bdt_elem->type = ep_ptr->type;
636  bdt_elem->direction = direction;
637 
638  temp = &g_bdtmap->ep_dsc[bdt_elem->bdtmap_index];
639 
640  /* Update BDTMAP for endpoint's EVEN Buffer */
641  temp->cnt = SWAP16((uint_16)ep_ptr->size);
642  temp->addr = SWAP32(g_bdt_address);
643  temp->Stat._byte = (_CPU | _DATA0 | _DTS);
644 
645  /* Update BDTMAP for endpoint's ODD Buffer */
646  temp = &g_bdtmap->ep_dsc[(uint8_t)((bdt_elem->bdtmap_index) ^ 1)];
647 
648  temp->cnt = SWAP16((uint_16)ep_ptr->size);
649  temp->addr = SWAP32(g_bdt_address);
650  temp->Stat._byte = (_CPU | _DATA1 | _DTS);
651 
652  /* update the buffer address for the next endpoint */
653  g_bdt_address += ep_ptr->size;
654 
655  if(direction == USB_RECV)
656  {
657  /*
658  For Recv Endpoints
659  Give SIE Control to DATA0
660  */
661  temp = &g_bdtmap->ep_dsc[bdt_elem->bdtmap_index];
662  temp->Stat._byte |= _SIE;
663  }
664 
665  /* enable handshake for Non-Isochronous Endpoints */
666  ep_ctrl[direction] |= ((ep_ptr->type != USB_ISOCHRONOUS_PIPE) ?
667  HSHK_EN:0x00);
668  /* set the EndPoint Control MCU Register*/
669  *((&USB0_ENDPT0) + (4 * ep_num)) |= ep_ctrl[direction];
670 #else // HIGH_SPEED_DEVICE
671  //printf_info("%s, ep_num is %d, dir is %d, type is %d\n", __func__, ep_ptr->ep_num, ep_ptr->direction, ep_ptr->type);
672 
673  unsigned char mult;
674 
675  // No need to initialize EP0
676  if (ep_ptr->ep_num == CONTROL_ENDPOINT)
677  return USB_OK;
678 
679  switch (ep_ptr->type & 0x3) {
680  case EP_TRANSFER_TYPE_CONTROL:
681  case EP_TRANSFER_TYPE_BULK:
682  case EP_TRANSFER_TYPE_INTERRUPT:
683  mult = 0;
684  break;
685  case EP_TRANSFER_TYPE_ISOCHRONOUS:
686  /* Calculate the ISO transfer High-Bandwidth Pipe Multiplier
687  * The ISO endpoints, must set Mult 1, 2, 3 for high speed
688  * and only 1 for full speed
689  */
690 #ifdef ULPI_FORCE_FULLSPEED
691  mult = 1;
692 #else
693  mult = (unsigned char)(1 + (((ep_ptr->size) >> 11) & 0x03));
694 #endif
695  }
696 
697  // setup dQH
698  usbd_ep_qh_init(controller_ID, ep_ptr->ep_num, ep_ptr->direction, ep_ptr->size, flag, mult, 0xDEAD0001);
699 
700  // enable endpoint
701  usbd_ep_setup(controller_ID, ep_ptr->ep_num, ep_ptr->direction, ep_ptr->type);
702 
703  return USB_OK;
704 #endif
705  return USB_OK;

2695 // Add TD to TD list and prime endpoint based on this algorithm:
2696 // Appendix
2697 // 5.5.3 Executing A Transfer Descriptor
2698 // To safely add a dTD, the DCD must be follow this procedure which will handle the event where the device
2699 // controller reaches the end of the dTD list at the same time a new dTD is being added to the end of the list.
2700 // Determine whether the link list is empty:
2701 // Check DCD driver to see if pipe is empty (internal representation of linked-list should indicate if any packets
2702 // are outstanding).
2703 // Case 1: Link list is empty
2704 // 1. Write dQH next pointer AND dQH terminate bit to 0 as a single DWord operation.
2705 // 2. Clear active & halt bit in dQH (in case set from a previous error).
2706 // 3. Prime endpoint by writing '1' to correct bit position in ENDPTPRIME.
2707 // Case 2: Link list is not empty
2708 // 1. Add dTD to end of linked list.
2709 // 2. Read correct prime bit in ENDPTPRIME - if '1' DONE.
2710 // 3. Set ATDTW bit in USBCMD register to '1'.
2711 // 4. Read correct status bit in ENDPTPRIME. (store in tmp. variable for later) [[this should be ENDPTSTATUS, not ENDPTPRIME]}
2712 // 5. Read ATDTW bit in USBCMD register.
2713 // If '0' goto 3.
2714 // If '1' continue to 6.
2715 // 6. Write ATDTW bit in USBCMD register to '0'.
2716 // 7. If status bit read in (4) is '1' DONE.
2717 // 8. If status bit read in (4) is '0' then Goto Case 1: Step 1.
2718 //
2719 static void usbd_add_td(uint_8 controller_ID, unsigned char ep_num, unsigned char direction, struct dtd_t *td)
2720 {
2721  // Get the index into the TD list for this endpoint + direction
2722  int td_index = (ep_num * 2) + direction;
2723 
2724  if (g_usbd_qh_tail[td_index] == NULL)
2725  {
2726  // Case 1: Link list is empty
2727 
2728  usbd_prime_ep(controller_ID, ep_num, direction, td);
2729  }
2730  else
2731  {
2732  // Case 2: Link list is not empty
2733 
2734  unsigned int ep_mask = (direction == OUT ? EPOUT_PRIME : EPIN_PRIME);
2735 
2736  // Add TD to tail next_link_ptr
2737  // Clear Terminate bit to indicate pointer is valid
2738  g_usbd_qh_tail[td_index]->dtd_word0 = td->dtd_base & 0xFFFFFFE0;
2739 
2740  // If EP is already primed, we are done
2741  if (!(USBHS_EPPRIME & (ep_mask << ep_num)))
2742  {
2743  // EP not primed, check if it is active
2744  unsigned int ep_status = 0;
2745  unsigned int temp;
2746 
2747  // Use Add dTD Tripwire to properly read endpoint status register
2748  do
2749  {
2750  /* write '1' to Add Tripwire (ATDTW) in USBCMD register */
2751  temp = USBHS_USBCMD;
2752  temp |= (0x1 << USBHS_USBCMD_ATDTW_SHIFT);
2753  USBHS_USBCMD = temp;
2754 
2755  // Read endpoint status
2756  ep_status = USBHS_EPSR & (ep_mask << ep_num);
2757 
2758  } while (!USBHS_USBCMD & (0x1 << USBHS_USBCMD_ATDTW_SHIFT));
2759 
2760  /* write '0' to Add Tripwire (ATDTW) in USBCMD register */
2761  temp = USBHS_USBCMD;
2762  temp &= ~(0x1 << USBHS_USBCMD_ATDTW_SHIFT);
2763  USBHS_USBCMD = temp;

2770  }
2771  }

2775 }
