126 void bsp_timer_scheduleIn(PORT_TIMER_WIDTH delayTicks)
127 {
128  PORT_TIMER_WIDTH newCompareValue;

136  bsp_timer_vars.last_compare_value = newCompareValue;

138  if (delayTicks < (TIM_GetCounter(TIM2)-temp_last_compare_value))

140  // setting the interrupt flag triggers an interrupt

142  TIM_GenerateEvent(TIM2,TIM_EventSource_CC1);

144  else
