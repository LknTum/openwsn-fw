507 void neighbors_updateMyDAGrankAndNeighborPreference() {
508  uint8_t i;
509  uint16_t rankIncrease;
510  uint32_t tentativeDAGrank; // 32-bit since is used to sum
511  uint8_t prefParentIdx;
512  bool prefParentFound;
513  uint32_t rankIncreaseIntermediary; // stores intermediary results of rankIncrease calculation
514 
515  // if I'm a DAGroot, my DAGrank is always MINHOPRANKINCREASE
516  if ((idmanager_getIsDAGroot())==TRUE) {
517  // the dagrank is not set through setting command, set rank to MINHOPRANKINCREASE here
518  neighbors_vars.myDAGrank=MINHOPRANKINCREASE;
519  return;
520  }
521 
522  // reset my DAG rank to max value. May be lowered below.
523  neighbors_vars.myDAGrank = MAXDAGRANK;
524 
525  // by default, I haven't found a preferred parent
526  prefParentFound = FALSE;
527  prefParentIdx = 0;
528 
529  // loop through neighbor table, update myDAGrank
530  for (i=0;i<MAXNUMNEIGHBORS;i++) {
531  if (neighbors_vars.neighbors[i].used==TRUE) {
532 
533  // reset parent preference
534  neighbors_vars.neighbors[i].parentPreference=0;
535 
536  // calculate link cost to this neighbor
537  if (neighbors_vars.neighbors[i].numTxACK==0) {
538  rankIncrease = DEFAULTLINKCOST*2*MINHOPRANKINCREASE;
539  } else {
540  //6TiSCH minimal draft using OF0 for rank computation
541  rankIncreaseIntermediary = (((uint32_t)neighbors_vars.neighbors[i].numTx) << 10);
542  rankIncreaseIntermediary = (rankIncreaseIntermediary * 2 * MINHOPRANKINCREASE) / ((uint32_t)neighbors_vars.neighbors[i].numTxACK);
543  rankIncrease = (uint16_t)(rankIncreaseIntermediary >> 10);
544  }
545 
546  tentativeDAGrank = neighbors_vars.neighbors[i].DAGrank+rankIncrease;
547  if ( tentativeDAGrank<neighbors_vars.myDAGrank &&
548  tentativeDAGrank<MAXDAGRANK) {
549  // found better parent, lower my DAGrank
550  neighbors_vars.myDAGrank = tentativeDAGrank;
551  prefParentFound = TRUE;
552  prefParentIdx = i;
553  }
554  }
555  }
556 
557  // update preferred parent
558  if (prefParentFound) {
559  neighbors_vars.neighbors[prefParentIdx].parentPreference = MAXPREFERENCE;
560  neighbors_vars.neighbors[prefParentIdx].stableNeighbor = TRUE;
561  neighbors_vars.neighbors[prefParentIdx].switchStabilityCounter = 0;
562  }
