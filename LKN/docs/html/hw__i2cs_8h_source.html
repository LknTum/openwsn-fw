46 #define I2CS_OAR 0x40020800 // I2C slave own address This
47  // register consists of seven
48  // address bits that identify the
49  // CC2538 I2C device on the I2C
50  // bus.
51 #define I2CS_STAT 0x40020804 // I2C slave control and status
52  // This register functions as a
53  // control register when written,
54  // and a status register when read.
55 #define I2CS_CTRL 0x40020804 // I2C slave control and status
56  // This register functions as a
57  // control register when written,
58  // and a status register when read.
59 #define I2CS_DR 0x40020808 // I2C slave data This register
60  // contains the data to be
61  // transmitted when in the slave
62  // transmit state, and the data
63  // received when in the slave
64  // receive state.
65 #define I2CS_IMR 0x4002080C // I2C slave interrupt mask This
66  // register controls whether a raw
67  // interrupt is promoted to a
68  // controller interrupt.

70  // This register specifies whether

74  // whether an interrupt was
