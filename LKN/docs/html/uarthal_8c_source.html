268 void
269 UARTConfigSetExpClk(uint32_t ui32Base, uint32_t ui32UARTClk,
270  uint32_t ui32Baud, uint32_t ui32Config)
271 {
272  uint32_t ui32Div;
273 
274  //
275  // Check the arguments.
276  //
277  ASSERT(UARTBaseValid(ui32Base));
278  ASSERT(ui32Baud != 0);
279 
280  //
281  // Stop the UART.
282  //
283  UARTDisable(ui32Base);
284 
285  //
286  // Is the required baud rate greater than the maximum rate supported
287  // without the use of high speed mode?
288  //
289  if((ui32Baud * 16) > ui32UARTClk)
290  {
291  //
292  // Enable high speed mode.
293  //
294  HWREG(ui32Base + UART_O_CTL) |= UART_CTL_HSE;
295 
296  //
297  // Half the supplied baud rate to compensate for enabling high speed
298  // mode. This allows the following code to be common to both cases.
299  //
300  ui32Baud /= 2;
301  }
302  else
303  {
304  //
305  // Disable high speed mode.
306  //
307  HWREG(ui32Base + UART_O_CTL) &= ~(UART_CTL_HSE);
308  }
309 
310  //
311  // Compute the fractional baud rate divider.
312  //
313  ui32Div = (((ui32UARTClk * 8) / ui32Baud) + 1) / 2;
314 
315  //
316  // Set the baud rate.
317  //
318  HWREG(ui32Base + UART_O_IBRD) = ui32Div / 64;
319  HWREG(ui32Base + UART_O_FBRD) = ui32Div % 64;
320 
321  //
322  // Set parity, data length, and number of stop bits.
323  //
324  HWREG(ui32Base + UART_O_LCRH) = ui32Config;
325 
326  //
327  // Clear the flags register.
328  //
329  HWREG(ui32Base + UART_O_FR) = 0;
330 }

364 UARTConfigGetExpClk(uint32_t ui32Base, uint32_t ui32UARTClk,
365  uint32_t *pui32Baud, uint32_t *pui32Config)
366 {
367  uint32_t ui32Int, ui32Frac;

370  // Check the arguments.

374  //

376  //

378  ui32Frac = HWREG(ui32Base + UART_O_FBRD);
379  *pui32Baud = (ui32UARTClk * 4) / ((64 * ui32Int) + ui32Frac);
380 
381  //

383  //
384  if(HWREG(ui32Base + UART_O_CTL) & UART_CTL_HSE)

386  //

388  // double what was just calculated.
389  //
390  *pui32Baud *= 2;
391  }
392 
393  //
394  // Get the parity, data length, and number of stop bits.
395  //
396  *pui32Config = (HWREG(ui32Base + UART_O_LCRH) &
397  (UART_LCRH_SPS | UART_LCRH_WLEN_M | UART_LCRH_STP2 |
398  UART_LCRH_EPS | UART_LCRH_PEN));
