207 void
208 SleepModeTimerCompareSet(uint32_t ui32Compare)
209 {
210  //
211  // Wait for ST0, ST3 regs to be ready for writing
212  //
213  while(!(HWREG(SMWDTHROSC_STLOAD) & SMWDTHROSC_STLOAD_STLOAD))
214  {
215  }
216 
217  HWREG(SMWDTHROSC_ST3) = (ui32Compare >> 24) & 0x000000ff;
218  HWREG(SMWDTHROSC_ST2) = (ui32Compare >> 16) & 0x000000ff;
219  HWREG(SMWDTHROSC_ST1) = (ui32Compare >> 8) & 0x000000ff;
220  HWREG(SMWDTHROSC_ST0) = ui32Compare & 0x000000ff;
221 }
