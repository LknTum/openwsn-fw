309 static bool private_spi_findBaudRate(uint32_t clk_div, uint8_t *p_BR, uint8_t *p_PBR, uint8_t *p_DBR){
310  bool div_found; /* Acceptable values found */
311  uint32_t best_distance; /* Distance between clk_div and calculated div */
312  uint8_t test_BR_idx; /* Auxiliar var to calculate the BR */
313  uint8_t test_PBR_idx; /* Auxiliar var to calculate the PBR */
314  uint8_t test_DBR; /* Auxiliar var to calculate the DBR */
315  uint32_t test_distance;
316  uint32_t test_div;
317  uint16_t BR_values[16] = {2, 4, 6, 8, 16, 32, 64, 128, /* Availabrle values for BR */
318  256, 512, 1024, 2048, 4096,
319  8192, 16384, 32768};
320  uint8_t PBR_values[4] = {2, 3, 5, 7}; /* Availabrle values for PBR */
321 
322  /* Try out all the Baud Rates and keep best fit */
323  div_found = FALSE;
324  best_distance = clk_div + 1; /* Initialize distance beyound maximum value */
325  for(test_BR_idx = 0u; test_BR_idx < 16; test_BR_idx++){
326  for(test_PBR_idx = 0u; test_PBR_idx < 4; test_PBR_idx++){
327  for(test_DBR = 0u; test_DBR < 2; test_DBR++){
328 
329  test_div = BR_values[test_BR_idx] * PBR_values[test_PBR_idx] / (1 + test_DBR);
330  if(test_div < clk_div){ /* See note #2 */
331  continue;
332  }
333  test_distance = test_div - clk_div;
334  if(test_distance < best_distance){ /* Keep test values if better than the previous. */
335  div_found = TRUE;
336  best_distance = test_distance;
337  *p_BR = test_BR_idx;
338  *p_PBR = test_PBR_idx;
339  *p_DBR = test_DBR;
340  }
341  }
342  }
343  }
344 
345  return div_found;
346 }

361

364  ns_per_cycle = 1;
365  }
366 
367  /* Try out all the Baud Rates and keep best fit */
368  delay_found = FALSE;
369  best_distance = 0xFFFFFFFF; /* Initialize distance beyound maximum value */
370  for(test_scaler_idx = 0u; test_scaler_idx < 16; test_scaler_idx++){
371  for(test_prescaler_idx = 0u; test_prescaler_idx < 4; test_prescaler_idx++){
372 
373  test_delay = scaler_values[test_scaler_idx] * prescaler_values[test_prescaler_idx] * ns_per_cycle;
374  if(test_delay < delay){ /* See note #2 */
375  continue;
376  }
377  test_distance = test_delay - delay;
378  if(test_distance < best_distance){ /* Keep test values if better than the previous. */
379  delay_found = TRUE;
380  best_distance = test_distance;
381  *p_scaler = test_scaler_idx;
382  *p_prescaler = test_prescaler_idx;
383  }
384  }
385  }
