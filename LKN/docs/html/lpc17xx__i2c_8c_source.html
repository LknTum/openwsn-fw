399 void I2C_MasterHandler (LPC_I2C_TypeDef *I2Cx)
400 {
401  int32_t tmp;
402  uint8_t returnCode;
403  I2C_M_SETUP_Type *txrx_setup;
404 
405  tmp = I2C_getNum(I2Cx);
406  txrx_setup = (I2C_M_SETUP_Type *) i2cdat[tmp].txrx_setup;
407 
408  returnCode = (I2Cx->I2STAT & I2C_STAT_CODE_BITMASK);
409  // Save current status
410  txrx_setup->status = returnCode;
411  // there's no relevant information
412  if (returnCode == I2C_I2STAT_NO_INF){
413  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
414  return;
415  }
416 
417  /* ----------------------------- TRANSMIT PHASE --------------------------*/
418  if (i2cdat[tmp].dir == 0){
419  switch (returnCode)
420  {
421  /* A start/repeat start condition has been transmitted -------------------*/
422  case I2C_I2STAT_M_TX_START:
423  case I2C_I2STAT_M_TX_RESTART:
424  I2Cx->I2CONCLR = I2C_I2CONCLR_STAC;
425  /*
426  * If there's any transmit data, then start to
427  * send SLA+W right now, otherwise check whether if there's
428  * any receive data for next state.
429  */
430  if ((txrx_setup->tx_data != NULL) && (txrx_setup->tx_length != 0)){
431  I2Cx->I2DAT = (txrx_setup->sl_addr7bit << 1);
432  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
433  } else {
434  goto next_stage;
435  }
436  break;
437 
438  /* SLA+W has been transmitted, ACK has been received ----------------------*/
439  case I2C_I2STAT_M_TX_SLAW_ACK:
440  /* Data has been transmitted, ACK has been received */
441  case I2C_I2STAT_M_TX_DAT_ACK:
442  /* Send more data */
443  if ((txrx_setup->tx_count < txrx_setup->tx_length) \
444  && (txrx_setup->tx_data != NULL)){
445  I2Cx->I2DAT = *(uint8_t *)(txrx_setup->tx_data + txrx_setup->tx_count);
446  txrx_setup->tx_count++;
447  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
448  }
449  // no more data, switch to next stage
450  else {
451 next_stage:
452  // change direction
453  i2cdat[tmp].dir = 1;
454  // Check if any data to receive
455  if ((txrx_setup->rx_length != 0) && (txrx_setup->rx_data != NULL)){
456  // check whether if we need to issue an repeat start
457  if ((txrx_setup->tx_length != 0) && (txrx_setup->tx_data != NULL)){
458  // Send out an repeat start command
459  I2Cx->I2CONSET = I2C_I2CONSET_STA;
460  I2Cx->I2CONCLR = I2C_I2CONCLR_AAC | I2C_I2CONCLR_SIC;
461  }
462  // Don't need issue an repeat start, just goto send SLA+R
463  else {
464  goto send_slar;
465  }
466  }
467  // no more data send, the go to end stage now
468  else {
469  // success, goto end stage
470  txrx_setup->status |= I2C_SETUP_STATUS_DONE;
471  goto end_stage;
472  }
473  }
474  break;
475 
476  /* SLA+W has been transmitted, NACK has been received ----------------------*/
477  case I2C_I2STAT_M_TX_SLAW_NACK:
478  /* Data has been transmitted, NACK has been received -----------------------*/
479  case I2C_I2STAT_M_TX_DAT_NACK:
480  // update status
481  txrx_setup->status |= I2C_SETUP_STATUS_NOACKF;
482  goto retry;
483  /* Arbitration lost in SLA+R/W or Data bytes -------------------------------*/
484  case I2C_I2STAT_M_TX_ARB_LOST:
485  // update status
486  txrx_setup->status |= I2C_SETUP_STATUS_ARBF;
487  default:
488  goto retry;
489  }
490  }
491 
492  /* ----------------------------- RECEIVE PHASE --------------------------*/
493  else if (i2cdat[tmp].dir == 1){
494  switch (returnCode){
495  /* A start/repeat start condition has been transmitted ---------------------*/
496  case I2C_I2STAT_M_RX_START:
497  case I2C_I2STAT_M_RX_RESTART:
498  I2Cx->I2CONCLR = I2C_I2CONCLR_STAC;
499  /*
500  * If there's any receive data, then start to
501  * send SLA+R right now, otherwise check whether if there's
502  * any receive data for end of state.
503  */
504  if ((txrx_setup->rx_data != NULL) && (txrx_setup->rx_length != 0)){
505 send_slar:
506  I2Cx->I2DAT = (txrx_setup->sl_addr7bit << 1) | 0x01;
507  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
508  } else {
509  // Success, goto end stage
510  txrx_setup->status |= I2C_SETUP_STATUS_DONE;
511  goto end_stage;
512  }
513  break;
514 
515  /* SLA+R has been transmitted, ACK has been received -----------------*/
516  case I2C_I2STAT_M_RX_SLAR_ACK:
517  if (txrx_setup->rx_count < (txrx_setup->rx_length - 1)) {
518  /*Data will be received, ACK will be return*/
519  I2Cx->I2CONSET = I2C_I2CONSET_AA;
520  }
521  else {
522  /*Last data will be received, NACK will be return*/
523  I2Cx->I2CONCLR = I2C_I2CONSET_AA;
524  }
525  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
526  break;
527 
528  /* Data has been received, ACK has been returned ----------------------*/
529  case I2C_I2STAT_M_RX_DAT_ACK:
530  // Note save data and increase counter first, then check later
531  /* Save data */
532  if ((txrx_setup->rx_data != NULL) && (txrx_setup->rx_count < txrx_setup->rx_length)){
533  *(uint8_t *)(txrx_setup->rx_data + txrx_setup->rx_count) = (I2Cx->I2DAT & I2C_I2DAT_BITMASK);
534  txrx_setup->rx_count++;
535  }
536  if (txrx_setup->rx_count < (txrx_setup->rx_length - 1)) {
537  /*Data will be received, ACK will be return*/
538  I2Cx->I2CONSET = I2C_I2CONSET_AA;
539  }
540  else {
541  /*Last data will be received, NACK will be return*/
542  I2Cx->I2CONCLR = I2C_I2CONSET_AA;
543  }
544 
545  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
546  break;
547 
548  /* Data has been received, NACK has been return -------------------------*/
549  case I2C_I2STAT_M_RX_DAT_NACK:
550  /* Save the last data */
551  if ((txrx_setup->rx_data != NULL) && (txrx_setup->rx_count < txrx_setup->rx_length)){
552  *(uint8_t *)(txrx_setup->rx_data + txrx_setup->rx_count) = (I2Cx->I2DAT & I2C_I2DAT_BITMASK);
553  txrx_setup->rx_count++;
554  }
555  // success, go to end stage
556  txrx_setup->status |= I2C_SETUP_STATUS_DONE;
557  goto end_stage;
558 
559  /* SLA+R has been transmitted, NACK has been received ------------------*/
560  case I2C_I2STAT_M_RX_SLAR_NACK:
561  // update status
562  txrx_setup->status |= I2C_SETUP_STATUS_NOACKF;
563  goto retry;
564 
565  /* Arbitration lost ----------------------------------------------------*/
566  case I2C_I2STAT_M_RX_ARB_LOST:
567  // update status
568  txrx_setup->status |= I2C_SETUP_STATUS_ARBF;
569  default:
570 retry:
571  // check if retransmission is available
572  if (txrx_setup->retransmissions_count < txrx_setup->retransmissions_max){
573  // Clear tx count
574  txrx_setup->tx_count = 0;
575  I2Cx->I2CONSET = I2C_I2CONSET_STA;
576  I2Cx->I2CONCLR = I2C_I2CONCLR_AAC | I2C_I2CONCLR_SIC;
577  txrx_setup->retransmissions_count++;
578  }
579  // End of stage
580  else {
581 end_stage:
582  // Disable interrupt
583  I2C_IntCmd(I2Cx, 0);
584  // Send stop
585  I2C_Stop(I2Cx);
586 
587  I2C_MasterComplete[tmp] = TRUE;
588  }
589  break;
590  }
591  }

603 void I2C_SlaveHandler (LPC_I2C_TypeDef *I2Cx)
604 {
605  int32_t tmp;
606  uint8_t returnCode;
607  I2C_S_SETUP_Type *txrx_setup;
608  uint32_t timeout;
609 
610  tmp = I2C_getNum(I2Cx);
611  txrx_setup = (I2C_S_SETUP_Type *) i2cdat[tmp].txrx_setup;
612 
613  returnCode = (I2Cx->I2STAT & I2C_STAT_CODE_BITMASK);
614  // Save current status
615  txrx_setup->status = returnCode;
616  // there's no relevant information
617  if (returnCode == I2C_I2STAT_NO_INF){
618  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
619  return;
620  }
621 
622 
623  switch (returnCode)
624  {
625 
626  /* No status information */
627  case I2C_I2STAT_NO_INF:
628  I2Cx->I2CONSET = I2C_I2CONSET_AA;
629  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
630  break;
631 
632  /* Reading phase -------------------------------------------------------- */
633  /* Own SLA+R has been received, ACK has been returned */
634  case I2C_I2STAT_S_RX_SLAW_ACK:
635  /* General call address has been received, ACK has been returned */
636  case I2C_I2STAT_S_RX_GENCALL_ACK:
637  I2Cx->I2CONSET = I2C_I2CONSET_AA;
638  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
639  break;
640 
641  /* Previously addressed with own SLA;
642  * DATA byte has been received;
643  * ACK has been returned */
644  case I2C_I2STAT_S_RX_PRE_SLA_DAT_ACK:
645  /* DATA has been received, ACK hasn been return */
646  case I2C_I2STAT_S_RX_PRE_GENCALL_DAT_ACK:
647  /*
648  * All data bytes that over-flow the specified receive
649  * data length, just ignore them.
650  */
651  if ((txrx_setup->rx_count < txrx_setup->rx_length) \
652  && (txrx_setup->rx_data != NULL)){
653  *(uint8_t *)(txrx_setup->rx_data + txrx_setup->rx_count) = (uint8_t)I2Cx->I2DAT;
654  txrx_setup->rx_count++;
655  }
656  I2Cx->I2CONSET = I2C_I2CONSET_AA;
657  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
658  break;
659 
660  /* Previously addressed with own SLA;
661  * DATA byte has been received;
662  * NOT ACK has been returned */
663  case I2C_I2STAT_S_RX_PRE_SLA_DAT_NACK:
664  /* DATA has been received, NOT ACK has been returned */
665  case I2C_I2STAT_S_RX_PRE_GENCALL_DAT_NACK:
666  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
667  break;
668 
669  /*
670  * Note that: Return code only let us know a stop condition mixed
671  * with a repeat start condition in the same code value.
672  * So we should provide a time-out. In case this is really a stop
673  * condition, this will return back after time out condition. Otherwise,
674  * next session that is slave receive data will be completed.
675  */
676 
677  /* A Stop or a repeat start condition */
678  case I2C_I2STAT_S_RX_STA_STO_SLVREC_SLVTRX:
679  // Temporally lock the interrupt for timeout condition
680  I2C_IntCmd(I2Cx, 0);
681  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
682  // enable time out
683  timeout = I2C_SLAVE_TIME_OUT;
684  while(1){
685  if (I2Cx->I2CONSET & I2C_I2CONSET_SI){
686  // re-Enable interrupt
687  I2C_IntCmd(I2Cx, 1);
688  break;
689  } else {
690  timeout--;
691  if (timeout == 0){
692  // timeout occur, it's really a stop condition
693  txrx_setup->status |= I2C_SETUP_STATUS_DONE;
694  goto s_int_end;
695  }
696  }
697  }
698  break;
699 
700  /* Writing phase -------------------------------------------------------- */
701  /* Own SLA+R has been received, ACK has been returned */
702  case I2C_I2STAT_S_TX_SLAR_ACK:
703  /* Data has been transmitted, ACK has been received */
704  case I2C_I2STAT_S_TX_DAT_ACK:
705  /*
706  * All data bytes that over-flow the specified receive
707  * data length, just ignore them.
708  */
709  if ((txrx_setup->tx_count < txrx_setup->tx_length) \
710  && (txrx_setup->tx_data != NULL)){
711  I2Cx->I2DAT = *(uint8_t *) (txrx_setup->tx_data + txrx_setup->tx_count);
712  txrx_setup->tx_count++;
713  }
714  I2Cx->I2CONSET = I2C_I2CONSET_AA;
715  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
716  break;
717 
718  /* Data has been transmitted, NACK has been received,
719  * that means there's no more data to send, exit now */
720  /*
721  * Note: Don't wait for stop event since in slave transmit mode,
722  * since there no proof lets us know when a stop signal has been received
723  * on slave side.
724  */

884  */

895  goto retry;
896  }
897  }
898  } else {

904  // update status

976 Status I2C_SlaveTransferData(LPC_I2C_TypeDef *I2Cx, I2C_S_SETUP_Type *TransferCfg, \
977  I2C_TRANSFER_OPT_Type Opt)
978 {
979  uint8_t *txdat;
980  uint8_t *rxdat;
981  uint32_t CodeStatus;
982  uint32_t timeout;
983  int32_t time_en;
984  int32_t tmp;
985 
986  // reset all default state
987  txdat = (uint8_t *) TransferCfg->tx_data;
988  rxdat = (uint8_t *) TransferCfg->rx_data;
989  // Reset I2C setup value to default state
990  TransferCfg->tx_count = 0;
991  TransferCfg->rx_count = 0;
992  TransferCfg->status = 0;
993 
994 
995  // Polling option
996  if (Opt == I2C_TRANSFER_POLLING){
997 
998  /* Set AA bit to ACK command on I2C bus */
999  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1000  /* Clear SI bit to be ready ... */
1001  I2Cx->I2CONCLR = (I2C_I2CONCLR_SIC | I2C_I2CONCLR_STAC);
1002 
1003  time_en = 0;
1004  timeout = 0;
1005 
1006  while (1)
1007  {
1008  /* Check SI flag ready */
1009  if (I2Cx->I2CONSET & I2C_I2CONSET_SI)
1010  {
1011  time_en = 0;
1012 
1013  switch (CodeStatus = (I2Cx->I2STAT & I2C_STAT_CODE_BITMASK))
1014  {
1015 
1016  /* No status information */
1017  case I2C_I2STAT_NO_INF:
1018  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1019  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1020  break;
1021 
1022  /* Reading phase -------------------------------------------------------- */
1023  /* Own SLA+R has been received, ACK has been returned */
1024  case I2C_I2STAT_S_RX_SLAW_ACK:
1025  /* General call address has been received, ACK has been returned */
1026  case I2C_I2STAT_S_RX_GENCALL_ACK:
1027  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1028  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1029  break;
1030 
1031  /* Previously addressed with own SLA;
1032  * DATA byte has been received;
1033  * ACK has been returned */
1034  case I2C_I2STAT_S_RX_PRE_SLA_DAT_ACK:
1035  /* DATA has been received, ACK hasn been return */
1036  case I2C_I2STAT_S_RX_PRE_GENCALL_DAT_ACK:
1037  /*
1038  * All data bytes that over-flow the specified receive
1039  * data length, just ignore them.
1040  */
1041  if ((TransferCfg->rx_count < TransferCfg->rx_length) \
1042  && (TransferCfg->rx_data != NULL)){
1043  *rxdat++ = (uint8_t)I2Cx->I2DAT;
1044  TransferCfg->rx_count++;
1045  }
1046  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1047  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1048  break;
1049 
1050  /* Previously addressed with own SLA;
1051  * DATA byte has been received;
1052  * NOT ACK has been returned */
1053  case I2C_I2STAT_S_RX_PRE_SLA_DAT_NACK:
1054  /* DATA has been received, NOT ACK has been returned */
1055  case I2C_I2STAT_S_RX_PRE_GENCALL_DAT_NACK:
1056  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1057  break;
1058 
1059  /*
1060  * Note that: Return code only let us know a stop condition mixed
1061  * with a repeat start condition in the same code value.
1062  * So we should provide a time-out. In case this is really a stop
1063  * condition, this will return back after time out condition. Otherwise,
1064  * next session that is slave receive data will be completed.
1065  */
1066 
1067  /* A Stop or a repeat start condition */
1068  case I2C_I2STAT_S_RX_STA_STO_SLVREC_SLVTRX:
1069  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1070  // enable time out
1071  time_en = 1;
1072  timeout = 0;
1073  break;
1074 
1075  /* Writing phase -------------------------------------------------------- */
1076  /* Own SLA+R has been received, ACK has been returned */
1077  case I2C_I2STAT_S_TX_SLAR_ACK:
1078  /* Data has been transmitted, ACK has been received */
1079  case I2C_I2STAT_S_TX_DAT_ACK:
1080  /*
1081  * All data bytes that over-flow the specified receive
1082  * data length, just ignore them.
1083  */
1084  if ((TransferCfg->tx_count < TransferCfg->tx_length) \
1085  && (TransferCfg->tx_data != NULL)){
1086  I2Cx->I2DAT = *txdat++;
1087  TransferCfg->tx_count++;
1088  }
1089  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1090  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1091  break;
1092 
1093  /* Data has been transmitted, NACK has been received,
1094  * that means there's no more data to send, exit now */
1095  /*
1096  * Note: Don't wait for stop event since in slave transmit mode,
1097  * since there no proof lets us know when a stop signal has been received
1098  * on slave side.
1099  */
1100  case I2C_I2STAT_S_TX_DAT_NACK:
1101  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1102  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1103  // enable time out
1104  time_en = 1;
1105  timeout = 0;
1106  break;
1107 
1108  // Other status must be captured
1109  default:
1110  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC;
1111  goto s_error;
1112  }
1113  } else if (time_en){
1114  if (timeout++ > I2C_SLAVE_TIME_OUT){
1115  // it's really a stop condition, goto end stage
1116  goto s_end_stage;
1117  }
1118  }
1119  }
1120 
1121 s_end_stage:
1122  /* Clear AA bit to disable ACK on I2C bus */
1123  I2Cx->I2CONCLR = I2C_I2CONCLR_AAC;
1124  // Check if there's no error during operation
1125  // Update status
1126  TransferCfg->status = CodeStatus | I2C_SETUP_STATUS_DONE;
1127  return SUCCESS;
1128 
1129 s_error:
1130  /* Clear AA bit to disable ACK on I2C bus */
1131  I2Cx->I2CONCLR = I2C_I2CONCLR_AAC;
1132  // Update status
1133  TransferCfg->status = CodeStatus;
1134  return ERROR;
1135  }
1136 
1137  else if (Opt == I2C_TRANSFER_INTERRUPT){
1138  // Setup tx_rx data, callback and interrupt handler
1139  tmp = I2C_getNum(I2Cx);
1140  i2cdat[tmp].txrx_setup = (uint32_t) TransferCfg;
1141  // Set direction phase, read first
1142  i2cdat[tmp].dir = 1;
1143 
1144  // Enable AA
1145  I2Cx->I2CONSET = I2C_I2CONSET_AA;
1146  I2Cx->I2CONCLR = I2C_I2CONCLR_SIC | I2C_I2CONCLR_STAC;
1147  I2C_IntCmd(I2Cx, 1);
1148 
1149  return (SUCCESS);
1150  }
1151 
1152  return ERROR;
1153 }
