198 static void clock_init(void) {
199  /* Disable global interrupts */
200  bool bIntDisabled = IntMasterDisable();
201 
202  /* Configure the 32 kHz pins, PD6 and PD7, for crystal operation */
203  /* By default they are configured as GPIOs */
204  GPIODirModeSet(GPIO_D_BASE, 0x40, GPIO_DIR_MODE_IN);
205  GPIODirModeSet(GPIO_D_BASE, 0x80, GPIO_DIR_MODE_IN);
206  IOCPadConfigSet(GPIO_D_BASE, 0x40, IOC_OVERRIDE_ANA);
207  IOCPadConfigSet(GPIO_D_BASE, 0x80, IOC_OVERRIDE_ANA);
208 
209  /* Set the real-time clock to use the 32 kHz external crystal */
210  /* Set the system clock to use the external 32 MHz crystal */
211  /* Set the system clock to 32 MHz */
212  SysCtrlClockSet(true, false, SYS_CTRL_SYSDIV_32MHZ);
213 
214  /* Set the IO clock to operate at 16 MHz */
215  /* This way peripherals can run while the system clock is gated */
216  SysCtrlIOClockSet(SYS_CTRL_SYSDIV_16MHZ);
217 
218  /* Wait until the selected clock configuration is stable */
219  while (!((HWREG(SYS_CTRL_CLOCK_STA)) & (SYS_CTRL_CLOCK_STA_XOSC_STB)));
220 
221  /* Define what peripherals run in each mode */
222  SysCtrlRunSetting();
223  SysCtrlSleepSetting();
224  SysCtrlDeepSleepSetting();
225  SysCtrlWakeupSetting();
226 
227  /* Re-enable interrupt if initially enabled */
228  if (!bIntDisabled) {
229  IntMasterEnable();
230  }
231 }
