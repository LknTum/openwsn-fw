419 uint_8 USB_Desc_Get_Descriptor
420  (
421  uint_8 controller_ID,
422  uint_8 type,
423  uint_8 str_num,
424  uint_16 index,
425  uint_8_ptr *descriptor,
426  uint_32 *size
427  )
428 {
429  /* Body */
430  UNUSED (controller_ID)
431  /* string descriptors are handled saperately */
432  if (type == USB_STRING_DESCRIPTOR)
433  {
434  if(index == 0)
435  {
436  /* return the string and size of all languages */
437  *descriptor = (uint_8_ptr)g_languages.languages_supported_string;
438  *size = g_languages.languages_supported_size;
439  }
440  else
441  {
442  uint_8 lang_id=0;
443  uint_8 lang_index=USB_MAX_LANGUAGES_SUPPORTED;
444  for(;lang_id< USB_MAX_LANGUAGES_SUPPORTED;lang_id++)
445  {
446  /* check whether we have a string for this language */
447  if(index == g_languages.usb_language[lang_id].language_id)
448  { /* check for max descriptors */
449  if(str_num < USB_MAX_STRING_DESCRIPTORS)
450  { /* setup index for the string to be returned */
451  lang_index=str_num;
452  } /* EndIf */
453  break;
454  } /* EndIf */
455  } /* EndFor */
456  /* set return val for descriptor and size */
457  *descriptor = (uint_8_ptr)
458  g_languages.usb_language[lang_id].lang_desc[lang_index];
459  *size =
460  g_languages.usb_language[lang_id].lang_desc_size[lang_index];
461  } /* EndIf */
462  }
463  else
464  {
465  if (type < USB_MAX_STD_DESCRIPTORS+1)
466  {
467  /* set return val for descriptor and size*/
468  *descriptor = (uint_8_ptr)g_std_descriptors [type];
469 
470  /* if there is no descriptor then return error */
471  if(*descriptor == NULL)
472  {
473  return USBERR_INVALID_REQ_TYPE;
474  }
475  *size = g_std_desc_size[type];
476  }
477  else /* invalid descriptor */
478  {
479  return USBERR_INVALID_REQ_TYPE;
480  } /* EndIf */
481  } /* EndIf */
482  return USB_OK;
483 } /* EndBody */
