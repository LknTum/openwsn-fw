246 void
247 IntRegister(uint32_t ui32Interrupt, void (*pfnHandler)(void))
248 {
249  uint32_t ui32Idx, ui32Value;
250 
251  //
252  // Check the arguments.
253  //
254  ASSERT(ui32Interrupt < NUM_INTERRUPTS);
255 
256  // Check below could be removed in final application
257 #ifdef CC2538_USE_ALTERNATE_INTERRUPT_MAP
258  ASSERT(IntAltMapIsEnabled());
259 #else
260  ASSERT(~IntAltMapIsEnabled());
261 #endif
262 
263  //
264  // Make sure that the RAM vector table is correctly aligned.
265  //
266  ASSERT(((uint32_t)g_pfnRAMVectors & 0x000003ff) == 0);
267 
268  //
269  // See if the RAM vector table has been initialized.
270  //
271  if(HWREG(NVIC_VTABLE) != (uint32_t)g_pfnRAMVectors)
272  {
273  //
274  // Copy the vector table from the beginning of FLASH to the RAM vector
275  // table.
276  //
277  ui32Value = HWREG(NVIC_VTABLE);
278  for(ui32Idx = 0; ui32Idx < NUM_INTERRUPTS; ui32Idx++)
279  {
280  g_pfnRAMVectors[ui32Idx] = (void (*)(void))HWREG((ui32Idx * 4) +
281  ui32Value);
282  }
283 
284  //
285  // Point NVIC at the RAM vector table.
286  //
287  HWREG(NVIC_VTABLE) = (uint32_t)g_pfnRAMVectors;
288  }
289 
290  //
291  // Save the interrupt handler.
292  //
293  g_pfnRAMVectors[ui32Interrupt] = pfnHandler;
294 }

436 void
437 IntPrioritySet(uint32_t ui32Interrupt, uint8_t ui8Priority)
438 {
439  uint32_t ui32Temp;
440 
441  //
442  // Check the arguments.
443  //
444  ASSERT((ui32Interrupt >= 4) && (ui32Interrupt < NUM_INTERRUPTS));
445 
446  //
447  // Set the interrupt priority.
448  //
449  ui32Temp = HWREG(g_pui32Regs[ui32Interrupt >> 2]);
450  ui32Temp &= ~(0xFF << (8 * (ui32Interrupt & 3)));
451  ui32Temp |= ui8Priority << (8 * (ui32Interrupt & 3));
452  HWREG(g_pui32Regs[ui32Interrupt >> 2]) = ui32Temp;
453 }

496 void
497 IntEnable(uint32_t ui32Interrupt)
498 {
499  //
500  // Check the arguments.
501  //
502  ASSERT(ui32Interrupt < NUM_INTERRUPTS);
503 
504  //
505  // Determine the interrupt to enable.
506  //
507  if(ui32Interrupt == FAULT_MPU)
508  {
509  //
510  // Enable the MemManage interrupt.
511  //
512  HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_MEM;
513  }
514  else if(ui32Interrupt == FAULT_BUS)
515  {
516  //
517  // Enable the bus fault interrupt.
518  //
519  HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_BUS;
520  }
521  else if(ui32Interrupt == FAULT_USAGE)
522  {
523  //
524  // Enable the usage fault interrupt.
525  //
526  HWREG(NVIC_SYS_HND_CTRL) |= NVIC_SYS_HND_CTRL_USAGE;
527  }
528  else if(ui32Interrupt == FAULT_SYSTICK)
529  {
530  //
531  // Enable the System Tick interrupt.
532  //
533  HWREG(NVIC_ST_CTRL) |= NVIC_ST_CTRL_INTEN;
534  }
535  else if(ui32Interrupt >= 16)
536  {
537  //
538  // Enable the general interrupt.
539  //
540  HWREG(g_pui32EnRegs[(ui32Interrupt - 16) / 32]) =
541  1 << ((ui32Interrupt - 16) & 31);
542  }
543 }
