956 port_INLINE bool sixtop_processIEs(OpenQueueEntry_t* pkt, uint16_t * lenIE) {
957  uint8_t ptr;
958  uint8_t temp_8b,gr_elem_id,subid;
959  uint16_t temp_16b,len,sublen;
960  uint8_t version,commandIdORcode;
961  uint8_t sfId;
962 
963  ptr=0;
964 
965  //candidate IE header if type ==0 header IE if type==1 payload IE
966  temp_8b = *((uint8_t*)(pkt->payload)+ptr);
967  ptr++;
968  temp_16b = temp_8b + ((*((uint8_t*)(pkt->payload)+ptr))<<8);
969  ptr++;
970  *lenIE = ptr;
971  if(
972  (temp_16b & IEEE802154E_DESC_TYPE_PAYLOAD_IE) ==
973  IEEE802154E_DESC_TYPE_PAYLOAD_IE
974  ){
975  //payload IE - last bit is 1
976  len = temp_16b & IEEE802154E_DESC_LEN_PAYLOAD_IE_MASK;
977  gr_elem_id =
978  (temp_16b & IEEE802154E_DESC_GROUPID_PAYLOAD_IE_MASK)>>
979  IEEE802154E_DESC_GROUPID_PAYLOAD_IE_SHIFT;
980  }else {
981  //header IE - last bit is 0
982  len = temp_16b & IEEE802154E_DESC_LEN_HEADER_IE_MASK;
983  gr_elem_id = (temp_16b & IEEE802154E_DESC_ELEMENTID_HEADER_IE_MASK)>>
984  IEEE802154E_DESC_ELEMENTID_HEADER_IE_SHIFT;
985  }
986 
987  *lenIE += len;
988  //now determine sub elements if any
989  switch(gr_elem_id){
990  //this is the only groupID that we parse. See page 82.
991  case IEEE802154E_MLME_IE_GROUPID:
992  //IE content can be any of the sub-IEs. Parse and see which
993  do{
994  //read sub IE header
995  temp_8b = *((uint8_t*)(pkt->payload)+ptr);
996  ptr = ptr + 1;
997  temp_16b = temp_8b + ((*((uint8_t*)(pkt->payload)+ptr))<<8);
998  ptr = ptr + 1;
999  len = len - 2; //remove header fields len
1000  if(
1001  (temp_16b & IEEE802154E_DESC_TYPE_LONG) ==
1002  IEEE802154E_DESC_TYPE_LONG
1003  ){
1004  //long sub-IE - last bit is 1
1005  sublen = temp_16b & IEEE802154E_DESC_LEN_LONG_MLME_IE_MASK;
1006  subid=
1007  (temp_16b & IEEE802154E_DESC_SUBID_LONG_MLME_IE_MASK)>>
1008  IEEE802154E_DESC_SUBID_LONG_MLME_IE_SHIFT;
1009  } else {
1010  //short IE - last bit is 0
1011  sublen = temp_16b & IEEE802154E_DESC_LEN_SHORT_MLME_IE_MASK;
1012  subid = (temp_16b & IEEE802154E_DESC_SUBID_SHORT_MLME_IE_MASK)>>
1013  IEEE802154E_DESC_SUBID_SHORT_MLME_IE_SHIFT;
1014  }
1015  switch(subid){
1016  case IANA_6TOP_SUBIE_ID:
1017  temp_8b = *((uint8_t*)(pkt->payload)+ptr);
1018  ptr = ptr + 1;
1019  // get 6P version and command ID
1020  version = temp_8b & 0x0f;
1021  commandIdORcode = (temp_8b >> 4) & 0x0f;
1022  // get sf_id
1023  sfId = *((uint8_t*)(pkt->payload)+ptr);
1024  ptr = ptr + 1;
1025  sixtop_notifyReceiveCommand(version,commandIdORcode,sfId,ptr,sublen-2,pkt);
1026  ptr += sublen-2;
1027  break;
1028  default:
1029  return FALSE;
1030  }
1031  len = len - sublen;
1032  } while(len>0);
1033  break;
1034  case SIXTOP_IE_GROUPID:
1035  subid = *((uint8_t*)(pkt->payload)+ptr);
1036  ptr += 1;
1037  sublen = len - 1;
1038  switch(subid){
1039  case IANA_6TOP_SUBIE_ID:
1040  temp_8b = *((uint8_t*)(pkt->payload)+ptr);
1041  ptr = ptr + 1;
1042  // get 6P version and command ID
1043  version = temp_8b & 0x0f;
1044  commandIdORcode = (temp_8b >> 4) & 0x0f;
1045  // get sf_id
1046  sfId = *((uint8_t*)(pkt->payload)+ptr);
1047  ptr = ptr + 1;
1048  sixtop_notifyReceiveCommand(version,commandIdORcode,sfId,ptr,sublen-2,pkt);
1049  ptr += sublen-2;
1050  break;
1051  default:
1052  return FALSE;
1053  }
1054  break;
1055  default:
1056  *lenIE = 0;//no header or not recognized.
1057  return FALSE;
1058  }
1059  if (*lenIE>127) {
1060  // log the error
1061  openserial_printError(COMPONENT_IEEE802154E,ERR_HEADER_TOO_LONG,
1062  (errorparameter_t)*lenIE,
1063  (errorparameter_t)1);
1064  }
1065  return TRUE;
1066 }
