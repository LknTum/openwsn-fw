49 /* function: llwu_configure
50 
51  description: Set up the LLWU for wakeup the MCU from LLS and VLLSx modes
52  from the selected pin or module.
53 
54  inputs:
55  pin_en - unsigned integer, bit position indicates the pin is enabled.
56  More than one bit can be set to enable more than one pin at a time.
57 
58  rise_fall - 0x00 = External input disabled as wakeup
59  0x01 - External input enabled as rising edge detection
60  0x02 - External input enabled as falling edge detection
61  0x03 - External input enablge as any edge detection
62  module_en - unsigned char, bit position indicates the module is enabled.
63  More than one bit can be set to enabled more than one module
64 
65  for example: if bit 0 and 1 need to be enabled as rising edge detect call this routine with
66  pin_en = 0x0003 and rise_fall = 0x02
67 
68  Note: to set up one set of pins for rising and another for falling, 2 calls to this
69  function are required, 1st for rising then the second for falling.

196 void llwu_configure_filter(unsigned int wu_pin_num, unsigned char filter_en, unsigned char rise_fall )
197 {
198  //wu_pin_num is the pin number to be written to FILTSEL. wu_pin_num is not the same as pin_en.
199  uint8_t temp;
200 
201  //Enabling Filter %x on WU Pin %x for WU sense %x \n",filter_en, wu_pin_num, rise_fall);
202 
203  temp = 0;
204  //first clear filter values and clear flag by writing a 1
205 #ifdef TOWER_K20
206  LLWU_FILT1 = LLWU_FILT1_FILTF_MASK;
207  LLWU_FILT2 = LLWU_FILT2_FILTF_MASK;
208 
209  if(filter_en == 1)
210  {
211  //clear the flag bit and set the others
212  temp |= (LLWU_FILT1_FILTF_MASK) | (LLWU_FILT1_FILTE(rise_fall) | LLWU_FILT1_FILTSEL(wu_pin_num));
213  LLWU_FILT1 = temp;
214 
215  }else if (filter_en == 2)
216  {
217  //clear the flag bit and set the others
218  temp |= (LLWU_FILT2_FILTF_MASK) | (LLWU_FILT2_FILTE(rise_fall) | LLWU_FILT2_FILTSEL(wu_pin_num));
219  LLWU_FILT2 = temp;
220  }else
221  {
222  //Error - invalid filter number %x\n",filter_en);
223  }
224 #endif
