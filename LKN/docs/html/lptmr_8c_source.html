21 /*
22  * Initialize the low power time to provide a delay measured in ms.
23  *
24  *
25  * Parameters:
26  * count_val number of ms to delay
27  *
28  * Returns:
29  * None
30  */
31 void time_delay_ms(uint32 count_val)
32 {
33  /* Make sure the clock to the LPTMR is enabled */
34  SIM_SCGC5|=SIM_SCGC5_LPTIMER_MASK;
35 
36  /* Reset LPTMR settings */
37  LPTMR0_CSR=0;
38 
39  /* Set the compare value to the number of ms to delay */
40  LPTMR0_CMR = count_val;
41 
42  /* Set up LPTMR to use 1kHz LPO with no prescaler as its clock source */
43  LPTMR0_PSR = LPTMR_PSR_PCS(1)|LPTMR_PSR_PBYP_MASK;
44 
45  /* Start the timer */
46  LPTMR0_CSR |= LPTMR_CSR_TEN_MASK;
47 
48  /* Wait for counter to reach compare value */
49  while (!(LPTMR0_CSR & LPTMR_CSR_TCF_MASK));
50 
51  /* Disable counter and Clear Timer Compare Flag */
52  LPTMR0_CSR &= ~LPTMR_CSR_TEN_MASK;
53 
54  return;
55 }

76 void lptmr_init(int count, int clock_source)
77 {
78  SIM_SCGC5 |= SIM_SCGC5_LPTIMER_MASK;

81 
82  RTC_CR |= RTC_CR_OSCE_MASK |
83  RTC_CR_CLKO_MASK |
84  RTC_CR_SC8P_MASK ;
85 
86  LPTMR0_PSR = ( LPTMR_PSR_PRESCALE(0) // 0000 is div 2
87  | LPTMR_PSR_PBYP_MASK // LPO feeds directly to LPT
88  | LPTMR_PSR_PCS(clock_source)) ; // use the choice of clock
89  if (clock_source== 0)
90  printf("\n LPTMR Clock source is the MCGIRCLK \n\r");
91  if (clock_source== 1)
92  printf("\n LPTMR Clock source is the LPOCLK \n\r");
93  if (clock_source== 2)
94  printf("\n LPTMR Clock source is the ERCLK32 \n\r");
95  if (clock_source== 3)
96  printf("\n LPTMR Clock source is the OSCERCLK \n\r");
97 
98  LPTMR0_CMR = LPTMR_CMR_COMPARE(count); //Set compare value
99 
100  LPTMR0_CSR =( LPTMR_CSR_TCF_MASK // Clear any pending interrupt
101  | LPTMR_CSR_TIE_MASK // LPT interrupt enabled
102  | LPTMR_CSR_TPS(0) //TMR pin select
103  |!LPTMR_CSR_TPP_MASK //TMR Pin polarity
104  |!LPTMR_CSR_TFC_MASK // Timer Free running counter is reset whenever TMR counter equals compare
105  |!LPTMR_CSR_TMS_MASK //LPTMR as Timer
106  );
107  LPTMR0_CSR |= LPTMR_CSR_TEN_MASK; //Turn on LPT and start counting
108 }
