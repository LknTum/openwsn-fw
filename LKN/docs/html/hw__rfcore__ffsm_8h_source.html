183 #define RFCORE_FFSM_SRCRESINDEX_SRCRESINDEX_M \
184  0x000000FF // The bit index of the
185  // least-significant entry (0-23
186  // for short addresses or 0-11 for
187  // extended addresses) in
188  // SRCRESMASK, or 0x3F when there
189  // is no source match On a match,
190  // bit 5 is 0 when the match is on
191  // a short address and 1 when it is
192  // on an extended address. On a
193  // match, bit 6 is 1 when the
194  // conditions for automatic pending
195  // bit in acknowledgment have been
196  // met (see the description of
197  // SRCMATCH.AUTOPEND). The bit does
198  // not indicate if the
199  // acknowledgment is actually
200  // transmitted, and does not
201  // consider the PENDING_OR register
202  // bit and the SACK/SACKPEND/SNACK
203  // strobes.
204 
205 #define RFCORE_FFSM_SRCRESINDEX_SRCRESINDEX_S 0
