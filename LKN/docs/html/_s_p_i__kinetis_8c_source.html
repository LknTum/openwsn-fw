61 /*********************************************************
62 * Name: SPI_Init
63 * Desc: Initialize SPI2 Module
64 * Parameter: None
65 * Return: None
66 **********************************************************/
67 void SPI_Init(void)
68 {
69  /*Body*/
70  /* set PORTE pin 1 to DSPI1.SOUT*/
71 #ifdef MCU_MK70F12
72  PORTE_PCR1 = PORT_PCR_MUX(7);
73 #else
74  PORTE_PCR1 = PORT_PCR_MUX(2);
75 #endif // MCU_MK70F12
76  /* set PORTE pin 2 to DSPI1.SCK*/
77  PORTE_PCR2 = PORT_PCR_MUX(2);
78  /* set PORTE pin 3 to DSPI1.SIN*/
79 #ifdef MCU_MK70F12
80  PORTE_PCR3 = PORT_PCR_MUX(7);
81 #else
82  PORTE_PCR3 = PORT_PCR_MUX(2);
83 #endif // MCU_MK70F12
84  /* set PORTE pin 4 to DSPI1.CS0*/
85  PORTE_PCR4 = PORT_PCR_MUX(2);
86 
87  SIM_SCGC5 |= SIM_SCGC5_PORTE_MASK;
88  /* Enable clock gate to DSPI1 module */
89 #ifdef MCU_MK70F12
90  SIM_SCGC6 |= SIM_SCGC6_DSPI1_MASK;
91 #else
92  SIM_SCGC6 |= SIM_SCGC6_SPI1_MASK;
93 #endif // MCU_MK70F12
94 
95  /* Enable master mode, disable both transmit and receive FIFO buffers,
96  set the inactive state for PCS2 high, enable the module (MDIS = 0),
97  delay the sample point from the leading edge of the clock and halt
98  any transfer
99  */
100  SPI1_MCR = (SPI_MCR_MSTR_MASK |
101  SPI_MCR_PCSIS(1) |
102  SPI_MCR_SMPL_PT(2) |
103  SPI_MCR_HALT_MASK);
104 
105  SPI1_MCR |= (SPI_MCR_CLR_RXF_MASK | SPI_MCR_CLR_TXF_MASK);
106  // K60: bus clock: 48MHz, DSPI clock: ~107 KHz
107  // K70: bus clock: 60MHz, DSPI clock: ~156 KHz
108  /* Value to be passed in SPI_Send_byte() function to DPI_CTAR0 register */
109 #ifdef MCU_MK70F12
110  gSPI_BaudRate = (SPI_CTAR_PBR(1) | SPI_CTAR_BR(0x07));
111 #else
112  gSPI_BaudRate = (SPI_CTAR_PBR(3) | SPI_CTAR_BR(0x06));
113 #endif
114  /* Configure the rest of the delays */
115  gSPI_BeforeTransfDelay = (SPI_CTAR_CSSCK(1) | SPI_CTAR_CSSCK(0x04));
116  gSPI_AfterTransfDelay = (SPI_CTAR_PASC(3) | SPI_CTAR_ASC(0x04));
117  gSPI_InterTransfDelay = (SPI_CTAR_PDT(3) | SPI_CTAR_DT(0x05));
118 }/*EndBody*/

122 /*********************************************************
123 * Name: SPI_Send_byte
124 * Desc: Send one byte
125 * Parameter: The byte to be sent
126 * Return: None
127 **********************************************************/
128 void SPI_Send_byte(uint_8 u8Data)
129 {
130  /*Body*/
131  /* Check the status flag */
132  if(SPI1_SR & SPI_SR_EOQF_MASK)
133  {
134  /* Clear the EOQF by writting a 1 to it */
135  SPI1_SR |= SPI_SR_EOQF_MASK;
136  }/*Endif*/
137 
138  /* Write the DSPI_PUSHR register */
139  SPI1_PUSHR = ( SPI_PUSHR_CTAS(0) |
140  SPI_PUSHR_EOQ_MASK |
141  SPI_PUSHR_CTCNT_MASK |
142  SPI_PUSHR_PCS(1) |
143  SPI_PUSHR_TXDATA(u8Data));
144  /* Write the clock and transfer attributes: master clock and frame size (8 bits) */
145  SPI1_CTAR0 = ( SPI_CTAR_SLAVE_FMSZ(7) |
146  gSPI_BeforeTransfDelay |
147  gSPI_AfterTransfDelay |
148  gSPI_InterTransfDelay |
149  gSPI_BaudRate);
150 
151  /* Start the transfer */
152  SPI1_MCR &= ~SPI_MCR_HALT_MASK;
153 
154  /* Wait until the transfer has been finished */
155  while(!(SPI1_SR & SPI_SR_EOQF_MASK))
156  {
157  Watchdog_Reset();
158  }/*EndWhile*/
159  /* Clear the EOQF by writting a 1 to it */
160  SPI1_SR |= SPI_SR_EOQF_MASK;
161 }/*EndBody*/

165 /*********************************************************
166 * Name: SPI_Receive_byte
167 * Desc: The byte received by SPI
168 * Parameter: None
169 * Return: Received byte
170 **********************************************************/
171 uint_8 SPI_Receive_byte(void)
172 {
173  uint_16 u8Data;
174 
175  /*Body*/
176  /* Check the status flag */
177  if(SPI1_SR & SPI_SR_EOQF_MASK)
178  {
179  /* Clear the EOQF by writting a 1 to it */
180  SPI1_SR |= SPI_SR_EOQF_MASK;
181  }/*EndIf*/
182 
183  /* Write the DSPI_PUSHR register */
184  SPI1_PUSHR = ( SPI_PUSHR_CTAS(0) |
185  SPI_PUSHR_EOQ_MASK |
186  SPI_PUSHR_CTCNT_MASK |
187  SPI_PUSHR_PCS(1) |
188  SPI_PUSHR_TXDATA(0xFF));
189  /* Write the clock and transfer attributes: master clock and frame size (8 bits) */
190  SPI1_CTAR0 = ( SPI_CTAR_FMSZ(7) |
191  gSPI_BeforeTransfDelay |
192  gSPI_AfterTransfDelay |
193  gSPI_InterTransfDelay |
194  gSPI_BaudRate);
195 
196  /* Start the transfer */
197  SPI1_MCR &= ~SPI_MCR_HALT_MASK;
198  /* Wait until the transfer has been finished */
199  while(!(SPI1_SR & SPI_SR_EOQF_MASK))
200  {
201  Watchdog_Reset();
202  }/*EndWhile*/
203  /* Clear the EOQF by writting a 1 to it */
204  SPI1_SR |= SPI_SR_EOQF_MASK;
205  /* Read the byte form the DSPI_POPR register */
206  u8Data = (uint_16)SPI_RXFR0_RXDATA(SPI1_POPR);
207 
208  return((uint_8)u8Data);
209 }/*EndBody*/
