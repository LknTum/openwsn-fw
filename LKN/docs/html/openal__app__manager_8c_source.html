46 openal_error_code_t openal_claim_pin(char pin) {
47  xTaskHandle curTask;
48  openal_pin_state_t * pin_state;
49  char c;
50  openal_error_code_t error;
51  xSemaphoreTake(pin_registry_mutex, portMAX_DELAY); // Going to be modifying the pin registry
52  xSemaphoreTake(app_data_mutex, portMAX_DELAY);
53 
54  // the pin parameter is 1-indexed
55  pin--;
56 
57  // Not a proper pin
58  if ( pin < 0 || pin >= OPENMOTE_NUM_PINS || openmote_pin_gpio_bank[pin] == NOT_VALID_PIN) {
59  error = INVALID_PIN;
60  goto fail;
61  }
62 
63  curTask = xTaskGetCurrentTaskHandle();
64 
65  pin_state = &openal_pin_registry[pin];
66 
67  // Already claimed by this task
68  if ( pin_state->owner == curTask ) {
69  goto success;
70  }
71 
72  for ( c = 0; c < OPENAL_NUM_APPS; c++ ) {
73  if (openal_app_data[c].handle == curTask ) {
74  goto taskfound;
75  } else if ( openal_app_data[c].handle == pin_state->owner ) { // Already claimed by a higher-priority task
76  error = PIN_ACCESS_DENIED;
77  goto fail;
78  }
79  }
80  error = UNKNOWN_CALLER;
81  goto fail;
82 
83  taskfound:
84  if ( app_registry[c].pin_permissions & (1<<pin) ) {
85  pin_state->owner = curTask;
86  goto success;
87  } else {
88  error = PIN_ACCESS_DENIED;
89  goto fail;
90  }
91 
92  success:
93  xSemaphoreGive((app_data_mutex));
94  xSemaphoreGive(pin_registry_mutex);
95  return SUCCESS;
96 
97  fail:
98  xSemaphoreGive((app_data_mutex));
99  xSemaphoreGive(pin_registry_mutex);
100  return error;
101 }

165 
166  xSemaphoreTake(app_data_mutex, portMAX_DELAY);

176  c--; // app numbers are 1-indexed

179  } else {
180  app_no--; // app numbers are 1-indexed
181 
182  if ( app_no < 0 || app_no >= OPENAL_NUM_APPS ) // invalid
183  goto error;
184 
185  curApp = &openal_app_data[app_no];
186 
187  if ( curApp->handle != curTask && !prvIsPrivileged()) {
188  // You are only allowed to kill yourself if not privileged
189  goto error;
190  }
191  }
192 
193  curTask = curApp->handle;
194  curApp->handle = NULL;
195  curApp->state = STOPPED;
196  xSemaphoreGive(app_data_mutex);
197 
198  // release pins from use
199  xSemaphoreTake(pin_registry_mutex, portMAX_DELAY);
200  for (c = 0; c < OPENMOTE_NUM_PINS; c++) {
201  if ( openal_pin_registry[c].owner == curTask ) {
202  openal_pin_registry[c].owner = NO_OWNER;
203  }
204  }
205  xSemaphoreGive(pin_registry_mutex);
206  vQueueDelete(openal_event_queues[app_no]);
207  vTaskDelete(curTask);
208 
209  // should not reach here if this was called on current task
210  return;
211
