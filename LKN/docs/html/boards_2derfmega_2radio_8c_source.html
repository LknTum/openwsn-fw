175 void radio_txNow() {
176  // change state
177  radio_vars.state = RADIOSTATE_TRANSMITTING;
178 
179  // send packet by forcing state to TX_START
180  radio_internalWriteReg(TRX_STATE, CMD_TX_START);
181  // The AT86RF231 does not generate an interrupt when the radio transmits the
182  // SFD, which messes up the MAC state machine. The danger is that, if we leave
183  // this funtion like this, any radio watchdog timer will expire.
184  // Instead, we cheat an mimick a start of frame event by calling
185  // ieee154e_startOfFrame from here. This also means that software can never catch
186  // a radio glitch by which #radio_txEnable would not be followed by a packet being
187  // transmitted (I've never seen that).
188  if (radio_vars.startFrame_cb!=NULL) {
189  // call the callback
190  radio_vars.startFrame_cb(radiotimer_getCapturedTime());
191  }
192 }
