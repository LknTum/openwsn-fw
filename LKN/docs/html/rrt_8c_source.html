86  owerror_t outcome;

90 
91  switch (coap_header->Code) {
92  case COAP_CODE_REQ_GET:
93 
94  //=== reset packet payload (we will reuse this packetBuffer)
95  msg->payload = &(msg->packet[127]);
96  msg->length = 0;
97 
98  //=== prepare CoAP response
99  rrt_setGETRespMsg(msg, rrt_vars.discovered);
100 
101 
102  // payload marker
103  packetfunctions_reserveHeaderSize(msg,1);
104  msg->payload[0] = COAP_PAYLOAD_MARKER;
105 
106  // set the CoAP header
107  coap_header->Code = COAP_CODE_RESP_CONTENT;
108 
109  outcome = E_SUCCESS;
110  break;
111  case COAP_CODE_REQ_PUT:
112  case COAP_CODE_REQ_POST:
113  mssgRecvd = msg->payload[0];
114 
115  if (mssgRecvd == 'C') {
116  rrt_vars.discovered = 1;
117  } else if (mssgRecvd == 'B' && rrt_vars.discovered == 1) {
118  //blink mote
119  leds_error_toggle();
120 
121  //send packet back saying it did action B - blink
122  rrt_sendCoAPMsg('B', NULL); //NULL for ringmaster
123  } else if (mssgRecvd == 'F' && rrt_vars.discovered == 1) { //format - FB[ipv6]
124  actionToFwd = msg->payload[1];
125  memcpy(&moteToSendTo, &msg->payload[2], 16);
126  rrt_sendCoAPMsg(actionToFwd, moteToSendTo);
127  }
128 
129  // reset packet payload
130  msg->payload = &(msg->packet[127]);
131  msg->length = 0;
132 
133  packetfunctions_reserveHeaderSize(msg, 1);
134  msg->payload[0] = COAP_PAYLOAD_MARKER;
135 
136  //set the CoAP header
137  coap_header->Code = COAP_CODE_RESP_CONTENT;
138 
139  outcome = E_SUCCESS;
140 
141  break;
142  case COAP_CODE_REQ_DELETE:
143  msg->payload = &(msg->packet[127]);
144  msg->length = 0;
145 
146  //unregister the current mote as 'discovered' by ringmaster
147  rrt_vars.discovered = 0;
148 
149  // payload marker
150  packetfunctions_reserveHeaderSize(msg,1);
151  msg->payload[0] = COAP_PAYLOAD_MARKER;
152

155 
156  outcome = E_SUCCESS;
