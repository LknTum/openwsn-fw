264 void abstimer_reschedule() {
265  // the goal here is to take the next compare time,
266  // i.e. the one with the smallest difference to currentTime which is armed
267 
268  uint8_t i; // iterator
269  bool found; // TRUE iff a timer could be found to schedule
270  uint16_t valToLoad; // the value to eventually load in the compare register
271  uint16_t thisDist; // the distance in the future of this timer
272  uint16_t minDist; // the minimem distance amoung all timers
273 
274  minDist = 0xffff;
275 
276  // find the timer closest in time to now
277  found = FALSE;
278  for (i=0;i<ABSTIMER_SRC_MAX;i++) {
279  if (abstimer_vars.isArmed[i]==TRUE) {
280  thisDist = abstimer_vars.compareVal[i]-abstimer_vars.currentTime;
281  if (
282  found==FALSE ||
283  thisDist<minDist
284  ) {
285  valToLoad = abstimer_vars.compareVal[i];
286  minDist = thisDist;
287  found = TRUE;
288  }
289  }
290  }
291 
292  // load that timer, if found
293  if (found==TRUE) {
294  sctimer_schedule(valToLoad);
295  abstimer_vars.nextCurrentTime = valToLoad;
296  } else {
297  sctimer_stop();
298  }
299 }

309  uint8_t i; // iterator
310  uint16_t now,timeSpent;
311  uint16_t calc;
312  uint16_t min;
313  uint16_t tempcompare;
314 
315  // update the current theoretical time -- nextCurrentTime MUST be NOW
316  abstimer_vars.currentTime = abstimer_vars.nextCurrentTime;
317 
318  //debug -- to detect nested behaviours in case of reentrant interrupt.. should never happen
319  abstimer_dbg.nested_isr++;
320 
321  //===== step 1. Find out which interrupts just fired
322 
323  // Note: we store the list of interrupts which fired in an 8-bit bitmap
324  abstimer_vars.bitmapInterruptsFired = 0;
325 
326  for (i=0;i<ABSTIMER_SRC_MAX;i++) {
327  if (
328  (abstimer_vars.isArmed[i]==TRUE) &&
329  (abstimer_vars.compareVal[i]==abstimer_vars.currentTime)
330  ) {
331  abstimer_vars.bitmapInterruptsFired |= (1<<i);
332  }
333  }
334  if ((abstimer_vars.bitmapInterruptsFired & (1<<ABSTIMER_SRC_RADIOTIMER_OVERFLOW)) &&
335  (abstimer_vars.bitmapInterruptsFired & (1<<ABSTIMER_SRC_RADIOTIMER_COMPARE))) {
336  abstimer_vars.bitmapInterruptsFired &= ~(1<<ABSTIMER_SRC_RADIOTIMER_COMPARE);
337  }
338 
339  // make sure at least one timer fired
340  if (abstimer_vars.bitmapInterruptsFired==0) {
341  while(1);
342  }
343 
344  //how many loops? -- this tells us how many timers have expired at the same time.
345  abstimer_dbg.consecutive_late=0;
346 
347  //let's execute all of them..
348  while (abstimer_vars.bitmapInterruptsFired!=0) {
349 //check that...
350  while (abstimer_vars.bitmapInterruptsFired!=0) {
351  //radiotimer overflow...
352  if (abstimer_vars.bitmapInterruptsFired & (1<<ABSTIMER_SRC_RADIOTIMER_OVERFLOW)) {
353 
354  // update debug stats
355  abstimer_dbg.num_radiotimer_overflow++;
356 
357  // keep previous value -- this is now poipoi
358  abstimer_vars.radiotimer_overflow_previousVal = \
359  abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_OVERFLOW];
360  // remember compare val in case the callback modifies it.
361  tempcompare = abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_OVERFLOW];
362 
363  // call the callback
364  abstimer_vars.callback[ABSTIMER_SRC_RADIOTIMER_OVERFLOW]();
365 
366  //==== step 2. Reschedule the timers which fired and need to be rescheduled
367 
368  // reschedule automatically if wasn't changed during callback
369  if (abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_OVERFLOW]==tempcompare) {
370  // this is a periodic timer, reschedule automatically -- only if the callback has not already set it.
371  abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_OVERFLOW] += abstimer_vars.radiotimer_period;
372  }
373 
374  // clear the interrupt flag
375  abstimer_vars.bitmapInterruptsFired &= ~(1<<ABSTIMER_SRC_RADIOTIMER_OVERFLOW);
376  //radiotimer compare inside the slot...
377  } else if (abstimer_vars.bitmapInterruptsFired & (1<<ABSTIMER_SRC_RADIOTIMER_COMPARE)) {
378 
379  // update debug stats
380  abstimer_dbg.num_radiotimer_compare++;
381 
382  // remember compare val
383  tempcompare = abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_COMPARE];
384 
385  // call the callback
386  abstimer_vars.callback[ABSTIMER_SRC_RADIOTIMER_COMPARE]();
387  // XV: I do not understand this ... why is rescheduled in the next slot?
388 
389  // reschedule automatically if wasn't changed during callback
390  if (abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_COMPARE]==tempcompare) {
391 
392  // reschedule automatically after *overflow*
393  abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_COMPARE] = abstimer_vars.compareVal[ABSTIMER_SRC_RADIOTIMER_OVERFLOW]+ \
394  abstimer_vars.radiotimer_compare_offset;
395  }
396 
397  // clear the interrupt flag
398  abstimer_vars.bitmapInterruptsFired &= ~(1<<ABSTIMER_SRC_RADIOTIMER_COMPARE);
399  //bsp timer
400  } else if (abstimer_vars.bitmapInterruptsFired & (1<<ABSTIMER_SRC_BSP_TIMER)) {
401 
402  // update debug stats
403  abstimer_dbg.num_bsp_timer++;
404 
405  // call the callback
406  abstimer_vars.callback[ABSTIMER_SRC_BSP_TIMER]();
407 
408  // clear the interrupt flag
409  abstimer_vars.bitmapInterruptsFired &= ~(1<<ABSTIMER_SRC_BSP_TIMER);
410 
411  // no automatic rescheduling
412 
413  }
414  }
415 
416  // make sure all interrupts have been handled
417  if (abstimer_vars.bitmapInterruptsFired!=0) {
418  while(1);
419  }
420 
421  //===== step 4. schedule the next operation
422 
423  abstimer_reschedule();
424 
425  //===== step 5. make sure I'm not late for my next schedule
426 
427  // evaluate how much time has passed since theoretical currentTime,
428  // (over estimate by ABSTIMER_GUARD_TICKS)
429  now = sctimer_getValue(); //this gets the value of the counter
430  timeSpent = (uint16_t)((uint16_t)now-(uint16_t)abstimer_vars.currentTime);
431  timeSpent += ABSTIMER_GUARD_TICKS;
432 
433  //debug -- to detect wraps...
434  if (timeSpent > 65000){
435  //in case of negative wrap, lets say that a small time has elapsed.
436  timeSpent=ABSTIMER_GUARD_TICKS;
437  }
438 
439  // verify that, for each timer, that duration doesn't exceed how much time is left
440  //find smallest next timer and mark it in the bitmap to be executed next. Can only be one marked at a time.
441  min=0xFFFF;
442  for (i=0;i<ABSTIMER_SRC_MAX;i++) {
443 
444  // calculate distance to next timeout
445  calc = (uint16_t)abstimer_vars.compareVal[i]-(uint16_t)abstimer_vars.currentTime;
446 
447  if ((abstimer_vars.isArmed[i]==TRUE) && (timeSpent>calc) && (calc<min)) {
448  //this is the nearest in time and we are already late or it is closer to its deadline, (distance < ABSTIMER_GUARD_TICKS)
449  min=calc;
450  // this interrupt needs to be serviced now
451  abstimer_vars.bitmapInterruptsFired = (1<<i);
452  // update the next timeout value
453  abstimer_vars.nextCurrentTime=abstimer_vars.compareVal[i];
454  // update debug statistics
455  abstimer_dbg.num_late_schedule++;
456  abstimer_dbg.consecutive_late++;
457 
458  }
459  }

461  abstimer_dbg.count_late++;

464  // 1) the isr function of that timer sets another timeout, also close in time, and the time it takes to execute the actual callback is higher
465  // than the difference between now and the next timeout.. this will cause a timer to be lost.
