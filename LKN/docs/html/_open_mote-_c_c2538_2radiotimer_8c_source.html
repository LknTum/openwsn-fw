222 /*
223  * The interrupt flags are given in the RFCORE_SFR_MTIRQF registers. The interrupt flag bits are set only
224 by hardware and are cleared only by writing to the SFR register.
225 Each interrupt source can be masked by its corresponding mask bit in the RFCORE_SFR_MTIRQM
226 register. An interrupt is generated when the corresponding mask bit is set; otherwise, the interrupt is not
227 generated. The interrupt flag bit is set, however, regardless of the state of the interrupt mask bit.
228  *
229  */
230 kick_scheduler_t radiotimer_isr() {
231 
232  uint8_t t2irqm;
233  uint8_t t2irqf;
234 
235  t2irqm = HWREG(RFCORE_SFR_MTIRQM);
236  t2irqf = HWREG(RFCORE_SFR_MTIRQF);
237 
238  IntPendClear(INT_MACTIMR);
239 
240  if ((t2irqf & RFCORE_SFR_MTIRQM_MACTIMER_OVF_COMPARE1M)& t2irqm){ // compare 1
241  debugpins_isr_toggle();
242  //clear interrupt
243  HWREG(RFCORE_SFR_MTIRQF)=~RFCORE_SFR_MTIRQM_MACTIMER_OVF_COMPARE1M;
244 
245  if (radiotimer_vars.compare_cb!=NULL) {
246  // call the callback
247  radiotimer_vars.compare_cb();
248  // kick the OS
249  return KICK_SCHEDULER;
250  }
251  } else if((t2irqf & RFCORE_SFR_MTIRQM_MACTIMER_OVF_PERM) & t2irqm){ // timer overflows
252  debugpins_isr_toggle();
253  //clear interrupt
254  HWREG(RFCORE_SFR_MTIRQF)=~RFCORE_SFR_MTIRQM_MACTIMER_OVF_PERM;
255 
256  if (radiotimer_vars.overflow_cb!=NULL) {
257  // call the callback
258  radiotimer_vars.overflow_cb();
259  // kick the OS
260  return KICK_SCHEDULER;
261  }
262  }
263  return DO_NOT_KICK_SCHEDULER;
264 }
