708 void
709 uDMAChannelTransferSet(uint32_t ui32ChannelStructIndex,
710  uint32_t ui32Mode, void *pvSrcAddr, void *pvDstAddr,
711  uint32_t ui32TransferSize)
712 {
713  tDMAControlTable *pControlTable;
714  uint32_t ui32Control;
715  uint32_t ui32Inc;
716  uint32_t ui32BufferBytes;
717 
718  //
719  // Check the arguments.
720  //
721  ASSERT((ui32ChannelStructIndex & 0xffff) < 64);
722  ASSERT(HWREG(UDMA_CTLBASE) != 0);
723  ASSERT(ui32Mode <= UDMA_MODE_PER_SCATTER_GATHER);
724  ASSERT((uint32_t)pvSrcAddr >= 0x20000000);
725  ASSERT((uint32_t)pvDstAddr >= 0x20000000);
726  ASSERT((ui32TransferSize != 0) && (ui32TransferSize <= 1024));
727 
728  //
729  // In case a channel selector macro (like UDMA_CH0_USB0EP1RX) was
730  // passed as the ui32ChannelStructIndex parameter, extract just the channel
731  // index from this parameter.
732  //
733  ui32ChannelStructIndex &= 0x3f;
734 
735  //
736  // Get the base address of the control table.
737  //
738  pControlTable = (tDMAControlTable *)HWREG(UDMA_CTLBASE);
739 
740  //
741  // Get the current control word value and mask off the mode and size
742  // fields.
743  //
744  ui32Control = (pControlTable[ui32ChannelStructIndex].ui32Control &
745  ~(UDMACHCTL_CHCTL_XFERSIZE_M | UDMACHCTL_CHCTL_XFERMODE_M));
746 
747  //
748  // Adjust the mode if the alt control structure is selected.
749  //
750  if(ui32ChannelStructIndex & UDMA_ALT_SELECT)
751  {
752  if((ui32Mode == UDMA_MODE_MEM_SCATTER_GATHER) ||
753  (ui32Mode == UDMA_MODE_PER_SCATTER_GATHER))
754  {
755  ui32Mode |= UDMA_MODE_ALT_SELECT;
756  }
757  }
758 
759  //
760  // Set the transfer size and mode in the control word (but don't write the
761  // control word yet as it could kick off a transfer).
762  //
763  ui32Control |= ui32Mode | ((ui32TransferSize - 1) << 4);
764 
765  //
766  // Get the address increment value for the source, from the control word.
767  //
768  ui32Inc = (ui32Control & UDMACHCTL_CHCTL_SRCINC_M);
769 
770  //
771  // Compute the ending source address of the transfer. If the source
772  // increment is set to none, then the ending address is the same as the
773  // beginning.
774  //
775  if(ui32Inc != UDMA_SRC_INC_NONE)
776  {
777  ui32Inc = ui32Inc >> 26;
778  ui32BufferBytes = ui32TransferSize << ui32Inc;
779  pvSrcAddr = (void *)((uint32_t)pvSrcAddr + ui32BufferBytes - 1);
780  }
781 
782  //
783  // Load the source ending address into the control block.
784  //
785  pControlTable[ui32ChannelStructIndex].pvSrcEndAddr = pvSrcAddr;
786 
787  //
788  // Get the address increment value for the destination, from the control
789  // word.
790  //
791  ui32Inc = ui32Control & UDMACHCTL_CHCTL_DSTINC_M;
792 
793  //
794  // Compute the ending destination address of the transfer. If the
795  // destination increment is set to none, then the ending address is the
796  // same as the beginning.
797  //
798  if(ui32Inc != UDMA_DST_INC_NONE)
799  {
800  //
801  // There is a special case if this is setting up a scatter-gather
802  // transfer. The destination pointer needs to point to the end of
803  // the alternate structure for this channel instead of calculating
804  // the end of the buffer in the normal way.
805  //
806  if((ui32Mode == UDMA_MODE_MEM_SCATTER_GATHER) ||
807  (ui32Mode == UDMA_MODE_PER_SCATTER_GATHER))
808  {
809  pvDstAddr =
810  (void *)&pControlTable[ui32ChannelStructIndex |
811  UDMA_ALT_SELECT].ui32Spare;
812  }
813  //
814  // Not a scatter-gather transfer, calculate end pointer normally.
815  //
816  else
817  {
818  ui32Inc = ui32Inc >> 30;
819  ui32BufferBytes = ui32TransferSize << ui32Inc;
820  pvDstAddr = (void *)((uint32_t)pvDstAddr + ui32BufferBytes - 1);
821  }
822  }
823 
824  //
825  // Load the destination ending address into the control block.
826  //
827  pControlTable[ui32ChannelStructIndex].pvDstEndAddr = pvDstAddr;
828 
829  //
830  // Write the new control word value.
831  //
832  pControlTable[ui32ChannelStructIndex].ui32Control = ui32Control;
833 }

859 void
860 uDMAChannelScatterGatherSet(uint32_t ui32ChannelNum,
861  uint32_t ui32TaskCount, void *pvTaskList,
862  uint32_t ui32IsPeriphSG)
863 {
864  tDMAControlTable *pControlTable;
865  tDMAControlTable *pTaskTable;
866 
867  //
868  // Check the parameters
869  //
870  ASSERT((ui32ChannelNum & 0xffff) < 32);
871  ASSERT(HWREG(UDMA_CTLBASE) != 0);
872  ASSERT(pvTaskList != 0);
873  ASSERT(ui32TaskCount <= 1024);
874  ASSERT(ui32TaskCount != 0);
875 
876  //
877  // In case a channel selector macro (like UDMA_CH0_USB0EP1RX) was
878  // passed as the ui32ChannelNum parameter, extract just the channel number
879  // from this parameter.
880  //
881  ui32ChannelNum &= 0x1f;
882 
883  //
884  // Get the base address of the control table.
885  //
886  pControlTable = (tDMAControlTable *)HWREG(UDMA_CTLBASE);
887 
888  //
889  // Get a handy pointer to the task list
890  //
891  pTaskTable = (tDMAControlTable *)pvTaskList;
892 
893  //
894  // Compute the ending address for the source pointer. This will be the
895  // last element of the last task in the task table
896  //
897  pControlTable[ui32ChannelNum].pvSrcEndAddr =
898  &pTaskTable[ui32TaskCount - 1].ui32Spare;
899 
900  //
901  // Compute the ending address for the destination pointer. This will be
902  // the end of the alternate structure for this channel.
903  //
904  pControlTable[ui32ChannelNum].pvDstEndAddr =
905  &pControlTable[ui32ChannelNum | UDMA_ALT_SELECT].ui32Spare;
906 
907  //
908  // Compute the control word. Most configurable items are fixed for
909  // scatter-gather. Item and increment sizes are all 32-bit and arb
910  // size must be 4. The count is the number of items in the task list
911  // times 4 (4 words per task).
912  //
913  pControlTable[ui32ChannelNum].ui32Control =
914  (UDMA_CHCTL_DSTINC_32 | UDMA_CHCTL_DSTSIZE_32 |
915  UDMA_CHCTL_SRCINC_32 | UDMA_CHCTL_SRCSIZE_32 |
916  UDMA_CHCTL_ARBSIZE_4 |
917  (((ui32TaskCount * 4) - 1) << UDMACHCTL_CHCTL_XFERSIZE_S) |
918  (ui32IsPeriphSG ? UDMA_CHCTL_XFERMODE_PER_SG :
919  UDMA_CHCTL_XFERMODE_MEM_SG));
920 }

986  //
987  return((ui32Control >> 4) + 1);
