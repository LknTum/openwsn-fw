275 
276  // On reception, when MODEMCTRL0.AUTOCRC is set, the CC2420 replaces the
277  // received CRC by:
278  // - [1B] the rssi, a signed value. The actual value in dBm is that - 45.
279  // - [1B] whether CRC checked (bit 7) and LQI (bit 6-0)
280  *rssi = *(bufRead+*lenRead-2);
281  *rssi -= 45;
282  //bufRead[*lenRead-3] = *(bufRead+*lenRead-2)-45;//*rssi;
283  *crc = ((*(bufRead+*lenRead-1))&0x80)>>7;
284  *lqi = (*(bufRead+*lenRead-1))&0x7f;
285 }
