122 void adaptive_sync_calculateCompensatedSlots(int16_t timeCorrection) {
123  bool isFirstSync; // is this the first sync after joining network?
124  uint16_t totalTimeCorrectionTicks; // how much error in ticks since last synchronization.
125 
126  if(adaptive_sync_vars.clockState == S_NONE) {
127  isFirstSync = TRUE;
128  } else {
129  isFirstSync = FALSE;
130  }
131  adaptive_sync_vars.elapsedSlots = ieee154e_asnDiff(&adaptive_sync_vars.oldASN);
132 
133  if(isFirstSync) {
134  if(timeCorrection > 1) {
135  adaptive_sync_vars.clockState = S_FASTER;
136  adaptive_sync_vars.compensationInfo_vars.compensationSlots = SYNC_ACCURACY*adaptive_sync_vars.elapsedSlots;
137  adaptive_sync_vars.compensationInfo_vars.compensationSlots /= timeCorrection;
138  } else {
139  if(timeCorrection < -1) {
140  adaptive_sync_vars.clockState = S_SLOWER;
141  adaptive_sync_vars.compensationInfo_vars.compensationSlots = SYNC_ACCURACY*adaptive_sync_vars.elapsedSlots;
142  adaptive_sync_vars.compensationInfo_vars.compensationSlots /= (-timeCorrection);
143  } else {
144  //timeCorrection = {-1,1}, it's not accurate when timeCorrection belongs to {-1,1}
145  //nothing is needed to do with this case.
146  }
147  }
148  } else {
149  if(adaptive_sync_vars.clockState == S_SLOWER) {
150  totalTimeCorrectionTicks = adaptive_sync_vars.compensateTicks;
151  totalTimeCorrectionTicks -= timeCorrection+adaptive_sync_vars.sumOfTC;
152  } else {
153  totalTimeCorrectionTicks = adaptive_sync_vars.compensateTicks;
154  totalTimeCorrectionTicks += timeCorrection+adaptive_sync_vars.sumOfTC;
155  }
156  if(totalTimeCorrectionTicks == 0) {
157  // totalTimeCorrectionTicks should be always positive if drift of clock is constant. if totalTimeCorrectionTIcks become zero, it means the drift changed for some reasons.
158  adaptive_sync_driftChanged();
159  } else {
160  adaptive_sync_vars.compensationInfo_vars.compensationSlots = SYNC_ACCURACY*adaptive_sync_vars.elapsedSlots;
161  adaptive_sync_vars.compensationInfo_vars.compensationSlots /= totalTimeCorrectionTicks;
162  }
163  }
164 
165  adaptive_sync_vars.compensationTimeout = adaptive_sync_vars.compensationInfo_vars.compensationSlots;
166 }

173 void adaptive_sync_countCompensationTimeout() {
174  uint16_t newSlotDuration;
175 
176  newSlotDuration = ieee154e_getSlotDuration();
177 
178  // if clockState is not set yet, don't compensate.
179  if (adaptive_sync_vars.clockState == S_NONE) {
180  return;
181  }
182 
183  if (adaptive_sync_vars.compensationTimeout == 0) {
184  return; // should not happen
185  }
186 
187  adaptive_sync_vars.compensationTimeout--;
188 
189  // when compensationTimeout, adjust current slot length
190  if(adaptive_sync_vars.compensationTimeout == 0) {
191  if(adaptive_sync_vars.clockState == S_SLOWER) {
192  newSlotDuration -= SYNC_ACCURACY;
193  adaptive_sync_vars.compensateTicks += SYNC_ACCURACY;
194  } else { // clock is fast
195  newSlotDuration += SYNC_ACCURACY;
196  adaptive_sync_vars.compensateTicks += SYNC_ACCURACY;
197  }
198  // update current slot duration and reload compensationTimeout
199  radio_setTimerPeriod(newSlotDuration);
200  adaptive_sync_vars.compensationTimeout = adaptive_sync_vars.compensationInfo_vars.compensationSlots;
