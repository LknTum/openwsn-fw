17 void pins_pinMode(char pin, pin_mode_t mode) {
18  portBASE_TYPE xRunningPrivileged;
19  char c;
20  unsigned long pwm_on;
21  // Make sure pin is valid
22  xSemaphoreTake(pin_configuration_mutex, portMAX_DELAY);
23 
24  // pin is 1-indexed
25  pin--;
26 
27  if ( pin < 0 || pin >= OPENMOTE_NUM_PINS || openmote_pin_gpio_bit[pin] == NOT_VALID_PIN ) goto error;
28 
29  // see if pin choice is supported
30  switch(mode) {
31  case INPUT: // already checked
32  case OUTPUT:
33  break;
34  case ANALOG_INPUT:
35  if ( openmote_adc_channels[pin] == NOT_VALID_PIN ) goto error;
36  break;
37  case ANALOG_OUTPUT:
38  if ( DAC_AOUT != pin ) goto error;
39  break;
40  case PWM:
41  if ( openmote_pwm_channels[pin] == NOT_VALID_PIN ) goto error;
42  break;
43  default:
44  goto error;
45  }
46 
47  // Valid pin, now see if we can claim it
48  if (openal_claim_pin(pin + 1) == SUCCESS) {
49  // see if we even need to configure anything
50  if ( openal_pin_registry[pin].mode == mode ) {
51  goto success;
52  }
53 
54  // we now need privileges!
55  xRunningPrivileged = prvRaisePrivilege();
56 
57  // switch out of the previous mode
58  switch(openal_pin_registry[pin].mode) {
59  case INPUT:
60  case OUTPUT:
61  case ANALOG_INPUT:
62  case ANALOG_OUTPUT: // power turns on and off automatically with PINSEL
63  break; // nothing we need to do
64  case PWM:
65  // turn off pwm pin
66  LPC_PWM1->PCR &= ~(1<<(openmote_pwm_channels[pin] + 9));
67  // check to see if any other pin was in PWM mode
68  for (c = 0; c < OPENMOTE_NUM_PINS; c++ ) {
69  if ( c == pin ) continue;
70  if ( openal_pin_registry[c].mode == PWM )
71  break;
72  }
73  if ( c == OPENMOTE_NUM_PINS ) { // turn off PWM module
74  LPC_PWM1->TCR = 0;
75  LPC_SC->PCONP &= ~(1<<6);
76  }
77  default:
78  break;
79  }
80 
81  openal_pin_registry[pin].mode = mode;
82  switch(mode) {
83  case INPUT:
84  LPC17xx_gpio_bank_addresses[openmote_pin_gpio_bank[pin]]->FIODIR &= ~(1<<openmote_pin_gpio_bit[pin]);
85  openmote_pin_function_select(openmote_pin_gpio_bank[pin],openmote_pin_gpio_bit[pin],0);
86  break;
87  case OUTPUT:
88  LPC17xx_gpio_bank_addresses[openmote_pin_gpio_bank[pin]]->FIODIR |= (1<<openmote_pin_gpio_bit[pin]);
89  openmote_pin_function_select(openmote_pin_gpio_bank[pin],openmote_pin_gpio_bit[pin],0);
90  break;
91  case ANALOG_INPUT:
92  openmote_pin_function_select(openmote_pin_gpio_bank[pin],openmote_pin_gpio_bit[pin], openmote_adc_pinsel_value[pin]);
93  case ANALOG_OUTPUT:
94  openmote_pin_function_select(openmote_pin_gpio_bank[pin],openmote_pin_gpio_bit[pin],DAC_PINSEL_VALUE);
95  pins_configure_pullups(pin+1, NONE);
96  pins_analogWrite(pin + 1, 0); // ends our privilege
97  case PWM:
98  // turn on PWM module if it is not already
99  pwm_on = (LPC_SC->PCONP & (1<<6));
100  if ( pwm_on == 0 ) {
101  LPC_SC->PCONP |= (1<<6);
102  LPC_PWM1->TCR = 0x09; // Turn counters on
103  }
104  openmote_pin_function_select(openmote_pin_gpio_bank[pin],openmote_pin_gpio_bit[pin], openmote_pwm_pinsel_value[pin]);
105  pins_pwm_write_adv(pin + 1, 0, 490); // ends our privileges
106  default:
107  break;
108  }
109 
110  // reset privileges!
111  portRESET_PRIVILEGE(xRunningPrivileged);
112 
113  goto success;
114  } else goto error;
115 
116  success:
117  xSemaphoreGive(pin_configuration_mutex);
118  return;
119 
120  error: // app will die for its atrocities
121  xSemaphoreGive(pin_configuration_mutex);
122  openal_kill_app(APP_CURRENT);
123  // shouldn't get to here
124 }

159 unsigned short pins_analogRead(char pin) {
160  portBASE_TYPE xRunningPrivileged;
161  unsigned short retval;
162  if ( openal_check_owner(pin) == SUCCESS ) {
163  pin--;
164 
165  xRunningPrivileged = prvRaisePrivilege();
166  xSemaphoreTake(adc_mutex, portMAX_DELAY);
167  LPC_SC->PCONP |= (1<<12); // enable ADC peripheral power
168 
169  xSemaphoreTake(adc_conversion_semaphore, 0); // make sure the semaphore is 0
170  LPC_ADC->ADINTEN = (1<<8); // interrupt on global DONE flag
171  LPC_ADC->ADCR = openmote_adc_channels[pin] | // channel
172  ((7) << 8) | // clock div 100/8 = 12.5MHz
173  (1 << 21) | // Power ON
174  (1 << 24); // start conversion NOW
175 
176  xSemaphoreTake(adc_conversion_semaphore, portMAX_DELAY); // wait for conversion to finish
177 
178  retval = ((LPC_ADC->ADGDR) >> 4) & 0x0FFF;
179 
180  LPC_ADC->ADCR = 0; // Disable ADC
181  LPC_SC->PCONP &= ~(1<<12); // enable ADC peripheral power
182  xSemaphoreGive(adc_mutex);
183  portRESET_PRIVILEGE(xRunningPrivileged);
184 
185  return retval;
186  } else {
187  openal_kill_app(APP_CURRENT);
188  return 0; // shouldn't get here
189  }
