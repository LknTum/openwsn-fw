96 {
97  uint8_t ui8Err;
98  uint32_t ui32N, ui32I;
99 
100  if(psMd == NULL)
101  {
102  return (SHA256_NULL_ERROR);
103  }
104  if(ui8In == NULL)
105  {
106  return (SHA256_NULL_ERROR);
107  }
108 
109  if(psMd->curlen > sizeof(psMd->buf))
110  {
111  return (SHA256_INVALID_PARAM);
112  }
113 
114  g_ui8CurrentAESOp = AES_SHA256;
115  if(ui32InLen > 0 && psMd->new_digest == true)
116  {
117  if(psMd->curlen == 0 && ui32InLen > SHA256_BLOCK_SIZE)
118  {
119  for(ui32I = 0; ui32I < SHA256_BLOCK_SIZE; ui32I++)
120  {
121  psMd->buf[psMd->curlen + ui32I] = ui8In[ui32I];
122  }
123  if((ui8Err = SHA256HashNew(psMd, (uint8_t *)psMd->buf,
124  (uint8_t *)psMd->state)) != SHA256_SUCCESS)
125  {
126  g_ui8CurrentAESOp = AES_NONE;
127  return (ui8Err);
128  }
129  psMd->new_digest = false;
130  psMd->length += SHA256_BLOCK_SIZE * 8;
131  ui32InLen -= SHA256_BLOCK_SIZE;
132  ui8In += SHA256_BLOCK_SIZE;
133 
134  }
135  else
136  {
137  ui32N = MIN(ui32InLen, (SHA256_BLOCK_SIZE - psMd->curlen));
138  for(ui32I = 0; ui32I < ui32N; ui32I++)
139  {
140  psMd->buf[psMd->curlen + ui32I] = ui8In[ui32I];
141  }
142  psMd->curlen += ui32N;
143  ui8In += ui32N;
144  ui32InLen -= ui32N;
145  if(psMd->curlen == SHA256_BLOCK_SIZE && ui32InLen > 0)
146  {
147  if((ui8Err = SHA256HashNew(psMd, (uint8_t *)psMd->buf,
148  (uint8_t *)psMd->state)) != SHA256_SUCCESS)
149  {
150  g_ui8CurrentAESOp = AES_NONE;
151  return (ui8Err);
152  }
153  psMd->new_digest = false;
154  psMd->length += 8 * SHA256_BLOCK_SIZE;
155  psMd->curlen = 0;
156  }
157  }
158  }
159 
160  while(ui32InLen > 0 && psMd->new_digest == false)

219 uint8_t SHA256Done(tSHA256State * psMd, uint8_t *ui8Out)
220 {
221  uint8_t ui8Err;
222  if(psMd == NULL || ui8Out == NULL)
223  {
224  return (SHA256_NULL_ERROR);
225  }
226 
227  if(psMd->curlen > sizeof(psMd->buf))
228  {
229  return (SHA256_INVALID_PARAM);
230  }
231 
232  g_ui8CurrentAESOp = AES_SHA256;
233 
234  // increase the length of the message
235  psMd->length += psMd->curlen * 8;
236  psMd->final_digest = true;
237  if(psMd->new_digest == true)
238  {
239  if((ui8Err = SHA256HashNew(psMd, (uint8_t *)psMd->buf,
240  (uint8_t *)ui8Out)) != SHA256_SUCCESS)
241  {
242  g_ui8CurrentAESOp = AES_NONE;
243  return (ui8Err);
244  }
245  }
246  else
247  {
248  if((ui8Err = SHA256HashResume(psMd, (uint8_t *)psMd->buf,
249  (uint8_t *)ui8Out)) != SHA256_SUCCESS)
250  {
251  g_ui8CurrentAESOp = AES_NONE;
252  return (ui8Err);
253  }
254  }
255  psMd->new_digest = false;
256  psMd->final_digest = false;
257 
258  g_ui8CurrentAESOp = AES_NONE;
259  return (SHA256_SUCCESS);

273 uint8_t SHA256HashNew(tSHA256State * psMd, uint8_t *ui8In, uint8_t *ui8Out)
274 {
275  // workaround for AES registers not retained after PM2
276  IntDisable(INT_AES);
277  HWREG(AES_CTRL_INT_CFG) = AES_CTRL_INT_CFG_LEVEL;
278  HWREG(AES_CTRL_INT_EN) = (AES_CTRL_INT_EN_RESULT_AV |
279  AES_CTRL_INT_EN_DMA_IN_DONE);
280 
281  // configure master control module
282  // enable DMA path to the SHA-256 engine + Digest readout
283  HWREG(AES_CTRL_ALG_SEL) = (AES_CTRL_ALG_SEL_TAG | AES_CTRL_ALG_SEL_HASH);
284  // clear any outstanding events
285  HWREG(AES_CTRL_INT_CLR) = AES_CTRL_INT_CLR_RESULT_AV;
286 
287  // configure hash engine
288  // indicate start of a new hash session and SHA256
289  HWREG(AES_HASH_MODE_IN) = (AES_HASH_MODE_IN_SHA256_MODE |
290  AES_HASH_MODE_IN_NEW_HASH);
291 
292  // if the final digest is required (pad the input DMA data),
293  // write the following register
294  //
295  if(psMd->final_digest)
296  {
297  // write length of the message (lo)
298  HWREG(AES_HASH_LENGTH_IN_L) = (uint32_t)psMd->length;
299  // write length of the message (hi)
300  HWREG(AES_HASH_LENGTH_IN_H) = (uint32_t)(psMd->length >> 16);
301  // pad the DMA-ed data
302  HWREG(AES_HASH_IO_BUF_CTRL) = AES_HASH_IO_BUF_CTRL_PAD_DMA_MESSAGE;
303  }
304 
305  // enable DMA channel 0 for message data
306  HWREG(AES_DMAC_CH0_CTRL) |= AES_DMAC_CH0_CTRL_EN;
307  // base address of the data in ext. memory
308  HWREG(AES_DMAC_CH0_EXTADDR) = (uint32_t)ui8In;
309  if(psMd->final_digest)
310  {
311  // input data length in bytes, equal to the message
312  HWREG(AES_DMAC_CH0_DMALENGTH) = psMd->curlen;
313  }
314  else
315  {
316  HWREG(AES_DMAC_CH0_DMALENGTH) = SHA256_BLOCK_SIZE;
317  }
318 
319  // enable DMA channel 1 for result digest
320  HWREG(AES_DMAC_CH1_CTRL) |= AES_DMAC_CH1_CTRL_EN;
321  // base address of the digest buffer
322  HWREG(AES_DMAC_CH1_EXTADDR) = (uint32_t)ui8Out;
323  // length of the result digest
324  HWREG(AES_DMAC_CH1_DMALENGTH) = SHA256_OUTPUT_LEN;
325 
326  // wait for completion of the operation
327  do
328  {
329  ASM_NOP;
330  }
331  while(!(HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_RESULT_AV));
332 
333 
334  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_DMA_BUS_ERR))
335  {
336  return (AES_DMA_BUS_ERROR);
337  }
338 
339  // clear the interrupt
340  HWREG(AES_CTRL_INT_CLR) = (AES_CTRL_INT_CLR_DMA_IN_DONE |
341  AES_CTRL_INT_CLR_RESULT_AV);
342  // disable master control/DMA clock
343  HWREG(AES_CTRL_ALG_SEL) = 0x00000000;
344  // clear mode
345  HWREG(AES_AES_CTRL) = 0x00000000;
346 
347  return (SHA256_SUCCESS);
348 }

362 uint8_t SHA256HashResume(tSHA256State * psMd, uint8_t *ui8In, uint8_t *ui8Out)
363 {
364  IntDisable(INT_AES);
365  // workaround for AES registers not retained after PM2
366  HWREG(AES_CTRL_INT_CFG) = AES_CTRL_INT_CFG_LEVEL;
367  HWREG(AES_CTRL_INT_EN) = (AES_CTRL_INT_EN_RESULT_AV |
368  AES_CTRL_INT_EN_DMA_IN_DONE);
369 
370  // configure master control module and enable
371  // the DMA path to the SHA-256 engine
372  //
373  HWREG(AES_CTRL_ALG_SEL) = AES_CTRL_ALG_SEL_HASH;
374 
375  // clear any outstanding events
376  HWREG(AES_CTRL_INT_CLR) = AES_CTRL_INT_CLR_RESULT_AV;
377 
378  // configure hash engine
379  // indicate the start of a resumed hash session and SHA256
380  HWREG(AES_HASH_MODE_IN) = AES_HASH_MODE_IN_SHA256_MODE;
381 
382  // if the final digest is required (pad the input DMA data)
383  if(psMd->final_digest)
384  {
385  // write length of the message (lo)
386  HWREG(AES_HASH_LENGTH_IN_L) = (uint32_t)psMd->length;
387  // write length of the message (hi)
388  HWREG(AES_HASH_LENGTH_IN_H) = (uint32_t)(psMd->length >> 16);
389  }
390 
391  // write the initial digest
392  HWREG(AES_HASH_DIGEST_A) = (uint32_t)psMd->state[0];
393  HWREG(AES_HASH_DIGEST_B) = (uint32_t)psMd->state[1];
394  HWREG(AES_HASH_DIGEST_C) = (uint32_t)psMd->state[2];
395  HWREG(AES_HASH_DIGEST_D) = (uint32_t)psMd->state[3];
396  HWREG(AES_HASH_DIGEST_E) = (uint32_t)psMd->state[4];
397  HWREG(AES_HASH_DIGEST_F) = (uint32_t)psMd->state[5];
398  HWREG(AES_HASH_DIGEST_G) = (uint32_t)psMd->state[6];
399  HWREG(AES_HASH_DIGEST_H) = (uint32_t)psMd->state[7];
400 
401  // If final digest, pad the DMA-ed data
402  if(psMd->final_digest)
403  {
404  HWREG(AES_HASH_IO_BUF_CTRL) = AES_HASH_IO_BUF_CTRL_PAD_DMA_MESSAGE;
405  }
406 
407  // enable DMA channel 0 for message data
408  HWREG(AES_DMAC_CH0_CTRL) |= AES_DMAC_CH0_CTRL_EN;
409  // base address of the data in ext. memory
410  HWREG(AES_DMAC_CH0_EXTADDR) = (uint32_t)ui8In;
411  // input data length in bytes, equal to the message
412  if(psMd->final_digest)
413  {
414  HWREG(AES_DMAC_CH0_DMALENGTH) = psMd->curlen;
415  }
416  else
417  {
418  HWREG(AES_DMAC_CH0_DMALENGTH) = SHA256_BLOCK_SIZE;
419  }
420 
421  // wait for completion of the operation
422  do
423  {
424  ASM_NOP;
425  }
426  while(!(HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_RESULT_AV));
427 
428  // check for any DMA Bus errors
429  if((HWREG(AES_CTRL_INT_STAT) & AES_CTRL_INT_STAT_DMA_BUS_ERR))
430  {
431  return (AES_DMA_BUS_ERROR);
432  }
433 
434  // read digest
435  ((uint32_t *)ui8Out)[0] = HWREG(AES_HASH_DIGEST_A);
436  ((uint32_t *)ui8Out)[1] = HWREG(AES_HASH_DIGEST_B);
437  ((uint32_t *)ui8Out)[2] = HWREG(AES_HASH_DIGEST_C);
438  ((uint32_t *)ui8Out)[3] = HWREG(AES_HASH_DIGEST_D);
439  ((uint32_t *)ui8Out)[4] = HWREG(AES_HASH_DIGEST_E);
440  ((uint32_t *)ui8Out)[5] = HWREG(AES_HASH_DIGEST_F);
441  ((uint32_t *)ui8Out)[6] = HWREG(AES_HASH_DIGEST_G);
442  ((uint32_t *)ui8Out)[7] = HWREG(AES_HASH_DIGEST_H);
443 
444  // acknowledge reading of the digest
445  HWREG(AES_HASH_IO_BUF_CTRL) = AES_HASH_IO_BUF_CTRL_OUTPUT_FULL;
446 
447  // clear the interrupt
448  HWREG(AES_CTRL_INT_CLR) = (AES_CTRL_INT_CLR_DMA_IN_DONE |
449  AES_CTRL_INT_CLR_RESULT_AV);
450  // acknowledge result and clear interrupts
451  // disable master control/DMA clock
452  HWREG(AES_CTRL_ALG_SEL) = 0x00000000;
453  // clear mode
454  HWREG(AES_AES_CTRL) = 0x00000000;
455 
456  return (SHA256_SUCCESS);
457 }
