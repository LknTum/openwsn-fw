205  // step 1. Identify expired timers

207  if (opentimers_vars.timersBuf[id].isrunning==TRUE) {
208 
209  if(opentimers_vars.currentTimeout >= opentimers_vars.timersBuf[id].ticks_remaining) {
210  // this timer has expired
211  //check to see if we have completed the whole timer, or we're just wrapping around the max 16bit value
212  if(opentimers_vars.timersBuf[id].wraps_remaining == 0){

219  opentimers_vars.timersBuf[id].ticks_remaining = (opentimers_vars.timersBuf[id].period_ticks) % MAX_TICKS_IN_SINGLE_CLOCK;
220  }else{
221  opentimers_vars.timersBuf[id].ticks_remaining = MAX_TICKS_IN_SINGLE_CLOCK;
222  }
223  }
224  } else {
225  // this timer is not expired

228  opentimers_vars.timersBuf[id].ticks_remaining -= opentimers_vars.currentTimeout;
229  }
230  }
231  }
