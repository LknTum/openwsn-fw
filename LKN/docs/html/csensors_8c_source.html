174  OpenQueueEntry_t* msg,
175  coap_header_iht* coap_header,
176  coap_option_iht* coap_options
177  ) {
178  owerror_t outcome;
179  uint8_t id;
180  uint8_t i;
181  uint32_t period;
182 
183 
184  switch (coap_header->Code) {
185  case COAP_CODE_REQ_GET:
186  // reset packet payload
187  msg->payload = &(msg->packet[127]);
188  msg->length = 0;
189 
190  if (coap_options[1].type != COAP_OPTION_NUM_URIPATH) {
191 
192  // have CoAP module write links to csensors resources
193  opencoap_writeLinks(msg,COMPONENT_CSENSORS);
194 
195  packetfunctions_reserveHeaderSize(msg,1);
196  msg->payload[0] = COAP_PAYLOAD_MARKER;
197 
198  // add return option
199  packetfunctions_reserveHeaderSize(msg,2);
200  msg->payload[0] = COAP_OPTION_NUM_CONTENTFORMAT << 4 | 1;
201  msg->payload[1] = COAP_MEDTYPE_APPLINKFORMAT;
202 
203  } else {
204  for(id=0;id<csensors_vars.numCsensors;id++) {
205  if (
206  memcmp(
207  coap_options[1].pValue,
208  csensors_vars.csensors_resource[id].desc.path1val,
209  csensors_vars.csensors_resource[id].desc.path1len
210  )==0
211  ) {
212  break;
213  }
214  }
215  csensors_fillpayload(msg,id);
216  packetfunctions_reserveHeaderSize(msg,2);
217  msg->payload[0] = (COAP_OPTION_NUM_CONTENTFORMAT << 4) | 1;
218  msg->payload[1] = COAP_MEDTYPE_APPOCTETSTREAM;
219  }
220  // set the CoAP header
221  coap_header->Code = COAP_CODE_RESP_CONTENT;
222 
223  outcome = E_SUCCESS;
224 
225  break;
226 
227  case COAP_CODE_REQ_PUT:
228  period = 0;
229  for(i=0;i<msg->length;i++) {
230  if ((uint8_t)msg->payload[i]>=(uint8_t)'0' && (uint8_t)msg->payload[i]<=(uint8_t)'0'+9) {
231  period = period*10 + (uint8_t)msg->payload[i] - (uint8_t)'0';
232  } else {
233  period = 0;
234  break;
235  }
236  }
237  if (coap_options[1].type == COAP_OPTION_NUM_URIPATH) {
238  for(id=0;id<csensors_vars.numCsensors;id++) {
239  if (
240  memcmp(
241  coap_options[1].pValue,
242  csensors_vars.csensors_resource[id].desc.path1val,
243  csensors_vars.csensors_resource[id].desc.path1len
244  )==0
245  ) {
246  break;
247  }
248  }
249  csensors_setPeriod(period,id);
250  } else {
251  for(id=0;id<csensors_vars.numCsensors;id++) {
252  csensors_setPeriod(period,id);
253  }
254  }
255 
256  // reset packet payload
257  msg->payload = &(msg->packet[127]);
258  msg->length = 0;
259  // set the CoAP header
260  coap_header->Code = COAP_CODE_RESP_CHANGED;
261 
262  outcome = E_SUCCESS;
263  break;
264 
265  default:
266  outcome = E_FAIL;
267  break;
268  }
269
