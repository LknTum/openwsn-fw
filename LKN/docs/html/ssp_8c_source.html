47 static void setSSPclock (LPC_SSP_TypeDef *SSPx, uint32_t target_clock)
48 {
49  uint32_t prescale, cr0_div, cmp_clk, ssp_clk;
50 
51 
52  /* The SSP clock is derived from the (main system oscillator / 2),
53  so compute the best divider from that clock */
54  if (SSPx == LPC_SSP0){
55  ssp_clk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_SSP0);
56  } else if (SSPx == LPC_SSP1) {
57  ssp_clk = CLKPWR_GetPCLK (CLKPWR_PCLKSEL_SSP1);
58  } else {
59  return;
60  }
61 
62  /* Find closest divider to get at or under the target frequency.
63  Use smallest prescale possible and rely on the divider to get
64  the closest target frequency */
65  cr0_div = 0;
66  cmp_clk = 0xFFFFFFFF;
67  prescale = 2;
68  while (cmp_clk > target_clock)
69  {
70  cmp_clk = ssp_clk / ((cr0_div + 1) * prescale);
71  if (cmp_clk > target_clock)
72  {
73  cr0_div++;
74  if (cr0_div > 0xFF)
75  {
76  cr0_div = 0;
77  prescale += 2;
78  }
79  }
80  }
81 
82  /* Write computed prescaler and divider back to register */
83  SSPx->CR0 &= (~SSP_CR0_SCR(0xFF)) & SSP_CR0_BITMASK;
84  SSPx->CR0 |= (SSP_CR0_SCR(cr0_div)) & SSP_CR0_BITMASK;
85  SSPx->CPSR = prescale & SSP_CPSR_BITMASK;
86 }

108 void SSP_Init(LPC_SSP_TypeDef *SSPx, SSP_CFG_Type *SSP_ConfigStruct)
109 {
110  uint32_t tmp;
111 
112 
113

115  /* Set up clock and power for SSP0 module */
116  CLKPWR_ConfigPPWR (CLKPWR_PCONP_PCSSP0, ENABLE);
117  } else if(SSPx == LPC_SSP1) {
118  /* Set up clock and power for SSP1 module */

120  } else {
121  return;

123 
124  /* Configure SSP, interrupt is disable, LoopBack mode is disable,
125  * SSP is disable, Slave output is disable as default
126  */
127  tmp = ((SSP_ConfigStruct->CPHA) | (SSP_ConfigStruct->CPOL) \
128  | (SSP_ConfigStruct->FrameFormat) | (SSP_ConfigStruct->Databit))
129  & SSP_CR0_BITMASK;
130  // write back to SSP control register
131  SSPx->CR0 = tmp;
132 
133  tmp = SSP_ConfigStruct->Mode & SSP_CR1_BITMASK;
134  // Write back to CR1
135  SSPx->CR1 = tmp;
136 
137  // Set clock rate for SSP peripheral
