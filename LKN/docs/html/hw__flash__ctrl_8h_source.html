122 #define FLASH_CTRL_FCTL_ABORT 0x00000020 // Abort status This bit is set to
123  // 1 when a write sequence or page
124  // erase is aborted. An operation
125  // is aborted when the accessed
126  // page is locked. Cleared when a
127  // write or page erase is started.
128  // If a write operation times out
129  // (because the FWDATA register is
130  // not written fast enough), the
131  // ABORT bit is not set even if the
132  // page is locked. If a page erase
133  // and a write operation are
134  // started simultaneously, the
135  // ABORT bit reflects the status of
136  // the last write operation. For
137  // example, if the page is locked
138  // and the write times out, the
139  // ABORT bit is not set because

144 #define FLASH_CTRL_FCTL_CM_M 0x0000000C // Cache Mode Disabling the cache
145  // increases the power consumption
146  // and reduces performance.
147  // Prefetching improves performance
148  // at the expense of a potential
149  // increase in power consumption.
150  // Real-time mode provides
151  // predictable flash read access
152  // time, the execution time is
153  // equal to cache disabled mode,
154  // but the power consumption is
155  // lower. 00: Cache disabled 01:
156  // Cache enabled 10: Cache enabled,
157  // with prefetch 11: Real-time mode
158  // Note: The read value always
159  // represents the current cache
160  // mode. Writing a new cache mode
161  // starts a cache mode change
162  // request that does not take
163  // effect until the controller is
164  // ready. Writes to this register
165  // are ignored if there is a
166  // current cache change request in
167  // progress.

196 // FLASH_CTRL_FADDR register.

199 #define FLASH_CTRL_FADDR_FADDR_M \
200  0x0001FFFF // Bit number [16:9] selects one
201  // of 256 pages for page erase. Bit
202  // number [8:7] selects one of the
203  // 4 row in a given page Bit number
204  // [6:1] selects one of the 64-bit
205  // wide locations in a give row.
206  // Bit number [0] will select
207  // upper/lower 32-bits in a given
208  // 64-bit location - 64Kbytes -->
209  // Bits [16:14] will always be 0. -
210  // 128Kbytes --> Bits [16:15] will
211  // always be 0. - 256Kbytes --> Bit
212  // [16] will always be 0. -
213  // 384/512Kbytes --> All bits
214  // written and valid. Writes to
215  // this register will be ignored
216  // when any of FCTL.WRITE and
217  // FCTL.ERASE is set. FADDR should
218  // be written with byte addressable
219  // location of the Flash to be
220  // programmed. Read back value
221  // always reflects a 32-bit aligned
222  // address. When the register is
223  // read back, the value that was
224  // written to FADDR gets right
225  // shift by 2 to indicate 32-bit
226  // aligned address. In other words
227  // lower 2 bits are discarded while
228  // reading back the register. Out
229  // of range address results in roll
230  // over. There is no status signal
231  // generated by flash controller to
232  // indicate this. Firmware is
233  // responsible to managing the
234  // addresses correctly.
235 
236 #define FLASH_CTRL_FADDR_FADDR_S 0
