213 void USB_Reset_Service (
214  PTR_USB_DEV_EVENT_STRUCT event /* [IN] Pointer to USB Event Structure */
215 )
216 {
217 
218  USB_EP_STRUCT ep_struct;
219  uint_8 err;
220 
221  /* Initialize the endpoint 0 in both directions */
222  ep_struct.direction = USB_SEND;
223  ep_struct.ep_num = CONTROL_ENDPOINT;
224  ep_struct.size = CONTROL_MAX_PACKET_SIZE;
225  ep_struct.type = USB_CONTROL_PIPE;
226 
227  /* Deinit Endpoint in case its already initialized */
228  err = _usb_device_deinit_endpoint(&(event->controller_ID),
229  ep_struct.ep_num, ep_struct.direction);
230  /* now initialize the endpoint */
231  err = _usb_device_init_endpoint(&(event->controller_ID),
232  ep_struct.ep_num, (uint_16)ep_struct.size, ep_struct.direction, ep_struct.type, TRUE);
233 
234  ep_struct.direction = USB_RECV;
235  /* Deinit Endpoint in case its already initialized */
236  (void)_usb_device_deinit_endpoint(&(event->controller_ID),
237  ep_struct.ep_num, ep_struct.direction);
238  /* now initialize the endpoint */
239  (void)_usb_device_init_endpoint(&(event->controller_ID),
240  ep_struct.ep_num, (uint_16)ep_struct.size, ep_struct.direction, ep_struct.type, TRUE);
241 
242  /* set the default device state */
243  (void)_usb_device_set_status(&(event->controller_ID), USB_STATUS_DEVICE_STATE,
244  USB_STATE_DEFAULT);
245 
246  /* set the default device state */
247  (void)_usb_device_set_status(&(event->controller_ID), USB_STATUS_DEVICE,
248  SELF_POWERED >> SELF_POWER_BIT_SHIFT);
249 
250  /* set the EndPoint Status as Idle in the device layer */
251  (void)_usb_device_set_status(&(event->controller_ID),
252  (uint_8)(USB_STATUS_ENDPOINT | CONTROL_ENDPOINT |
253  (USB_SEND << USB_COMPONENT_DIRECTION_SHIFT)),
254  USB_STATUS_IDLE);
255  /* let the application know that bus reset has taken place */
256  g_class_callback(event->controller_ID, USB_APP_BUS_RESET, NULL);
257 
258  UNUSED(err);
259  return;
260 }
