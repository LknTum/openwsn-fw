1021  unsigned char hgo_val,

1028  )
1029 {
1030 #ifndef MCU_MK70F12
1031  /*
1032  * First move to FBE mode
1033  * Enable external oscillator, RANGE=0, HGO=, EREFS=, LP=, IRCS=
1034  */
1035  #if ((defined MCU_MK60N512VMD100) || (defined MCU_MK53N512CMD100))
1036  MCG_C2 = 0;
1037  #else

1042  #endif
1043  #endif

1047  * CLKS = 2, FRDIV = 3, IREFS = 0, IRCLKEN = 0, IREFSTEN = 0

1054  #elif defined (MCU_MK40N512VMD100)

1056  #endif

1058  #ifndef MCU_MK20D5

1060  while (MCG_S & MCG_S_IREFST_MASK) {};
1061  #endif

1068  * Configure PLL Reference Divider, PLLCLKEN = 0, PLLSTEN = 0, PRDIV = 0x18
