420 /**************************************************************************/
433 void USB_Control_Service (
434  PTR_USB_DEV_EVENT_STRUCT event /* [IN] Pointer to USB Event Structure */
435 )
436 {
437  uint_8 device_state = 0;
438  uint_8 status = USBERR_INVALID_REQ_TYPE;
439  uint_8_ptr data = NULL;
440  USB_PACKET_SIZE size;
441 
442  /* get the device state */
443  (void)_usb_device_get_status(&(event->controller_ID), USB_STATUS_DEVICE_STATE,
444  &device_state);
445 
446  if (event->setup == TRUE)
447  {
448  /* get the setup packet */
449  (void)memcpy(&g_setup_pkt, event->buffer_ptr, USB_SETUP_PKT_SIZE);
450 
451  /* take care of endianess of the 16 bt fields correctly */
452  g_setup_pkt.index = BYTE_SWAP16(g_setup_pkt.index);
453  g_setup_pkt.value = BYTE_SWAP16(g_setup_pkt.value);
454  g_setup_pkt.length = BYTE_SWAP16(g_setup_pkt.length);
455 
456  /* if the request is standard request */
457  if ((g_setup_pkt.request_type & USB_REQUEST_CLASS_MASK) ==
458  USB_REQUEST_CLASS_STRD)
459  {
460  /* if callback is not NULL */
461  if (g_standard_request[g_setup_pkt.request] != NULL)
462  {
463  /* if the request is valid in this device state */
464  if((device_state < USB_STATE_POWERED) &&
465  (g_validate_request[g_setup_pkt.request][device_state]
466  ==TRUE))
467  {
468  /* Standard Request function pointers */
469  status = g_standard_request[g_setup_pkt.request]
470  (event->controller_ID, &g_setup_pkt, &data,
471  (USB_PACKET_SIZE *)&size);
472  }
473  }
474  }
475  else /* for Class/Vendor requests */
476  {
477  /*get the length from the setup_request*/
478  size = (USB_PACKET_SIZE)g_setup_pkt.length;
479 
480  if( (size != 0) &&
481  ((g_setup_pkt.request_type & USB_DATA_DIREC_MASK) ==
482  USB_DATA_TO_DEVICE) )
483  {
484  (void)memcpy(ext_req_to_host, &g_setup_pkt, USB_SETUP_PKT_SIZE);
485 
486  /* expecting host to send data (OUT TRANSACTION)*/
487  (void)_usb_device_recv_data(&(event->controller_ID),
488  CONTROL_ENDPOINT, ext_req_to_host+USB_SETUP_PKT_SIZE,
489  (USB_PACKET_SIZE)(size));
490  return;
491  }
492  else if(g_other_req_callback != NULL)/*call class/vendor request*/
493  {
494  status = g_other_req_callback(event->controller_ID,
495  &g_setup_pkt, &data, (USB_PACKET_SIZE*)&size);
496  }
497  }
498 
499  USB_Control_Service_Handler(event->controller_ID,
500  status, &g_setup_pkt, &data, (USB_PACKET_SIZE*)&size);
501  }
502  /* if its not a setup request */
503  else if(device_state == USB_STATE_PENDING_ADDRESS)
504  {
505  /* Device state is PENDING_ADDRESS */
506  /* Assign the new adddress to the Device */
507  (void)USB_Strd_Req_Assign_Address(event->controller_ID);
508  return;
509  }
510  else if( ((g_setup_pkt.request_type &
511  USB_DATA_DIREC_MASK) == USB_DATA_TO_DEVICE) &&
512  (event->direction == USB_RECV) )
513  {
514  /* execution enters Control Service because of
515  OUT transaction on CONTROL_ENDPOINT*/
516  if(g_other_req_callback != NULL)
517  { /* class or vendor request */
518  size = (USB_PACKET_SIZE)(event->len + USB_SETUP_PKT_SIZE);
519  status = g_other_req_callback(event->controller_ID,
520  (USB_SETUP_STRUCT*)(ext_req_to_host), &data,
521  (USB_PACKET_SIZE*)&size);
522  }
523 
524  USB_Control_Service_Handler(event->controller_ID,
525  status, &g_setup_pkt, &data,
526  (USB_PACKET_SIZE*)&size);
527  }
528  return;
529 }

549 static void USB_Control_Service_Handler (
550  uint_8 controller_ID, /* [IN] Controller ID */
551  uint_8 status, /* [IN] Device Status */
552  USB_SETUP_STRUCT * setup_packet, /* [IN] Setup packet received */
553  uint_8_ptr *data, /* [OUT] Data to be send back */
554  USB_PACKET_SIZE *size /* [OUT] Size to be returned */
555 )
556 {
557  if(status == USBERR_INVALID_REQ_TYPE)
558  {
559  /* incase of error Stall endpoint */
560  (void)_usb_device_set_status(&controller_ID,
561  (uint_8)(USB_STATUS_ENDPOINT | CONTROL_ENDPOINT |
562  (USB_SEND << USB_COMPONENT_DIRECTION_SHIFT)),
563  USB_STATUS_STALLED);
564  }
565  else /* Need to send Data to the USB Host */
566  {
567  /* send the data prepared by the handlers.*/
568  if(*size > setup_packet->length)
569  {
570  *size = (USB_PACKET_SIZE)setup_packet->length;
571  }
572 
573  /* send the data to the host */
574  (void)USB_Class_Send_Data(controller_ID,
575  CONTROL_ENDPOINT, *data, *size);
576  if((setup_packet->request_type & USB_DATA_DIREC_MASK) ==
577  USB_DATA_TO_HOST)
578  { /* Request was to Get Data from device */
579  /* setup rcv to get status from host */
580  (void)_usb_device_recv_data(&controller_ID,
581  CONTROL_ENDPOINT,NULL,0);
582  }
583 
584  }
585  return;
586 }

606 static uint_8 USB_Strd_Req_Get_Status (
607  uint_8 controller_ID, /* [IN] Controller ID */
608  USB_SETUP_STRUCT * setup_packet, /* [IN] Setup packet received */
609  uint_8_ptr *data, /* [OUT] Data to be send back */
610  USB_PACKET_SIZE *size /* [OUT] Size to be returned */
611 )
612 {
613  uint_8_ptr pu8ConfigDesc = NULL;
614  uint_16 u16DescSize;
615  uint_8 interface, endpoint = 0;
616  uint_8 status;
617  uint_8 device_state = USB_STATE_POWERED;
618 
619 
620  status = _usb_device_get_status(&controller_ID, USB_STATUS_DEVICE_STATE, &device_state);
621 
622  if((setup_packet->request_type & USB_REQUEST_SRC_MASK) == USB_REQUEST_SRC_DEVICE)
623  {
624  #ifdef OTG_BUILD
625  if(setup_packet->index == USB_WINDEX_OTG_STATUS_SEL)
626  {
627  uint_8 temp;
628  /* Request for Device */
629  status = _usb_device_get_status(&controller_ID, USB_STATUS_OTG, &temp);
630  g_std_framework_data = (uint_16)temp;
631  g_std_framework_data &= GET_STATUS_OTG_MASK;
632  g_std_framework_data = BYTE_SWAP16(g_std_framework_data);
633  *size = OTG_STATUS_SIZE;
634  }
635  else
636  #endif
637  {
638  uint_8 device_state1;
639 
640  /* Request for Device */
641  status = _usb_device_get_status(&controller_ID, USB_STATUS_DEVICE, &device_state1);
642  g_std_framework_data = (uint_16)device_state1;
643  g_std_framework_data &= GET_STATUS_DEVICE_MASK;
644  g_std_framework_data = BYTE_SWAP16(g_std_framework_data);
645 
646  /* Set Self Powered bit in device state according to configuration */
647  status = USB_Desc_Get_Descriptor(controller_ID, USB_CONFIG_DESCRIPTOR,
648  0, 0, &pu8ConfigDesc, (USB_PACKET_SIZE *)&u16DescSize);
649 
650  if((pu8ConfigDesc[7] & SELF_POWERED) != 0)
651  {
652  g_std_framework_data |= 0x0100;
653  }
654  else
655  {
656  g_std_framework_data &= ~0x0100;
657  }
658  *size = DEVICE_STATUS_SIZE;
659  }
660  }
661  else if((setup_packet->request_type & USB_REQUEST_SRC_MASK) == USB_REQUEST_SRC_INTERFACE)
662  {
663  /* Request for Interface */
664  interface = (uint_8) setup_packet->index;
665  if((device_state == USB_STATE_ADDRESS) && (interface > 0))
666  {
667  status = USBERR_INVALID_REQ_TYPE;
668  }
669  else
670  {
671  /* Interface requests always return 0 */
672  g_std_framework_data = 0x0000;
673  *size = INTERFACE_STATUS_SIZE;
674  }
675  }
676  else if((setup_packet->request_type & USB_REQUEST_SRC_MASK) == USB_REQUEST_SRC_ENDPOINT)
677  {
678  /* Request for Endpoint */
679  if((device_state == USB_STATE_ADDRESS) && (endpoint > 0))
680  {
681  status = USBERR_INVALID_REQ_TYPE;
682  }
683  else
684  {
685  uint_8 device_state2;
686  endpoint = (uint_8)(((uint_8) setup_packet->index) |
687  USB_STATUS_ENDPOINT);
688  status = _usb_device_get_status(&controller_ID,
689  endpoint,
690  &device_state2);
691  g_std_framework_data = (uint_16)device_state2;
692  g_std_framework_data = BYTE_SWAP16(g_std_framework_data);
693  *size = ENDP_STATUS_SIZE;
694 
695  /* All other fields except HALT must be zero */
696  g_std_framework_data &= 0x0100;
697  }
698  }
699 
700  *data = (uint_8_ptr)&g_std_framework_data;
701  return status;
702 }
