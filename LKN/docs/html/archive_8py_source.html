10 # Archive nesdoc information from a given compilation in a nesdoc repository.
11 #
12 # A nesdoc repository is a directory containing separate XML files for each
13 # known interface and component (in the "interfaces" and "components"
14 # subdirectories respectively). These files have basically the same format
15 # as a single interfacedef and component element from the nesC XML dump
16 # format, with the following changes:
17 # - all "interfacedef", "interfacedef-ref", "component" and "component-ref"
18 # elements have an extra "nicename" attribute which stores the full
19 # `package name' for the corresponding element (see below for details)
20 # - components have two extra sub-elements:
21 # o "specification": a list of the "interface" and "function"
22 # elements that form the components specification
23 # o "referenced": the "interface" and "function" elements referenced
24 # from the wiring graph
25 # the "component" elements referenced from these "interface" and
26 # "function" elements
27 # (but the "referenced" list excludes self-references, i.e., the
28 # component itself and the "interface" and "function" elements
29 # already found in "specification")
30 
31 # A `package name' is a component or interface name prefixed with its path
32 # relative to a set of "top directories", and with path separators replaced
33 # by dots. The root of a TinyOS installation is usually the single top
34 # directory, but more can be specified with the -topdir option.

83 # simplify file names (for generating package names, so no need to
84 # preserve path validity):
85 # empty strings become ./
86 # windows paths are made unix-like:
87 # c: becomes /c/
88 # all \ become /
89 def canonicalise(name):
90  if name == "":
91  name = "."
92  if (name[1:2] == ":"): # windows disk names
93  name = "/%s/%s" %(name[0], name[2:])
94  name = replace(name, "\\", "/")
95  return name

97 # canonicalise a directory. like canonicalise, but ensures

99 def canonicalisedir(dirname):
100  dirname = canonicalise(dirname)
101  if dirname[-1] != "/":
102  return dirname + "/"

104  return dirname
