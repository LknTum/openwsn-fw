357 void
358 FlashCacheModeSet(uint32_t ui32CacheMode)
359 {
360  uint32_t ui32Busy;
361  uint32_t ui32TempValue;
362 
363  //
364  // Check the arguments.
365  //
366  ASSERT((ui32CacheMode == FLASH_CTRL_CACHE_MODE_DISABLE) ||
367  (ui32CacheMode == FLASH_CTRL_CACHE_MODE_ENABLE) ||
368  (ui32CacheMode == FLASH_CTRL_CACHE_MODE_PREFETCH_ENABLE) ||
369  (ui32CacheMode == FLASH_CTRL_CACHE_MODE_REALTIME));
370 
371  //
372  // Wait until FLASH is not busy.
373  //
374  ui32Busy = 1;
375  while(ui32Busy)
376  {
377  ui32TempValue = HWREG(FLASH_CTRL_FCTL);
378  ui32Busy = ui32TempValue & FLASH_CTRL_FCTL_BUSY;
379  }
380 
381  //
382  // Set desired cache mode.
383  //
384  ui32TempValue &= ~FLASH_CTRL_FCTL_CM_M;
385  HWREG(FLASH_CTRL_FCTL) = ui32TempValue | ui32CacheMode;
386 }
